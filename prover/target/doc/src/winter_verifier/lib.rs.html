6//! This crate contains Winterfell STARK verifier.
7//!
8//! This verifier can be used to verify STARK proofs generated by the Winterfell STARK prover.
9//!
10//! # Usage
11//! To verify a proof that a computation was executed correctly, you'll need to do the following:
12//!
13//! 1. Define an *algebraic intermediate representation* (AIR) for you computation. This can be
14//!    done by implementing [Air] trait.
15//! 2. Execute [verify()] function and supply the AIR of your computation together with the
16//!    [Proof] and related public inputs as parameters.
17//!
18//! # Performance
19//! Proof verification is extremely fast and is nearly independent of the complexity of the
20//! computation being verified. In vast majority of cases proofs can be verified in 3 - 5 ms
21//! on a modern mid-range laptop CPU (using a single core).
22//!
23//! There is one exception, however: if a computation requires a lot of `sequence` assertions
24//! (see [Assertion] for more info), the verification time will grow linearly in the number of
25//! asserted values. But for the impact to be noticeable, the number of asserted values would
26//! need to be in tens of thousands. And even for hundreds of thousands of asserted values, the
27//! verification time should not exceed 50 ms.

69/// Verifies that the specified computation was executed correctly against the specified inputs.
70///
71/// Specifically, for a computation specified by `AIR` and `HashFn` type parameter, verifies that
72/// the provided `proof` attests to the correct execution of the computation against public inputs
73/// specified by `pub_inputs`. If the verification is successful, `Ok(())` is returned.
74///
75/// # Errors
76/// Returns an error if combination of the provided proof and public inputs does not attest to
77/// a correct execution of the computation. This could happen for many various reasons, including:
78/// - The specified proof was generated for a different computation.
79/// - The specified proof was generated for this computation but for different public inputs.
80/// - The specified proof was generated with parameters not providing an acceptable security level.
81pub fn verify<AIR, HashFn, RandCoin, VC>(
82    proof: Proof,
83    pub_inputs: AIR::PublicInputs,
84    acceptable_options: &AcceptableOptions,
85) -> Result<(), VerifierError>
86where
87    AIR: Air,
88    HashFn: ElementHasher<BaseField = AIR::BaseField>,
89    RandCoin: RandomCoin<BaseField = AIR::BaseField, Hasher = HashFn>,
90    VC: VectorCommitment<HashFn>,
91{
92    // check that `proof` was generated with an acceptable set of parameters from the point of view
93    // of the verifier
94    acceptable_options.validate::<HashFn>(&proof)?;
95
96    // build a seed for the public coin; the initial seed is a hash of the proof context and the
97    // public inputs, but as the protocol progresses, the coin will be reseeded with the info
98    // received from the prover
99    let mut public_coin_seed = proof.context.to_elements();
100    public_coin_seed.append(&mut pub_inputs.to_elements());
101
102    // create AIR instance for the computation specified in the proof
103    let air = AIR::new(proof.trace_info().clone(), pub_inputs, proof.options().clone());
104
105    // figure out which version of the generic proof verification procedure to run. this is a sort
106    // of static dispatch for selecting two generic parameter: extension field and hash function.
107    match air.options().field_extension() {
108        FieldExtension::None => {
109            let public_coin = RandCoin::new(&public_coin_seed);
110            let channel = VerifierChannel::new(&air, proof)?;
111            perform_verification::<AIR, AIR::BaseField, HashFn, RandCoin, VC>(
112                air,
113                channel,
114                public_coin,
115            )
116        },
117        FieldExtension::Quadratic => {
118            if !<QuadExtension<AIR::BaseField>>::is_supported() {
119                return Err(VerifierError::UnsupportedFieldExtension(2));
120            }
121            let public_coin = RandCoin::new(&public_coin_seed);
122            let channel = VerifierChannel::new(&air, proof)?;
123            perform_verification::<AIR, QuadExtension<AIR::BaseField>, HashFn, RandCoin, VC>(
124                air,
125                channel,
126                public_coin,
127            )
128        },
129        FieldExtension::Cubic => {
130            if !<CubeExtension<AIR::BaseField>>::is_supported() {
131                return Err(VerifierError::UnsupportedFieldExtension(3));
132            }
133            let public_coin = RandCoin::new(&public_coin_seed);
134            let channel = VerifierChannel::new(&air, proof)?;
135            perform_verification::<AIR, CubeExtension<AIR::BaseField>, HashFn, RandCoin, VC>(
136                air,
137                channel,
138                public_coin,
139            )
140        },
141    }
142}

144// VERIFICATION PROCEDURE
145// ================================================================================================
146/// Performs the actual verification by reading the data from the `channel` and making sure it
147/// attests to a correct execution of the computation specified by the provided `air`.
148fn perform_verification<A, E, H, R, V>(
149    air: A,
150    mut channel: VerifierChannel<E, H, V>,
151    mut public_coin: R,
152) -> Result<(), VerifierError>
153where
154    E: FieldElement<BaseField = A::BaseField>,
155    A: Air,
156    H: ElementHasher<BaseField = A::BaseField>,
157    R: RandomCoin<BaseField = A::BaseField, Hasher = H>,
158    V: VectorCommitment<H>,
159{
160    // 1 ----- trace commitment -------------------------------------------------------------------
161    // Read the commitments to evaluations of the trace polynomials over the LDE domain sent by the
162    // prover. The commitments are used to update the public coin, and draw sets of random elements
163    // from the coin (in the interactive version of the protocol the verifier sends these random
164    // elements to the prover after each commitment is made). When there are multiple trace
165    // commitments (i.e., the trace consists of more than one segment), each previous commitment is
166    // used to draw random elements needed to construct the next trace segment. The last trace
167    // commitment is used to draw a set of random coefficients which the prover uses to compute
168    // constraint composition polynomial.
169    const MAIN_TRACE_IDX: usize = 0;
170    const AUX_TRACE_IDX: usize = 1;
171    let trace_commitments = channel.read_trace_commitments();
172
173    // reseed the coin with the commitment to the main trace segment
174    public_coin.reseed(trace_commitments[MAIN_TRACE_IDX]);
175
176    // process auxiliary trace segments (if any), to build a set of random elements for each segment
177    let aux_trace_rand_elements = if air.trace_info().is_multi_segment() {
178        if air.context().has_lagrange_kernel_aux_column() {
179            let gkr_proof = {
180                let gkr_proof_serialized = channel
181                    .read_gkr_proof()
182                    .expect("Expected an a GKR proof because trace has lagrange kernel column");
183
184                Deserializable::read_from_bytes(gkr_proof_serialized)
185                    .map_err(|err| VerifierError::ProofDeserializationError(err.to_string()))?
186            };
187            let gkr_rand_elements = air
188                .get_gkr_proof_verifier::<E>()
189                .verify::<E, _>(gkr_proof, &mut public_coin)
190                .map_err(|err| VerifierError::GkrProofVerificationFailed(err.to_string()))?;
191
192            let rand_elements = air.get_aux_rand_elements(&mut public_coin).expect(
193                "failed to generate the random elements needed to build the auxiliary trace",
194            );
195
196            public_coin.reseed(trace_commitments[AUX_TRACE_IDX]);
197
198            Some(AuxRandElements::new_with_gkr(rand_elements, gkr_rand_elements))
199        } else {
200            let rand_elements = air.get_aux_rand_elements(&mut public_coin).expect(
201                "failed to generate the random elements needed to build the auxiliary trace",
202            );
203
204            public_coin.reseed(trace_commitments[AUX_TRACE_IDX]);
205
206            Some(AuxRandElements::new(rand_elements))
207        }
208    } else {
209        None
210    };
211
212    // build random coefficients for the composition polynomial
213    let constraint_coeffs = air
214        .get_constraint_composition_coefficients(&mut public_coin)
215        .map_err(|_| VerifierError::RandomCoinError)?;
216
217    // 2 ----- constraint commitment --------------------------------------------------------------
218    // read the commitment to evaluations of the constraint composition polynomial over the LDE
219    // domain sent by the prover, use it to update the public coin, and draw an out-of-domain point
220    // z from the coin; in the interactive version of the protocol, the verifier sends this point z
221    // to the prover, and the prover evaluates trace and constraint composition polynomials at z,
222    // and sends the results back to the verifier.
223    let constraint_commitment = channel.read_constraint_commitment();
224    public_coin.reseed(constraint_commitment);
225    let z = public_coin.draw::<E>().map_err(|_| VerifierError::RandomCoinError)?;
226
227    // 3 ----- OOD consistency check --------------------------------------------------------------
228    // make sure that evaluations obtained by evaluating constraints over the out-of-domain frame
229    // are consistent with the evaluations of composition polynomial columns sent by the prover
230
231    // read the out-of-domain trace frames (the main trace frame and auxiliary trace frame, if
232    // provided) sent by the prover and evaluate constraints over them; also, reseed the public
233    // coin with the OOD frames received from the prover.
234    let ood_trace_frame = channel.read_ood_trace_frame();
235    let ood_main_trace_frame = ood_trace_frame.main_frame();
236    let ood_aux_trace_frame = ood_trace_frame.aux_frame();
237    let ood_lagrange_kernel_frame = ood_trace_frame.lagrange_kernel_frame();
238    let ood_constraint_evaluation_1 = evaluate_constraints(
239        &air,
240        constraint_coeffs,
241        &ood_main_trace_frame,
242        &ood_aux_trace_frame,
243        ood_lagrange_kernel_frame,
244        aux_trace_rand_elements.as_ref(),
245        z,
246    );
247    public_coin.reseed(ood_trace_frame.hash::<H>());
248
249    // read evaluations of composition polynomial columns sent by the prover, and reduce them into
250    // a single value by computing \sum_{i=0}^{m-1}(z^(i * l) * value_i), where value_i is the
251    // evaluation of the ith column polynomial H_i(X) at z, l is the trace length and m is
252    // the number of composition column polynomials. This computes H(z) (i.e.
253    // the evaluation of the composition polynomial at z) using the fact that
254    // H(X) = \sum_{i=0}^{m-1} X^{i * l} H_i(X).
255    // Also, reseed the public coin with the OOD constraint evaluations received from the prover.
256    let ood_constraint_evaluations = channel.read_ood_constraint_evaluations();
257    let ood_constraint_evaluation_2 =
258        ood_constraint_evaluations
259            .iter()
260            .enumerate()
261            .fold(E::ZERO, |result, (i, &value)| {
262                result + z.exp_vartime(((i * (air.trace_length())) as u32).into()) * value
263            });
264    public_coin.reseed(H::hash_elements(&ood_constraint_evaluations));
265
266    // finally, make sure the values are the same
267    if ood_constraint_evaluation_1 != ood_constraint_evaluation_2 {
268        return Err(VerifierError::InconsistentOodConstraintEvaluations);
269    }
270
271    // 4 ----- FRI commitments --------------------------------------------------------------------
272    // draw coefficients for computing DEEP composition polynomial from the public coin; in the
273    // interactive version of the protocol, the verifier sends these coefficients to the prover
274    // and the prover uses them to compute the DEEP composition polynomial. the prover, then
275    // applies FRI protocol to the evaluations of the DEEP composition polynomial.
276    let deep_coefficients = air
277        .get_deep_composition_coefficients::<E, R>(&mut public_coin)
278        .map_err(|_| VerifierError::RandomCoinError)?;
279
280    // instantiates a FRI verifier with the FRI layer commitments read from the channel. From the
281    // verifier's perspective, this is equivalent to executing the commit phase of the FRI protocol.
282    // The verifier uses these commitments to update the public coin and draw random points alpha
283    // from them; in the interactive version of the protocol, the verifier sends these alphas to
284    // the prover, and the prover uses them to compute and commit to the subsequent FRI layers.
285    let fri_verifier = FriVerifier::new(
286        &mut channel,
287        &mut public_coin,
288        air.options().to_fri_options(),
289        air.trace_poly_degree(),
290    )
291    .map_err(VerifierError::FriVerificationFailed)?;
292    // TODO: make sure air.lde_domain_size() == fri_verifier.domain_size()
293
294    // 5 ----- trace and constraint queries -------------------------------------------------------
295    // read proof-of-work nonce sent by the prover
296    let pow_nonce = channel.read_pow_nonce();
297
298    // make sure the proof-of-work specified by the grinding factor is satisfied
299    if public_coin.check_leading_zeros(pow_nonce) < air.options().grinding_factor() {
300        return Err(VerifierError::QuerySeedProofOfWorkVerificationFailed);
301    }
302
303    // draw pseudo-random query positions for the LDE domain from the public coin; in the
304    // interactive version of the protocol, the verifier sends these query positions to the prover,
305    // and the prover responds with decommitments against these positions for trace and constraint
306    // composition polynomial evaluations.
307    let mut query_positions = public_coin
308        .draw_integers(air.options().num_queries(), air.lde_domain_size(), pow_nonce)
309        .map_err(|_| VerifierError::RandomCoinError)?;
310
311    // remove any potential duplicates from the positions as the prover will send openings only
312    // for unique queries
313    query_positions.sort_unstable();
314    query_positions.dedup();
315
316    // read evaluations of trace and constraint composition polynomials at the queried positions;
317    // this also checks that the read values are valid against trace and constraint commitments
318    let (queried_main_trace_states, queried_aux_trace_states) =
319        channel.read_queried_trace_states(&query_positions)?;
320    let queried_constraint_evaluations = channel.read_constraint_evaluations(&query_positions)?;
321
322    // 6 ----- DEEP composition -------------------------------------------------------------------
323    // compute evaluations of the DEEP composition polynomial at the queried positions
324    let composer = DeepComposer::new(&air, &query_positions, z, deep_coefficients);
325    let t_composition = composer.compose_trace_columns(
326        queried_main_trace_states,
327        queried_aux_trace_states,
328        ood_main_trace_frame,
329        ood_aux_trace_frame,
330        ood_lagrange_kernel_frame,
331    );
332    let c_composition = composer
333        .compose_constraint_evaluations(queried_constraint_evaluations, ood_constraint_evaluations);
334    let deep_evaluations = composer.combine_compositions(t_composition, c_composition);
335
336    // 7 ----- Verify low-degree proof -------------------------------------------------------------
337    // make sure that evaluations of the DEEP composition polynomial we computed in the previous
338    // step are in fact evaluations of a polynomial of degree equal to trace polynomial degree
339    fri_verifier
340        .verify(&mut channel, &deep_evaluations, &query_positions)
341        .map_err(VerifierError::FriVerificationFailed)
342}

344// ACCEPTABLE OPTIONS
345// ================================================================================================
346// Specifies either the minimal, conjectured or proven, security level or a set of
347// `ProofOptions` that are acceptable by the verification procedure.
348pub enum AcceptableOptions {
349    /// Minimal acceptable conjectured security level
350    MinConjecturedSecurity(u32),
351    /// Minimal acceptable proven security level
352    MinProvenSecurity(u32),
353    /// Set of acceptable proof parameters
354    OptionSet(Vec<ProofOptions>),
355}
356
357impl AcceptableOptions {
358    /// Checks that a proof was generated using an acceptable set of parameters.
359    pub fn validate<H: Hasher>(&self, proof: &Proof) -> Result<(), VerifierError> {
360        match self {
361            AcceptableOptions::MinConjecturedSecurity(minimal_security) => {
362                let proof_security = proof.security_level::<H>(true);
363                if proof_security < *minimal_security {
364                    return Err(VerifierError::InsufficientConjecturedSecurity(
365                        *minimal_security,
366                        proof_security,
367                    ));
368                }
369            },
370            AcceptableOptions::MinProvenSecurity(minimal_security) => {
371                let proof_security = proof.security_level::<H>(false);
372                if proof_security < *minimal_security {
373                    return Err(VerifierError::InsufficientProvenSecurity(
374                        *minimal_security,
375                        proof_security,
376                    ));
377                }
378            },
379            AcceptableOptions::OptionSet(options) => {
380                if !options.iter().any(|opt| opt == proof.options()) {
381                    return Err(VerifierError::UnacceptableProofOptions);
382                }
383            },
384        }
385        Ok(())
386    }
