19// VERIFIER CHANNEL
20// ================================================================================================
21
22/// A view into a [Proof] for a computation structured to simulate an "interactive" channel.
23///
24/// A channel is instantiated for a specific proof, which is parsed into structs over the
25/// appropriate field (specified by type parameter `E`). This also validates that the proof is
26/// well-formed in the context of the computation for the specified [Air].
27pub struct VerifierChannel<
28    E: FieldElement,
29    H: ElementHasher<BaseField = E::BaseField>,
30    V: VectorCommitment<H>,
31> {
32    // trace queries
33    trace_commitments: Vec<H::Digest>,
34    trace_queries: Option<TraceQueries<E, H, V>>,
35    // constraint queries
36    constraint_commitment: H::Digest,
37    constraint_queries: Option<ConstraintQueries<E, H, V>>,
38    // partition sizes for the rows of main, auxiliary and constraint traces rows
39    partition_size_main: usize,
40    partition_size_aux: usize,
41    partition_size_constraint: usize,
42    // FRI proof
43    fri_commitments: Option<Vec<H::Digest>>,
44    fri_layer_proofs: Vec<V::MultiProof>,
45    fri_layer_queries: Vec<Vec<E>>,
46    fri_remainder: Option<Vec<E>>,
47    fri_num_partitions: usize,
48    // out-of-domain frame
49    ood_trace_frame: Option<TraceOodFrame<E>>,
50    ood_constraint_evaluations: Option<Vec<E>>,
51    // query proof-of-work
52    pow_nonce: u64,
53    gkr_proof: Option<Vec<u8>>,
54}
55
56impl<E, H, V> VerifierChannel<E, H, V>
57where
58    E: FieldElement,
59    H: ElementHasher<BaseField = E::BaseField>,
60    V: VectorCommitment<H>,
61{
62    // CONSTRUCTOR
63    // --------------------------------------------------------------------------------------------
64    /// Creates and returns a new [VerifierChannel] initialized from the specified `proof`.
65    pub fn new<A: Air<BaseField = E::BaseField>>(
66        air: &A,
67        proof: Proof,
68    ) -> Result<Self, VerifierError> {
69        let Proof {
70            context,
71            num_unique_queries,
72            commitments,
73            trace_queries,
74            constraint_queries,
75            ood_frame,
76            fri_proof,
77            pow_nonce,
78            gkr_proof,
79        } = proof;
80
81        // make sure AIR and proof base fields are the same
82        if E::BaseField::get_modulus_le_bytes() != context.field_modulus_bytes() {
83            return Err(VerifierError::InconsistentBaseField);
84        }
85        let constraint_frame_width = air.context().num_constraint_composition_columns();
86
87        let num_trace_segments = air.trace_info().num_segments();
88        let main_trace_width = air.trace_info().main_trace_width();
89        let aux_trace_width = air.trace_info().aux_segment_width();
90        let lde_domain_size = air.lde_domain_size();
91        let fri_options = air.options().to_fri_options();
92        let partition_options = air.options().partition_options();
93
94        // --- parse commitments ------------------------------------------------------------------
95        let (trace_commitments, constraint_commitment, fri_commitments) = commitments
96            .parse::<H>(num_trace_segments, fri_options.num_fri_layers(lde_domain_size))
97            .map_err(|err| VerifierError::ProofDeserializationError(err.to_string()))?;
98
99        // --- parse trace and constraint queries -------------------------------------------------
100        let trace_queries =
101            TraceQueries::<E, H, V>::new(trace_queries, air, num_unique_queries as usize)?;
102        let constraint_queries = ConstraintQueries::<E, H, V>::new(
103            constraint_queries,
104            air,
105            num_unique_queries as usize,
106        )?;
107
108        // --- parse FRI proofs -------------------------------------------------------------------
109        let fri_num_partitions = fri_proof.num_partitions();
110        let fri_remainder = fri_proof
111            .parse_remainder()
112            .map_err(|err| VerifierError::ProofDeserializationError(err.to_string()))?;
113        let (fri_layer_queries, fri_layer_proofs) = fri_proof
114            .parse_layers::<E, H, V>(lde_domain_size, fri_options.folding_factor())
115            .map_err(|err| VerifierError::ProofDeserializationError(err.to_string()))?;
116
117        // --- parse out-of-domain evaluation frame -----------------------------------------------
118        let (ood_trace_frame, ood_constraint_evaluations) = ood_frame
119            .parse(main_trace_width, aux_trace_width, constraint_frame_width)
120            .map_err(|err| VerifierError::ProofDeserializationError(err.to_string()))?;
121
122        // --- compute the partition size for each trace ------------------------------------------
123        let partition_size_main = partition_options
124            .partition_size::<E::BaseField>(air.context().trace_info().main_trace_width());
125        let partition_size_aux =
126            partition_options.partition_size::<E>(air.context().trace_info().aux_segment_width());
127        let partition_size_constraint = partition_options
128            .partition_size::<E>(air.context().num_constraint_composition_columns());
129
130        Ok(VerifierChannel {
131            // trace queries
132            trace_commitments,
133            trace_queries: Some(trace_queries),
134            // constraint queries
135            constraint_commitment,
136            constraint_queries: Some(constraint_queries),
137            // num partitions used in commitment
138            partition_size_main,
139            partition_size_aux,
140            partition_size_constraint,
141            // FRI proof
142            fri_commitments: Some(fri_commitments),
143            fri_layer_proofs,
144            fri_layer_queries,
145            fri_remainder: Some(fri_remainder),
146            fri_num_partitions,
147            // out-of-domain evaluation
148            ood_trace_frame: Some(ood_trace_frame),
149            ood_constraint_evaluations: Some(ood_constraint_evaluations),
150            // query seed
151            pow_nonce,
152            gkr_proof,
153        })
154    }
155
156    // DATA READERS
157    // --------------------------------------------------------------------------------------------
158
159    /// Returns execution trace commitments sent by the prover.
160    ///
161    /// For computations requiring multiple trace segment, the returned slice will contain a
162    /// commitment for each trace segment.
163    pub fn read_trace_commitments(&self) -> &[H::Digest] {
164        &self.trace_commitments
165    }
166
167    /// Returns constraint evaluation commitment sent by the prover.
168    pub fn read_constraint_commitment(&self) -> H::Digest {
169        self.constraint_commitment
170    }
171
172    /// Returns trace polynomial evaluations at out-of-domain points z and z * g, where g is the
173    /// generator of the LDE domain.
174    ///
175    /// For computations requiring multiple trace segments, evaluations of auxiliary trace
176    /// polynomials are also included.
177    pub fn read_ood_trace_frame(&mut self) -> TraceOodFrame<E> {
178        self.ood_trace_frame.take().expect("already read")
179    }
180
181    /// Returns evaluations of composition polynomial columns at z^m, where z is the out-of-domain
182    /// point, and m is the number of composition polynomial columns.
183    pub fn read_ood_constraint_evaluations(&mut self) -> Vec<E> {
184        self.ood_constraint_evaluations.take().expect("already read")
185    }
186
187    /// Returns query proof-of-work nonce sent by the prover.
188    pub fn read_pow_nonce(&self) -> u64 {
189        self.pow_nonce
190    }
191
192    /// Returns the serialized GKR proof, if any.
193    pub fn read_gkr_proof(&self) -> Option<&Vec<u8>> {
194        self.gkr_proof.as_ref()
195    }
196
197    /// Returns trace states at the specified positions of the LDE domain. This also checks if
198    /// the trace states are valid against the trace commitment sent by the prover.
199    ///
200    /// For computations requiring multiple trace segments, trace states for auxiliary segments
201    /// are also included as the second value of the returned tuple (trace states for all auxiliary
202    /// segments are merged into a single table). Otherwise, the second value is None.
203    #[allow(clippy::type_complexity)]
204    pub fn read_queried_trace_states(
205        &mut self,
206        positions: &[usize],
207    ) -> Result<(Table<E::BaseField>, Option<Table<E>>), VerifierError> {
208        let queries = self.trace_queries.take().expect("already read");
209
210        // make sure the states included in the proof correspond to the trace commitment
211        let items: Vec<H::Digest> = queries
212            .main_states
213            .rows()
214            .map(|row| hash_row::<H, E::BaseField>(row, self.partition_size_main))
215            .collect();
216
217        <V as VectorCommitment<H>>::verify_many(
218            self.trace_commitments[0],
219            positions,
220            &items,
221            &queries.query_proofs[0],
222        )
223        .map_err(|_| VerifierError::TraceQueryDoesNotMatchCommitment)?;
224
225        if let Some(ref aux_states) = queries.aux_states {
226            let items: Vec<H::Digest> = aux_states
227                .rows()
228                .map(|row| hash_row::<H, E>(row, self.partition_size_aux))
229                .collect();
230
231            <V as VectorCommitment<H>>::verify_many(
232                self.trace_commitments[1],
233                positions,
234                &items,
235                &queries.query_proofs[1],
236            )
237            .map_err(|_| VerifierError::TraceQueryDoesNotMatchCommitment)?;
238        }
239
240        Ok((queries.main_states, queries.aux_states))
241    }
242
243    /// Returns constraint evaluations at the specified positions of the LDE domain. This also
244    /// checks if the constraint evaluations are valid against the constraint commitment sent by
245    /// the prover.
246    pub fn read_constraint_evaluations(
247        &mut self,
248        positions: &[usize],
249    ) -> Result<Table<E>, VerifierError> {
250        let queries = self.constraint_queries.take().expect("already read");
251
252        let items: Vec<H::Digest> = queries
253            .evaluations
254            .rows()
255            .map(|row| hash_row::<H, E>(row, self.partition_size_constraint))
256            .collect();
257
258        <V as VectorCommitment<H>>::verify_many(
259            self.constraint_commitment,
260            positions,
261            &items,
262            &queries.query_proofs,
263        )
264        .map_err(|_| VerifierError::ConstraintQueryDoesNotMatchCommitment)?;
265
266        Ok(queries.evaluations)
267    }
268}

303// TRACE QUERIES

305
306/// Container of trace query data, including:
307/// * Queried states for all trace segments.
308/// * Batch opening proof for all queries.
309///
310/// Trace states for all auxiliary segments are stored in a single table.
311struct TraceQueries<
312    E: FieldElement,
313    H: ElementHasher<BaseField = E::BaseField>,
314    V: VectorCommitment<H>,
315> {
316    query_proofs: Vec<V::MultiProof>,
317    main_states: Table<E::BaseField>,
318    aux_states: Option<Table<E>>,
319    _h: PhantomData<H>,
320}
321
322impl<E, H, V> TraceQueries<E, H, V>
323where
324    E: FieldElement,
325    H: ElementHasher<BaseField = E::BaseField>,
326    V: VectorCommitment<H>,
327{
328    /// Parses the provided trace queries into trace states in the specified field and
329    /// corresponding batch opening proof.
330    pub fn new<A: Air<BaseField = E::BaseField>>(
331        mut queries: Vec<Queries>,
332        air: &A,
333        num_queries: usize,
334    ) -> Result<Self, VerifierError> {
335        assert_eq!(
336            queries.len(),
337            air.trace_info().num_segments(),
338            "expected {} trace segment queries, but received {}",
339            air.trace_info().num_segments(),
340            queries.len()
341        );
342
343        // parse main trace segment queries
344        let main_segment_width = air.trace_info().main_trace_width();
345        let main_segment_queries = queries.remove(0);
346        let (main_segment_query_proofs, main_segment_states) = main_segment_queries
347            .parse::<E::BaseField, H, V>(air.lde_domain_size(), num_queries, main_segment_width)
348            .map_err(|err| {
349                VerifierError::ProofDeserializationError(format!(
350                    "main trace segment query deserialization failed: {err}"
351                ))
352            })?;
353
354        // all query proofs will be aggregated into a single vector
355        let mut query_proofs = vec![main_segment_query_proofs];
356
357        // parse auxiliary trace segment queries (if any), and merge resulting tables into a
358        // single table
359        let aux_trace_states = if air.trace_info().is_multi_segment() {
360            let mut aux_trace_states = Vec::new();
361            let segment_queries = queries.remove(0);
362            let segment_width = air.trace_info().get_aux_segment_width();
363            let (segment_query_proof, segment_trace_states) = segment_queries
364                .parse::<E, H, V>(air.lde_domain_size(), num_queries, segment_width)
365                .map_err(|err| {
366                    VerifierError::ProofDeserializationError(format!(
367                        "auxiliary trace segment query deserialization failed: {err}"
368                    ))
369                })?;
370
371            query_proofs.push(segment_query_proof);
372            aux_trace_states.push(segment_trace_states);
373
374            // merge tables for each auxiliary segment into a single table
375            Some(Table::merge(aux_trace_states))
376        } else {
377            None
378        };
379
380        Ok(Self {
381            query_proofs,
382            main_states: main_segment_states,
383            aux_states: aux_trace_states,
384            _h: PhantomData,
385        })
386    }
387}

389// CONSTRAINT QUERIES
390// ================================================================================================
391
392/// Container of constraint evaluation query data, including:
393/// * Queried constraint evaluation values.
394/// * Batch opening proof for all queries.
395struct ConstraintQueries<
396    E: FieldElement,
397    H: ElementHasher<BaseField = E::BaseField>,
398    V: VectorCommitment<H>,
399> {
400    query_proofs: V::MultiProof,
401    evaluations: Table<E>,
402    _h: PhantomData<H>,
403}
404
405impl<E, H, V> ConstraintQueries<E, H, V>
406where
407    E: FieldElement,
408    H: ElementHasher<BaseField = E::BaseField>,
409    V: VectorCommitment<H>,
410{
411    /// Parses the provided constraint queries into evaluations in the specified field and
412    /// corresponding batch opening proof.
413    pub fn new<A: Air<BaseField = E::BaseField>>(
414        queries: Queries,
415        air: &A,
416        num_queries: usize,
417    ) -> Result<Self, VerifierError> {
418        let constraint_frame_width = air.context().num_constraint_composition_columns();
419
420        let (query_proofs, evaluations) = queries
421            .parse::<E, H, V>(air.lde_domain_size(), num_queries, constraint_frame_width)
422            .map_err(|err| {
423                VerifierError::ProofDeserializationError(format!(
424                    "constraint evaluation query deserialization failed: {err}"
425                ))
426            })?;
427
428        Ok(Self {
429            query_proofs,
430            evaluations,
431            _h: PhantomData,
432        })
433    }
434}

439/// Hashes a row of a trace in batches where each batch is of size at most `partition_size`.
440fn hash_row<H, E>(row: &[E], partition_size: usize) -> H::Digest
441where
442    E: FieldElement,
443    H: ElementHasher<BaseField = E::BaseField>,
444{
445    if partition_size == row.len() {
446        H::hash_elements(row)
447    } else {
448        let num_partitions = row.len().div_ceil(partition_size);
449
450        let mut buffer = vec![H::Digest::default(); num_partitions];
451
452        row.chunks(partition_size)
453            .zip(buffer.iter_mut())
454            .for_each(|(chunk, buf)| *buf = H::hash_elements(chunk));
455        H::merge_many(&buffer)
456    }
