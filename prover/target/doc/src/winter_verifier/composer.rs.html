6use alloc::vec::Vec;

8use air::{

13// DEEP COMPOSER
14// ================================================================================================
15
16pub struct DeepComposer<E: FieldElement> {
17    cc: DeepCompositionCoefficients<E>,
18    x_coordinates: Vec<E>,
19    z: [E; 2],
20    g_trace: E::BaseField,
21    lagrange_kernel_column_idx: Option<usize>,
22}
23
24impl<E: FieldElement> DeepComposer<E> {
25    /// Creates a new composer for computing DEEP composition polynomial values.
26    pub fn new<A: Air<BaseField = E::BaseField>>(
27        air: &A,
28        query_positions: &[usize],
29        z: E,
30        cc: DeepCompositionCoefficients<E>,
31    ) -> Self {
32        // compute LDE domain coordinates for all query positions
33        let g_lde = air.lde_domain_generator();
34        let domain_offset = air.domain_offset();
35        let x_coordinates: Vec<E> = query_positions
36            .iter()
37            .map(|&p| E::from(g_lde.exp_vartime((p as u64).into()) * domain_offset))
38            .collect();
39        let g_trace = air.trace_domain_generator();
40
41        DeepComposer {
42            cc,
43            x_coordinates,
44            z: [z, z * E::from(g_trace)],
45            g_trace,
46            lagrange_kernel_column_idx: air.context().lagrange_kernel_aux_column_idx(),
47        }
48    }
49
50    /// For each queried trace state, combines column values into a single value by computing
51    /// their random linear combinations as follows:
52    ///
53    /// - Assume each column value is an evaluation of a trace polynomial T_i(x).
54    /// - For each T_i(x) compute T'_i(x) = (T_i(x) - T_i(z)) / (x - z) and
55    ///   T''_i = (T_i(x) - T_i(z * g)) / (x - z * g), where z is the out-of-domain point and
56    ///   g is the the LDE domain generator.
57    /// - Then, combine all T'_i(x) and T''_i(x) values together by computing
58    ///   T(x) = sum((T'_i(x) + T''_i(x)) * cc_i) for all i, where cc_i is the coefficient for
59    ///   for the random linear combination drawn from the public coin.
60    /// - If a Lagrange kernel is present, combine one additional term defined as
61    ///   (T_l(x) - p_S(x)) / Z_S(x), where:
62    ///
63    /// 1. $T_l(x) is the evaluation of the Lagrange trace polynomial at $x$.
64    /// 2. $S$ is the set of opening points for the Lagrange kernel i.e.,
65    ///    $S := {z, z.g, z.g^2, ..., z.g^{2^{log_2(\nu) - 1}}}$.
66    /// 3. $p_S(X)$ is the polynomial of minimal degree interpolating the set
67    ///    ${(a, T_l(a)): a \in S}$.
68    /// 4. $Z_S(X)$ is the polynomial of minimal degree vanishing over the set $S$.
69    ///
70    /// Note that values of T_i(z) and T_i(z * g) are received from the prover and passed into
71    /// this function via the `ood_main_frame` and `ood_aux_frame` parameters.
72    ///
73    /// If a Lagrange kernel is present, the evaluations of $T_l$ over the set $S$ are received
74    /// from the prover and passed separately via `ood_lagrange_kernel_frame`.
75    pub fn compose_trace_columns(
76        &self,
77        queried_main_trace_states: Table<E::BaseField>,
78        queried_aux_trace_states: Option<Table<E>>,
79        ood_main_frame: EvaluationFrame<E>,
80        ood_aux_frame: Option<EvaluationFrame<E>>,
81        ood_lagrange_kernel_frame: Option<&LagrangeKernelEvaluationFrame<E>>,
82    ) -> Vec<E> {
83        let ood_main_trace_states = [ood_main_frame.current(), ood_main_frame.next()];
84
85        // compose columns of of the main trace segment; we do this separately for numerators of
86        // each query; we also track common denominator for each query separately; this way we can
87        // use a batch inversion in the end.
88        let n = queried_main_trace_states.num_rows();
89        let mut result_num = Vec::<E>::with_capacity(n);
90        let mut result_den = Vec::<E>::with_capacity(n);
91
92        for ((_, row), &x) in (0..n).zip(queried_main_trace_states.rows()).zip(&self.x_coordinates)
93        {
94            let mut t1_num = E::ZERO;
95            let mut t2_num = E::ZERO;
96
97            for (i, &value) in row.iter().enumerate() {
98                let value = E::from(value);
99                // compute the numerator of T'_i(x) as (T_i(x) - T_i(z)), multiply it by a
100                // composition coefficient, and add the result to the numerator aggregator
101                t1_num += (value - ood_main_trace_states[0][i]) * self.cc.trace[i];
102
103                // compute the numerator of T''_i(x) as (T_i(x) - T_i(z * g)), multiply it by a
104                // composition coefficient, and add the result to the numerator aggregator
105                t2_num += (value - ood_main_trace_states[1][i]) * self.cc.trace[i];
106            }
107            // compute the common denominator as (x - z) * (x - z * g)
108            let t1_den = x - self.z[0];
109            let t2_den = x - self.z[1];
110            result_den.push(t1_den * t2_den);
111
112            // add the numerators of T'_i(x) and T''_i(x) together; we can do this because later on
113            // we'll use the common denominator computed above.
114            result_num.push(t1_num * t2_den + t2_num * t1_den);
115        }
116
117        // if the trace has auxiliary segments, compose columns from these segments as well; we
118        // also do this separately for numerators and denominators.
119        if let Some(queried_aux_trace_states) = queried_aux_trace_states {
120            let ood_aux_frame = ood_aux_frame.expect("missing auxiliary OOD frame");
121            let ood_aux_trace_states = [ood_aux_frame.current(), ood_aux_frame.next()];
122
123            // we define this offset here because composition of the main trace columns has
124            // consumed some number of composition coefficients already.
125            let cc_offset = queried_main_trace_states.num_columns();
126
127            // we treat the Lagrange column separately if present
128            let lagrange_ker_col_idx =
129                self.lagrange_kernel_column_idx.unwrap_or(ood_aux_trace_states[0].len());
130
131            for ((j, row), &x) in
132                (0..n).zip(queried_aux_trace_states.rows()).zip(&self.x_coordinates)
133            {
134                let mut t1_num = E::ZERO;
135                let mut t2_num = E::ZERO;
136
137                let row = &row[..lagrange_ker_col_idx];
138                for (i, &value) in row.iter().enumerate() {
139                    // compute the numerator of T'_i(x) as (T_i(x) - T_i(z)), multiply it by a
140                    // composition coefficient, and add the result to the numerator aggregator
141                    t1_num += (value - ood_aux_trace_states[0][i]) * self.cc.trace[cc_offset + i];
142
143                    // compute the numerator of T''_i(x) as (T_i(x) - T_i(z * g)), multiply it by a
144                    // composition coefficient, and add the result to the numerator aggregator
145                    t2_num += (value - ood_aux_trace_states[1][i]) * self.cc.trace[cc_offset + i];
146                }
147
148                // compute the common denominators (x - z) and (x - z * g), and use the to aggregate
149                // numerators into the common numerator computed for the main trace of this query
150                let t1_den = x - self.z[0];
151                let t2_den = x - self.z[1];
152                result_num[j] += t1_num * t2_den + t2_num * t1_den;
153            }
154
155            // if a Lagrange kernel trace polynomial is present, we include its associated term
156            // separately. Note that, for performance reasons, we divide by Z_{S^{'}} instead of
157            // Z_S, where  $S^{'} := {z.g^2, ..., z.g^{2^{log_2(\nu) - 1}}}$. This is done as
158            // the final linear combination is divided by `(x - z) . (x - z.g)`.
159            if let Some(ood_lagrange_kernel_frame) = ood_lagrange_kernel_frame {
160                let mut result_lag_num = Vec::<E>::with_capacity(n);
161                let mut result_lag_den = Vec::<E>::with_capacity(n);
162
163                let log_trace_len_plus_1 = ood_lagrange_kernel_frame.num_rows();
164                let mut xs = Vec::with_capacity(log_trace_len_plus_1);
165                let mut ys = Vec::with_capacity(log_trace_len_plus_1);
166
167                xs.push(self.z[0]);
168                ys.push(ood_lagrange_kernel_frame.inner()[0]);
169
170                let mut g_exp = self.g_trace;
171                for i in 0..(log_trace_len_plus_1 - 1) {
172                    let x = self.z[0].mul_base(g_exp);
173                    let lagrange_poly_at_x = ood_lagrange_kernel_frame.inner()[i + 1];
174
175                    xs.push(x);
176                    ys.push(lagrange_poly_at_x);
177
178                    // takes on the values `g`, `g^2`, `g^4`, `g^8`, ...
179                    g_exp *= g_exp;
180                }
181                let p_s = polynom::interpolate(&xs, &ys, true);
182                let z_s_prime = polynom::poly_from_roots(&xs[2..]);
183
184                for (row, &x) in queried_aux_trace_states.rows().zip(&self.x_coordinates) {
185                    let value = row[lagrange_ker_col_idx];
186                    let cc = self.cc.lagrange.unwrap();
187
188                    result_lag_num.push((value - polynom::eval(&p_s, x)) * cc);
189                    result_lag_den.push(polynom::eval(&z_s_prime, x));
190                }
191
192                result_lag_den = batch_inversion(&result_lag_den);
193                for (j, res) in result_num.iter_mut().enumerate() {
194                    *res += result_lag_num[j] * result_lag_den[j];
195                }
196            }
197        }
198
199        result_den = batch_inversion(&result_den);
200        result_num.iter().zip(result_den).map(|(n, d)| *n * d).collect()
201    }
202
203    /// For each queried set of composition polynomial column evaluations, combine evaluations
204    /// into a single value by computing their random linear combination as follows:
205    ///
206    /// - Assume each queried value is an evaluation of a composition polynomial column H_i(x).
207    /// - For each H_i(x), compute H'_i(x) = (H_i(x) - H(z)) / (x - z).
208    /// - Then, combine all H_i(x) values together by computing H(x) = sum(H_i(x) * cc_i) for
209    ///   all i, where cc_i is the coefficient for the random linear combination drawn from the
210    ///   public coin.
211    ///
212    /// Note that values of H_i(z) are received from the prover and passed into this function
213    /// via the `ood_evaluations` parameter.
214    pub fn compose_constraint_evaluations(
215        &self,
216        queried_evaluations: Table<E>,
217        ood_evaluations: Vec<E>,
218    ) -> Vec<E> {
219        assert_eq!(queried_evaluations.num_rows(), self.x_coordinates.len());
220
221        let n = queried_evaluations.num_rows();
222        let mut result_num = Vec::<E>::with_capacity(n);
223        let mut result_den = Vec::<E>::with_capacity(n);
224
225        let z = self.z[0];
226
227        // combine composition polynomial columns separately for numerators and denominators;
228        // this way we can use batch inversion in the end.
229        for (query_values, &x) in queried_evaluations.rows().zip(&self.x_coordinates) {
230            let mut composition_num = E::ZERO;
231            for (i, &evaluation) in query_values.iter().enumerate() {
232                // compute the numerator of H'_i(x) as (H_i(x) - H_i(z)), multiply it by a
233                // composition coefficient, and add the result to the numerator aggregator
234                composition_num += (evaluation - ood_evaluations[i]) * self.cc.constraints[i];
235            }
236            result_num.push(composition_num);
237            result_den.push(x - z);
238        }
239
240        result_den = batch_inversion(&result_den);
241        result_num.iter().zip(result_den).map(|(n, d)| *n * d).collect()
242    }
243
244    /// Combines trace and constraint compositions together.
245    pub fn combine_compositions(&self, t_composition: Vec<E>, c_composition: Vec<E>) -> Vec<E> {
246        assert_eq!(t_composition.len(), self.x_coordinates.len());
247        assert_eq!(c_composition.len(), self.x_coordinates.len());
248
249        let mut result = Vec::with_capacity(self.x_coordinates.len());
250        for (t, c) in t_composition.iter().zip(c_composition) {
251            // compute C(x) by adding the two compositions together
252            result.push(*t + c);
253        }
254
255        result
256    }
