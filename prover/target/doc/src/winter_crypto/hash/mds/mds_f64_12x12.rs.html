6//! This module contains helper functions as well as constants used to perform a 12x12 vector-matrix
7//! multiplication. The special form of our MDS matrix i.e. being circulant, allows us to reduce
8//! the vector-matrix multiplication to a Hadamard product of two vectors in "frequency domain".
9//! This follows from the simple fact that every circulant matrix has the columns of the discrete
10//! Fourier transform matrix as orthogonal eigenvectors.
11//! The implementation also avoids the use of 3-point FFTs, and 3-point iFFTs, and substitutes that
12//! with explicit expressions. It also avoids, due to the form of our matrix in the frequency domain,
13//! divisions by 2 and repeated modular reductions. This is because of our explicit choice of
14//! an MDS matrix that has small powers of 2 entries in frequency domain.
15//! The following implementation has benefited greatly from the discussions and insights of
16//! Hamish Ivey-Law and Jacqueline Nabaglo of Polygon Zero and is based on Nabaglo's implementation
17//! in [Plonky2](https://github.com/mir-protocol/plonky2).
18//! The circulant matrix is identified by its first row: [7, 23, 8, 26, 13, 10, 9, 7, 6, 22, 21, 8].
19
20// FFT-BASED MDS MULTIPLICATION HELPER FUNCTIONS
21// ================================================================================================
22
23use math::{
24    fft::real_u64::{fft4_real, ifft4_real_unreduced},
25    fields::f64::BaseElement,
26    FieldElement,
27};

29// MDS matrix in frequency domain.
30// More precisely, this is the output of the three 4-point (real) FFTs of the first column of
31// the MDS matrix i.e. just before the multiplication with the appropriate twiddle factors
32// and application of the final four 3-point FFT in order to get the full 12-point FFT.
33// The entries have been scaled appropriately in order to avoid divisions by 2 in iFFT2 and iFFT4.
34// The code to generate the matrix in frequency domain is based on an adaptation of a code, to generate
35// MDS matrices efficiently in original domain, that was developed by the Polygon Zero team.
36const MDS_FREQ_BLOCK_ONE: [i64; 3] = [16, 8, 16];
37const MDS_FREQ_BLOCK_TWO: [(i64, i64); 3] = [(-1, 2), (-1, 1), (4, 8)];
38const MDS_FREQ_BLOCK_THREE: [i64; 3] = [-8, 1, 1];
39
40pub(crate) fn mds_multiply(state: &mut [BaseElement; 12]) {
41    let mut result = [BaseElement::ZERO; 12];
42
43    // Using the linearity of the operations we can split the state into a low||high decomposition
44    // and operate on each with no overflow and then combine/reduce the result to a field element.
45    let mut state_l = [0u64; 12];
46    let mut state_h = [0u64; 12];
47
48    for r in 0..12 {
49        let s = state[r].inner();
50        state_h[r] = s >> 32;
51        state_l[r] = (s as u32) as u64;
52    }
53
54    let state_h = mds_multiply_freq(state_h);
55    let state_l = mds_multiply_freq(state_l);
56
57    for r in 0..12 {
58        let s = state_l[r] as u128 + ((state_h[r] as u128) << 32);
59        let s_hi = (s >> 64) as u64;
60        let s_lo = s as u64;
61        let z = (s_hi << 32) - s_hi;
62        let (res, over) = s_lo.overflowing_add(z);
63
64        result[r] = BaseElement::from_mont(res.wrapping_add(0u32.wrapping_sub(over as u32) as u64));
65    }
66    *state = result;
67}

69// We use split 3 x 4 FFT transform in order to transform our vectors into the frequency domain.
70#[inline(always)]
71pub(crate) fn mds_multiply_freq(state: [u64; 12]) -> [u64; 12] {
72    let [s0, s1, s2, s3, s4, s5, s6, s7, s8, s9, s10, s11] = state;
73
74    let (u0, u1, u2) = fft4_real([s0, s3, s6, s9]);
75    let (u4, u5, u6) = fft4_real([s1, s4, s7, s10]);
76    let (u8, u9, u10) = fft4_real([s2, s5, s8, s11]);
77
78    // This where the multiplication in frequency domain is done. More precisely, and with
79    // the appropriate permuations in between, the sequence of
80    // 3-point FFTs --> multiplication by twiddle factors --> Hadamard multiplication -->
81    // 3 point iFFTs --> multiplication by (inverse) twiddle factors
82    // is "squashed" into one step composed of the functions "block1", "block2" and "block3".
83    // The expressions in the aforementioned functions are the result of explicit computations
84    // combined with the Karatsuba trick for the multiplication of Complex numbers.
85
86    let [v0, v4, v8] = block1([u0, u4, u8], MDS_FREQ_BLOCK_ONE);
87    let [v1, v5, v9] = block2([u1, u5, u9], MDS_FREQ_BLOCK_TWO);
88    let [v2, v6, v10] = block3([u2, u6, u10], MDS_FREQ_BLOCK_THREE);
89    // The 4th block is not computed as it is similar to the 2nd one, up to complex conjugation,
90    // and is, due to the use of the real FFT and iFFT, redundant.
91
92    let [s0, s3, s6, s9] = ifft4_real_unreduced((v0, v1, v2));
93    let [s1, s4, s7, s10] = ifft4_real_unreduced((v4, v5, v6));
94    let [s2, s5, s8, s11] = ifft4_real_unreduced((v8, v9, v10));
95
96    [s0, s1, s2, s3, s4, s5, s6, s7, s8, s9, s10, s11]
97}

99#[inline(always)]
100fn block1(x: [i64; 3], y: [i64; 3]) -> [i64; 3] {
101    let [x0, x1, x2] = x;
102    let [y0, y1, y2] = y;
103    let z0 = x0 * y0 + x1 * y2 + x2 * y1;
104    let z1 = x0 * y1 + x1 * y0 + x2 * y2;
105    let z2 = x0 * y2 + x1 * y1 + x2 * y0;
106
107    [z0, z1, z2]
108}

110#[inline(always)]
111fn block2(x: [(i64, i64); 3], y: [(i64, i64); 3]) -> [(i64, i64); 3] {
112    let [(x0r, x0i), (x1r, x1i), (x2r, x2i)] = x;
113    let [(y0r, y0i), (y1r, y1i), (y2r, y2i)] = y;
114    let x0s = x0r + x0i;
115    let x1s = x1r + x1i;
116    let x2s = x2r + x2i;
117    let y0s = y0r + y0i;
118    let y1s = y1r + y1i;
119    let y2s = y2r + y2i;
120
121    // Compute x0​y0 ​− ix1​y2​ − ix2​y1​ using Karatsuba for complex numbers multiplication
122    let m0 = (x0r * y0r, x0i * y0i);
123    let m1 = (x1r * y2r, x1i * y2i);
124    let m2 = (x2r * y1r, x2i * y1i);
125    let z0r = (m0.0 - m0.1) + (x1s * y2s - m1.0 - m1.1) + (x2s * y1s - m2.0 - m2.1);
126    let z0i = (x0s * y0s - m0.0 - m0.1) + (-m1.0 + m1.1) + (-m2.0 + m2.1);
127    let z0 = (z0r, z0i);
128
129    // Compute x0​y1​ + x1​y0​ − ix2​y2 using Karatsuba for complex numbers multiplication
130    let m0 = (x0r * y1r, x0i * y1i);
131    let m1 = (x1r * y0r, x1i * y0i);
132    let m2 = (x2r * y2r, x2i * y2i);
133    let z1r = (m0.0 - m0.1) + (m1.0 - m1.1) + (x2s * y2s - m2.0 - m2.1);
134    let z1i = (x0s * y1s - m0.0 - m0.1) + (x1s * y0s - m1.0 - m1.1) + (-m2.0 + m2.1);
135    let z1 = (z1r, z1i);
136
137    // Compute x0​y2​ + x1​y1 ​+ x2​y0​ using Karatsuba for complex numbers multiplication
138    let m0 = (x0r * y2r, x0i * y2i);
139    let m1 = (x1r * y1r, x1i * y1i);
140    let m2 = (x2r * y0r, x2i * y0i);
141    let z2r = (m0.0 - m0.1) + (m1.0 - m1.1) + (m2.0 - m2.1);
142    let z2i = (x0s * y2s - m0.0 - m0.1) + (x1s * y1s - m1.0 - m1.1) + (x2s * y0s - m2.0 - m2.1);
143    let z2 = (z2r, z2i);
144
145    [z0, z1, z2]
146}

148#[inline(always)]
149fn block3(x: [i64; 3], y: [i64; 3]) -> [i64; 3] {
150    let [x0, x1, x2] = x;
151    let [y0, y1, y2] = y;
152    let z0 = x0 * y0 - x1 * y2 - x2 * y1;
153    let z1 = x0 * y1 + x1 * y0 - x2 * y2;
154    let z2 = x0 * y2 + x1 * y1 + x2 * y0;
155
156    [z0, z1, z2]
157}
