16// BYTE READER TRAIT
17// ================================================================================================
18
19/// Defines how primitive values are to be read from `Self`.
20///
21/// Whenever data is read from the reader using any of the `read_*` functions, the reader advances
22/// to the next unread byte. If the error occurs, the reader is not rolled back to the state prior
23/// to calling any of the function.
24pub trait ByteReader {
25    // REQUIRED METHODS
26    // --------------------------------------------------------------------------------------------
27
28    /// Returns a single byte read from `self`.
29    ///
30    /// # Errors
31    /// Returns a [DeserializationError] error the reader is at EOF.
32    fn read_u8(&mut self) -> Result<u8, DeserializationError>;
33
34    /// Returns the next byte to be read from `self` without advancing the reader to the next byte.
35    ///
36    /// # Errors
37    /// Returns a [DeserializationError] error the reader is at EOF.
38    fn peek_u8(&self) -> Result<u8, DeserializationError>;
39
40    /// Returns a slice of bytes of the specified length read from `self`.
41    ///
42    /// # Errors
43    /// Returns a [DeserializationError] if a slice of the specified length could not be read
44    /// from `self`.
45    fn read_slice(&mut self, len: usize) -> Result<&[u8], DeserializationError>;
46
47    /// Returns a byte array of length `N` read from `self`.
48    ///
49    /// # Errors
50    /// Returns a [DeserializationError] if an array of the specified length could not be read
51    /// from `self`.
52    fn read_array<const N: usize>(&mut self) -> Result<[u8; N], DeserializationError>;
53
54    /// Checks if it is possible to read at least `num_bytes` bytes from this ByteReader
55    ///
56    /// # Errors
57    /// Returns an error if, when reading the requested number of bytes, we go beyond the
58    /// the data available in the reader.
59    fn check_eor(&self, num_bytes: usize) -> Result<(), DeserializationError>;
60
61    /// Returns true if there are more bytes left to be read from `self`.
62    fn has_more_bytes(&self) -> bool;
63
64    // PROVIDED METHODS
65    // --------------------------------------------------------------------------------------------
66
67    /// Returns a boolean value read from `self` consuming 1 byte from the reader.
68    ///
69    /// # Errors
70    /// Returns a [DeserializationError] if a u16 value could not be read from `self`.
71    fn read_bool(&mut self) -> Result<bool, DeserializationError> {
72        let byte = self.read_u8()?;
73        match byte {
74            0 => Ok(false),
75            1 => Ok(true),
76            _ => Err(DeserializationError::InvalidValue(format!("{byte} is not a boolean value"))),
77        }
78    }
79
80    /// Returns a u16 value read from `self` in little-endian byte order.
81    ///
82    /// # Errors
83    /// Returns a [DeserializationError] if a u16 value could not be read from `self`.
84    fn read_u16(&mut self) -> Result<u16, DeserializationError> {
85        let bytes = self.read_array::<2>()?;
86        Ok(u16::from_le_bytes(bytes))
87    }
88
89    /// Returns a u32 value read from `self` in little-endian byte order.
90    ///
91    /// # Errors
92    /// Returns a [DeserializationError] if a u32 value could not be read from `self`.
93    fn read_u32(&mut self) -> Result<u32, DeserializationError> {
94        let bytes = self.read_array::<4>()?;
95        Ok(u32::from_le_bytes(bytes))
96    }
97
98    /// Returns a u64 value read from `self` in little-endian byte order.
99    ///
100    /// # Errors
101    /// Returns a [DeserializationError] if a u64 value could not be read from `self`.
102    fn read_u64(&mut self) -> Result<u64, DeserializationError> {
103        let bytes = self.read_array::<8>()?;
104        Ok(u64::from_le_bytes(bytes))
105    }
106
107    /// Returns a u128 value read from `self` in little-endian byte order.
108    ///
109    /// # Errors
110    /// Returns a [DeserializationError] if a u128 value could not be read from `self`.
111    fn read_u128(&mut self) -> Result<u128, DeserializationError> {
112        let bytes = self.read_array::<16>()?;
113        Ok(u128::from_le_bytes(bytes))
114    }
115
116    /// Returns a usize value read from `self` in [vint64](https://docs.rs/vint64/latest/vint64/)
117    /// format.
118    ///
119    /// # Errors
120    /// Returns a [DeserializationError] if:
121    /// * usize value could not be read from `self`.
122    /// * encoded value is greater than `usize` maximum value on a given platform.
123    fn read_usize(&mut self) -> Result<usize, DeserializationError> {
124        let first_byte = self.peek_u8()?;
125        let length = first_byte.trailing_zeros() as usize + 1;
126
127        let result = if length == 9 {
128            // 9-byte special case
129            self.read_u8()?;
130            let value = self.read_array::<8>()?;
131            u64::from_le_bytes(value)
132        } else {
133            let mut encoded = [0u8; 8];
134            let value = self.read_slice(length)?;
135            encoded[..length].copy_from_slice(value);
136            u64::from_le_bytes(encoded) >> length
137        };
138
139        // check if the result value is within acceptable bounds for `usize` on a given platform
140        if result > usize::MAX as u64 {
141            return Err(DeserializationError::InvalidValue(format!(
142                "Encoded value must be less than {}, but {} was provided",
143                usize::MAX,
144                result
145            )));
146        }
147
148        Ok(result as usize)
149    }
150
151    /// Returns a byte vector of the specified length read from `self`.
152    ///
153    /// # Errors
154    /// Returns a [DeserializationError] if a vector of the specified length could not be read
155    /// from `self`.
156    fn read_vec(&mut self, len: usize) -> Result<Vec<u8>, DeserializationError> {
157        let data = self.read_slice(len)?;
158        Ok(data.to_vec())
159    }
160
161    /// Returns a String of the specified length read from `self`.
162    ///
163    /// # Errors
164    /// Returns a [DeserializationError] if a String of the specified length could not be read
165    /// from `self`.
166    fn read_string(&mut self, num_bytes: usize) -> Result<String, DeserializationError> {
167        let data = self.read_vec(num_bytes)?;
168        String::from_utf8(data).map_err(|err| DeserializationError::InvalidValue(format!("{err}")))
169    }
170
171    /// Reads a deserializable value from `self`.
172    ///
173    /// # Errors
174    /// Returns a [DeserializationError] if the specified value could not be read from `self`.
175    fn read<D>(&mut self) -> Result<D, DeserializationError>
176    where
177        Self: Sized,
178        D: Deserializable,
179    {
180        D::read_from(self)
181    }
182
183    /// Reads a sequence of bytes from `self`, attempts to deserialize these bytes into a vector
184    /// with the specified number of `D` elements, and returns the result.
185    ///
186    /// # Errors
187    /// Returns a [DeserializationError] if the specified number elements could not be read from
188    /// `self`.
189    fn read_many<D>(&mut self, num_elements: usize) -> Result<Vec<D>, DeserializationError>
190    where
191        Self: Sized,
192        D: Deserializable,
193    {
194        let mut result = Vec::with_capacity(num_elements);
195        for _ in 0..num_elements {
196            let element = D::read_from(self)?;
197            result.push(element)
198        }
199        Ok(result)
200    }
201}

206/// An adapter of [ByteReader] to any type that implements [std::io::Read]
207///
208/// In particular, this covers things like [std::fs::File], standard input, etc.
209#[cfg(feature = "std")]
210pub struct ReadAdapter<'a> {
211    // NOTE: The [ByteReader] trait does not currently support reader implementations that require
212    // mutation during `peek_u8`, `has_more_bytes`, and `check_eor`. These (or equivalent)
213    // operations on the standard library [std::io::BufRead] trait require a mutable reference, as
214    // it may be necessary to read from the underlying input to implement them.
215    //
216    // To handle this, we wrap the underlying reader in an [RefCell], this allows us to mutate the
217    // reader if necessary during a call to one of the above-mentioned trait methods, without
218    // sacrificing safety - at the cost of enforcing Rust's borrowing semantics dynamically.
219    //
220    // This should not be a problem in practice, except in the case where `read_slice` is called,
221    // and the reference returned is from `reader` directly, rather than `buf`. If a call to one
222    // of the above-mentioned methods is made while that reference is live, and we attempt to read
223    // from `reader`, a panic will occur.
224    //
225    // Ultimately, this should be addressed by making the [ByteReader] trait align with the standard
226    // library I/O traits, so this is a temporary solution.
227    reader: RefCell<std::io::BufReader<&'a mut dyn std::io::Read>>,
228    // A temporary buffer to store chunks read from `reader` that are larger than what is required for
229    // the higher-level [ByteReader] APIs.
230    //
231    // By default we attempt to satisfy reads from `reader` directly, but that is not always possible.
232    buf: alloc::vec::Vec<u8>,
233    // The position in `buf` at which we should start reading the next byte, when `buf` is non-empty.
234    pos: usize,
235    // This is set when we attempt to read from `reader` and get an empty buffer. This indicates that
236    // once we exhaust `buf`, we have truly reached end-of-file.
237    //
238    // We will use this to more accurately handle functions like `has_more_bytes` when this is set.
239    guaranteed_eof: bool,
240}
241
242#[cfg(feature = "std")]
243impl<'a> ReadAdapter<'a> {
244    /// Create a new [ByteReader] adapter for the given implementation of [std::io::Read]
245    pub fn new(reader: &'a mut dyn std::io::Read) -> Self {
246        Self {
247            reader: RefCell::new(std::io::BufReader::with_capacity(256, reader)),
248            buf: Default::default(),
249            pos: 0,
250            guaranteed_eof: false,
251        }
252    }
253
254    /// Get the internal adapter buffer as a (possibly empty) slice of bytes
255    #[inline(always)]
256    fn buffer(&self) -> &[u8] {
257        self.buf.get(self.pos..).unwrap_or(&[])
258    }
259
260    /// Get the internal adapter buffer as a slice of bytes, or `None` if the buffer is empty
261    #[inline(always)]
262    fn non_empty_buffer(&self) -> Option<&[u8]> {
263        self.buf.get(self.pos..).filter(|b| !b.is_empty())
264    }
265
266    /// Return the current reader buffer as a (possibly empty) slice of bytes.
267    ///
268    /// This buffer being empty _does not_ mean we're at EOF, you must call [non_empty_reader_buffer_mut] first.
269    #[inline(always)]
270    fn reader_buffer(&self) -> Ref<'_, [u8]> {
271        Ref::map(self.reader.borrow(), |r| r.buffer())
272    }
273
274    /// Return the current reader buffer, reading from the underlying reader
275    /// if the buffer is empty.
276    ///
277    /// Returns `Ok` only if the buffer is non-empty, and no errors occurred
278    /// while filling it (if filling was needed).
279    fn non_empty_reader_buffer_mut(&mut self) -> Result<&[u8], DeserializationError> {
280        use std::io::ErrorKind;
281        let buf = self.reader.get_mut().fill_buf().map_err(|e| match e.kind() {
282            ErrorKind::UnexpectedEof => DeserializationError::UnexpectedEOF,
283            e => DeserializationError::UnknownError(e.to_string()),
284        })?;
285        if buf.is_empty() {
286            self.guaranteed_eof = true;
287            Err(DeserializationError::UnexpectedEOF)
288        } else {
289            Ok(buf)
290        }
291    }
292
293    /// Same as [non_empty_reader_buffer_mut], but with dynamically-enforced
294    /// borrow check rules so that it can be called in functions like `peek_u8`.
295    ///
296    /// This comes with overhead for the dynamic checks, so you should prefer
297    /// to call [non_empty_reader_buffer_mut] if you already have a mutable
298    /// reference to `self`
299    fn non_empty_reader_buffer(&self) -> Result<Ref<'_, [u8]>, DeserializationError> {
300        use std::io::ErrorKind;
301        let mut reader = self.reader.borrow_mut();
302        let buf = reader.fill_buf().map_err(|e| match e.kind() {
303            ErrorKind::UnexpectedEof => DeserializationError::UnexpectedEOF,
304            e => DeserializationError::UnknownError(e.to_string()),
305        })?;
306        if buf.is_empty() {
307            Err(DeserializationError::UnexpectedEOF)
308        } else {
309            // Re-borrow immutably
310            drop(reader);
311            Ok(self.reader_buffer())
312        }
313    }
314
315    /// Returns true if there is sufficient capacity remaining in `buf` to hold `n` bytes
316    #[inline]
317    fn has_remaining_capacity(&self, n: usize) -> bool {
318        let remaining = self.buf.capacity() - self.buffer().len();
319        remaining >= n
320    }
321
322    /// Takes the next byte from the input, returning an error if the operation fails
323    fn pop(&mut self) -> Result<u8, DeserializationError> {
324        if let Some(byte) = self.non_empty_buffer().map(|b| b[0]) {
325            self.pos += 1;
326            return Ok(byte);
327        }
328        let result = self.non_empty_reader_buffer_mut().map(|b| b[0]);
329        if result.is_ok() {
330            self.reader.get_mut().consume(1);
331        } else {
332            self.guaranteed_eof = true;
333        }
334        result
335    }
336
337    /// Takes the next `N` bytes from the input as an array, returning an error if the operation fails
338    fn read_exact<const N: usize>(&mut self) -> Result<[u8; N], DeserializationError> {
339        let buf = self.buffer();
340        let mut output = [0; N];
341        match buf.len() {
342            0 => {
343                let buf = self.non_empty_reader_buffer_mut()?;
344                if buf.len() < N {
345                    return Err(DeserializationError::UnexpectedEOF);
346                }
347                // SAFETY: This copy is guaranteed to be safe, as we have validated above
348                // that `buf` has at least N bytes, and `output` is defined to be exactly
349                // N bytes.
350                unsafe {
351                    core::ptr::copy_nonoverlapping(buf.as_ptr(), output.as_mut_ptr(), N);
352                }
353                self.reader.get_mut().consume(N);
354            },
355            n if n >= N => {
356                // SAFETY: This copy is guaranteed to be safe, as we have validated above
357                // that `buf` has at least N bytes, and `output` is defined to be exactly
358                // N bytes.
359                unsafe {
360                    core::ptr::copy_nonoverlapping(buf.as_ptr(), output.as_mut_ptr(), N);
361                }
362                self.pos += N;
363            },
364            n => {
365                // We have to fill from both the local and reader buffers
366                self.non_empty_reader_buffer_mut()?;
367                let reader_buf = self.reader_buffer();
368                match reader_buf.len() {
369                    #[cfg(debug_assertions)]
370                    0 => unreachable!("expected reader buffer to be non-empty to reach here"),
371                    #[cfg(not(debug_assertions))]
372                    // SAFETY: The call to `non_empty_reader_buffer_mut` will return an error
373                    // if `reader_buffer` is non-empty, as a result is is impossible to reach
374                    // here with a length of 0.
375                    0 => unsafe { core::hint::unreachable_unchecked() },
376                    // We got enough in one request
377                    m if m + n >= N => {
378                        let needed = N - n;
379                        let dst = output.as_mut_ptr();
380                        // SAFETY: Both copies are guaranteed to be in-bounds:
381                        //
382                        // * `output` is defined to be exactly N bytes
383                        // * `buf` is guaranteed to be < N bytes
384                        // * `reader_buf` is guaranteed to have the remaining bytes needed,
385                        // and we only copy exactly that many bytes
386                        unsafe {
387                            core::ptr::copy_nonoverlapping(self.buffer().as_ptr(), dst, n);
388                            core::ptr::copy_nonoverlapping(reader_buf.as_ptr(), dst.add(n), needed);
389                            drop(reader_buf);
390                        }
391                        self.pos += n;
392                        self.reader.get_mut().consume(needed);
393                    },
394                    // We didn't get enough, but haven't necessarily reached eof yet, so fall back
395                    // to filling `self.buf`
396                    m => {
397                        let needed = N - (m + n);
398                        drop(reader_buf);
399                        self.buffer_at_least(needed)?;
400                        debug_assert!(self.buffer().len() >= N, "expected buffer to be at least {N} bytes after call to buffer_at_least");
401                        // SAFETY: This is guaranteed to be an in-bounds copy
402                        unsafe {
403                            core::ptr::copy_nonoverlapping(
404                                self.buffer().as_ptr(),
405                                output.as_mut_ptr(),
406                                N,
407                            );
408                        }
409                        self.pos += N;
410                        return Ok(output);
411                    },
412                }
413            },
414        }
415
416        // Check if we should reset our internal buffer
417        if self.buffer().is_empty() && self.pos > 0 {
418            unsafe {
419                self.buf.set_len(0);
420            }
421        }
422
423        Ok(output)
424    }
425
426    /// Fill `self.buf` with `count` bytes
427    ///
428    /// This should only be called when we can't read from the reader directly
429    fn buffer_at_least(&mut self, mut count: usize) -> Result<(), DeserializationError> {
430        // Read until we have at least `count` bytes, or until we reach end-of-file,
431        // which ever comes first.
432        loop {
433            // If we have succesfully read `count` bytes, we're done
434            if count == 0 || self.buf.len() >= count {
435                break Ok(());
436            }
437
438            // This operation will return an error if the underlying reader hits EOF
439            self.non_empty_reader_buffer_mut()?;
440
441            // Extend `self.buf` with the bytes read from the underlying reader.
442            //
443            // NOTE: We have to re-borrow the reader buffer here, since we can't get a mutable
444            // reference to `self.buf` while holding an immutable reference to the reader buffer.
445            let reader = self.reader.get_mut();
446            let buf = reader.buffer();
447            let consumed = buf.len();
448            self.buf.extend_from_slice(buf);
449            reader.consume(consumed);
450            count = count.saturating_sub(consumed);
451        }
452    }
453}

455#[cfg(feature = "std")]
456impl ByteReader for ReadAdapter<'_> {
457    #[inline(always)]
458    fn read_u8(&mut self) -> Result<u8, DeserializationError> {
459        self.pop()
460    }
461
462    /// NOTE: If we happen to not have any bytes buffered yet when this is called, then we will be
463    /// forced to try and read from the underlying reader. This requires a mutable reference, which
464    /// is obtained dynamically via [RefCell].
465    ///
466    /// <div class="warning">
467    /// Callers must ensure that they do not hold any immutable references to the buffer of this
468    /// reader when calling this function so as to avoid a situtation in which the dynamic borrow
469    /// check fails. Specifically, you must not be holding a reference to the result of
470    /// [Self::read_slice] when this function is called.
471    /// </div>
472    fn peek_u8(&self) -> Result<u8, DeserializationError> {
473        if let Some(byte) = self.buffer().first() {
474            return Ok(*byte);
475        }
476        self.non_empty_reader_buffer().map(|b| b[0])
477    }
478
479    fn read_slice(&mut self, len: usize) -> Result<&[u8], DeserializationError> {
480        // Edge case
481        if len == 0 {
482            return Ok(&[]);
483        }
484
485        // If we have unused buffer, and the consumed portion is
486        // large enough, we will move the unused portion of the buffer
487        // to the start, freeing up bytes at the end for more reads
488        // before forcing a reallocation
489        let should_optimize_storage = self.pos >= 16 && !self.has_remaining_capacity(len);
490        if should_optimize_storage {
491            // We're going to optimize storage first
492            let buf = self.buffer();
493            let src = buf.as_ptr();
494            let count = buf.len();
495            let dst = self.buf.as_mut_ptr();
496            unsafe {
497                core::ptr::copy(src, dst, count);
498                self.buf.set_len(count);
499                self.pos = 0;
500            }
501        }
502
503        // Fill the buffer so we have at least `len` bytes available,
504        // this will return an error if we hit EOF first
505        self.buffer_at_least(len)?;
506
507        let slice = &self.buf[self.pos..(self.pos + len)];
508        self.pos += len;
509        Ok(slice)
510    }
511
512    #[inline]
513    fn read_array<const N: usize>(&mut self) -> Result<[u8; N], DeserializationError> {
514        if N == 0 {
515            return Ok([0; N]);
516        }
517        self.read_exact()
518    }
519
520    fn check_eor(&self, num_bytes: usize) -> Result<(), DeserializationError> {
521        // Do we have sufficient data in the local buffer?
522        let buffer_len = self.buffer().len();
523        if buffer_len >= num_bytes {
524            return Ok(());
525        }
526
527        // What about if we include what is in the local buffer and the reader's buffer?
528        let reader_buffer_len = self.non_empty_reader_buffer().map(|b| b.len())?;
529        let buffer_len = buffer_len + reader_buffer_len;
530        if buffer_len >= num_bytes {
531            return Ok(());
532        }
533
534        // We have no more input, thus can't fulfill a request of `num_bytes`
535        if self.guaranteed_eof {
536            return Err(DeserializationError::UnexpectedEOF);
537        }
538
539        // Because this function is read-only, we must optimistically assume we can read `num_bytes`
540        // from the input, and fail later if that does not hold. We know we're not at EOF yet, but
541        // that's all we can say without buffering more from the reader. We could make use of
542        // `buffer_at_least`, which would guarantee a correct result, but it would also impose
543        // additional restrictions on the use of this function, e.g. not using it while holding a
544        // reference returned from `read_slice`. Since it is not a memory safety violation to return
545        // an optimistic result here, it makes for a better tradeoff.
546        Ok(())
547    }
548
549    #[inline]
550    fn has_more_bytes(&self) -> bool {
551        !self.buffer().is_empty() || self.non_empty_reader_buffer().is_ok()
552    }
553}

558#[cfg(feature = "std")]
559macro_rules! cursor_remaining_buf {
560    ($cursor:ident) => {{
561        let buf = $cursor.get_ref().as_ref();
562        let start = $cursor.position().min(buf.len() as u64) as usize;
563        &buf[start..]
564    }};
565}
566
567#[cfg(feature = "std")]
568impl<T: AsRef<[u8]>> ByteReader for std::io::Cursor<T> {
569    fn read_u8(&mut self) -> Result<u8, DeserializationError> {
570        let buf = cursor_remaining_buf!(self);
571        if buf.is_empty() {
572            Err(DeserializationError::UnexpectedEOF)
573        } else {
574            let byte = buf[0];
575            self.set_position(self.position() + 1);
576            Ok(byte)
577        }
578    }
579
580    fn peek_u8(&self) -> Result<u8, DeserializationError> {
581        cursor_remaining_buf!(self)
582            .first()
583            .copied()
584            .ok_or(DeserializationError::UnexpectedEOF)
585    }
586
587    fn read_slice(&mut self, len: usize) -> Result<&[u8], DeserializationError> {
588        let pos = self.position();
589        let size = self.get_ref().as_ref().len() as u64;
590        if size.saturating_sub(pos) < len as u64 {
591            Err(DeserializationError::UnexpectedEOF)
592        } else {
593            self.set_position(pos + len as u64);
594            let start = pos.min(size) as usize;
595            Ok(&self.get_ref().as_ref()[start..(start + len)])
596        }
597    }
598
599    fn read_array<const N: usize>(&mut self) -> Result<[u8; N], DeserializationError> {
600        self.read_slice(N).map(|bytes| {
601            let mut result = [0u8; N];
602            result.copy_from_slice(bytes);
603            result
604        })
605    }
606
607    fn check_eor(&self, num_bytes: usize) -> Result<(), DeserializationError> {
608        if cursor_remaining_buf!(self).len() >= num_bytes {
609            Ok(())
610        } else {
611            Err(DeserializationError::UnexpectedEOF)
612        }
613    }
614
615    #[inline]
616    fn has_more_bytes(&self) -> bool {
617        let pos = self.position();
618        let size = self.get_ref().as_ref().len() as u64;
619        pos < size
620    }
621}

623// SLICE READER
624// ================================================================================================
625
626/// Implements [ByteReader] trait for a slice of bytes.
627///
628/// NOTE: If you are building with the `std` feature, you should probably prefer [std::io::Cursor]
629/// instead. However, [SliceReader] is still useful in no-std environments until stabilization of
630/// the `core_io_borrowed_buf` feature.
631pub struct SliceReader<'a> {
632    source: &'a [u8],
633    pos: usize,
634}
635
636impl<'a> SliceReader<'a> {
637    /// Creates a new slice reader from the specified slice.
638    pub fn new(source: &'a [u8]) -> Self {
639        SliceReader { source, pos: 0 }
640    }
641}
642
643impl ByteReader for SliceReader<'_> {
644    fn read_u8(&mut self) -> Result<u8, DeserializationError> {
645        self.check_eor(1)?;
646        let result = self.source[self.pos];
647        self.pos += 1;
648        Ok(result)
649    }
650
651    fn peek_u8(&self) -> Result<u8, DeserializationError> {
652        self.check_eor(1)?;
653        Ok(self.source[self.pos])
654    }
655
656    fn read_slice(&mut self, len: usize) -> Result<&[u8], DeserializationError> {
657        self.check_eor(len)?;
658        let result = &self.source[self.pos..self.pos + len];
659        self.pos += len;
660        Ok(result)
661    }
662
663    fn read_array<const N: usize>(&mut self) -> Result<[u8; N], DeserializationError> {
664        self.check_eor(N)?;
665        let mut result = [0_u8; N];
666        result.copy_from_slice(&self.source[self.pos..self.pos + N]);
667        self.pos += N;
668        Ok(result)
669    }
670
671    fn check_eor(&self, num_bytes: usize) -> Result<(), DeserializationError> {
672        if self.pos + num_bytes > self.source.len() {
673            return Err(DeserializationError::UnexpectedEOF);
674        }
675        Ok(())
676    }
677
678    fn has_more_bytes(&self) -> bool {
679        self.pos < self.source.len()
680    }
681}
