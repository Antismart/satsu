27/// Boundary constraints for a computation.
28///
29/// Boundary constraints are arranged into two categories: constraints against columns of the main
30/// trace segment, and constraints against columns of the auxiliary trace segment. Within each
31/// category, the constraints are grouped by their divisor (see [BoundaryConstraintGroup] for
32/// more info on boundary constraint structure).
33///
34/// When the protocol is run in a large field, types `B` and `E` are the same. However, when
35/// working with small fields, these types are used as follows:
36/// * Constraints against columns of the main segment of the execution trace are defined over the
37///   field specified by `B`.
38/// * Constraints against columns of auxiliary segments of the execution trace (if any) are defined
39///   over the field specified by `E`.
40/// * Constraint composition coefficients are defined over the field specified by `E`.
41/// * Constraint divisors are defined over the field specified by `B`.
42pub struct BoundaryConstraints<E: FieldElement> {
43    main_constraints: Vec<BoundaryConstraintGroup<E::BaseField, E>>,
44    aux_constraints: Vec<BoundaryConstraintGroup<E, E>>,
45}
46
47impl<E: FieldElement> BoundaryConstraints<E> {
48    // CONSTRUCTOR
49    // --------------------------------------------------------------------------------------------
50    /// Returns a new instance of [BoundaryConstraints] for a computation described by the provided
51    /// assertions and AIR context.
52    ///
53    /// # Panics
54    /// Panics if:
55    /// * The number of provided assertions does not match the number of assertions described by
56    ///   the context.
57    /// * The number of assertions does not match the number of the provided composition
58    ///   coefficients.
59    /// * The specified assertions are not valid in the context of the computation (e.g., assertion
60    ///   column index is out of bounds).
61    pub fn new(
62        context: &AirContext<E::BaseField>,
63        main_assertions: Vec<Assertion<E::BaseField>>,
64        aux_assertions: Vec<Assertion<E>>,
65        composition_coefficients: &[E],
66    ) -> Self {
67        // make sure the provided assertions are consistent with the specified context
68        assert_eq!(
69            main_assertions.len(),
70            context.num_main_assertions,
71            "expected {} assertions against main trace segment, but received {}",
72            context.num_main_assertions,
73            main_assertions.len(),
74        );
75
76        assert_eq!(
77            aux_assertions.len(),
78            context.num_aux_assertions,
79            "expected {} assertions against the auxiliary trace segment, but received {}",
80            context.num_aux_assertions,
81            aux_assertions.len(),
82        );
83
84        assert_eq!(
85            context.num_assertions(),
86            composition_coefficients.len(),
87            "number of assertions must match the number of composition coefficient tuples"
88        );
89
90        let trace_length = context.trace_info.length();
91        let main_trace_width = context.trace_info.main_trace_width();
92        let aux_trace_width = context.trace_info.aux_segment_width();
93
94        // make sure the assertions are valid in the context of their respective trace segments;
95        // also, sort the assertions in the deterministic order so that changing the order of
96        // assertions does not change random coefficients that get assigned to them.
97        let main_assertions = prepare_assertions(main_assertions, main_trace_width, trace_length);
98        let aux_assertions = prepare_assertions(aux_assertions, aux_trace_width, trace_length);
99
100        // compute inverse of the trace domain generator; this will be used for offset
101        // computations when creating sequence constraints
102        let inv_g = context.trace_domain_generator.inv();
103
104        // cache inverse twiddles for multi-value assertions in this map so that we don't have
105        // to re-build them for assertions with identical strides
106        let mut twiddle_map = BTreeMap::new();
107
108        // split composition coefficients into main and auxiliary parts
109        let (main_composition_coefficients, aux_composition_coefficients) =
110            composition_coefficients.split_at(main_assertions.len());
111
112        // build constraints for the assertions against the main trace segment
113        let main_constraints = group_constraints(
114            main_assertions,
115            context,
116            main_composition_coefficients,
117            inv_g,
118            &mut twiddle_map,
119        );
120
121        // build constraints for the assertions against the auxiliary trace segment
122        let aux_constraints = group_constraints(
123            aux_assertions,
124            context,
125            aux_composition_coefficients,
126            inv_g,
127            &mut twiddle_map,
128        );
129
130        Self { main_constraints, aux_constraints }
131    }
132
133    // PUBLIC ACCESSORS
134    // --------------------------------------------------------------------------------------------
135
136    /// Returns a reference to the boundary constraints against the main segment of an execution
137    /// trace. The constraints are grouped by their divisors.
138    pub fn main_constraints(&self) -> &[BoundaryConstraintGroup<E::BaseField, E>] {
139        &self.main_constraints
140    }
141
142    /// Returns a reference to the boundary constraints against auxiliary segments of an execution
143    /// trace. The constraints are grouped by their divisors.
144    pub fn aux_constraints(&self) -> &[BoundaryConstraintGroup<E, E>] {
145        &self.aux_constraints
146    }
147}

152/// Translates the provided assertions into boundary constraints, groups the constraints by their
153/// divisor, and sorts the resulting groups by the degree adjustment factor.
154fn group_constraints<F, E>(
155    assertions: Vec<Assertion<F>>,
156    context: &AirContext<F::BaseField>,
157    composition_coefficients: &[E],
158    inv_g: F::BaseField,
159    twiddle_map: &mut BTreeMap<usize, Vec<F::BaseField>>,
160) -> Vec<BoundaryConstraintGroup<F, E>>
161where
162    F: FieldElement,
163    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
164{
165    // iterate over all assertions, which are sorted first by stride and then by first_step
166    // in ascending order
167    let mut groups = BTreeMap::new();
168    for (assertion, &cc) in assertions.into_iter().zip(composition_coefficients) {
169        let key = (assertion.stride(), assertion.first_step());
170        let group = groups.entry(key).or_insert_with(|| {
171            BoundaryConstraintGroup::new(ConstraintDivisor::from_assertion(
172                &assertion,
173                context.trace_len(),
174            ))
175        });
176
177        // add a new assertion constraint to the current group (last group in the list)
178        group.add(assertion, inv_g, twiddle_map, cc);
179    }
180
181    //return a vector of groups
182    groups.into_iter().map(|e| e.1).collect::<Vec<_>>()
183}
184
185/// Makes sure the assertions are valid in the context of this computation and don't overlap with
186/// each other - i.e. no two assertions are placed against the same column and step combination.
187///
188/// This also sorts the assertions in their 'natural order'. The natural order is defined as
189/// sorting first by stride, then by first step, and finally by column, all in ascending order.
190fn prepare_assertions<E: FieldElement>(
191    assertions: Vec<Assertion<E>>,
192    trace_width: usize,
193    trace_length: usize,
194) -> Vec<Assertion<E>> {
195    // we use a sorted set to help us sort the assertions by their 'natural' order
196    let mut result = BTreeSet::<Assertion<E>>::new();
197
198    for assertion in assertions.into_iter() {
199        assertion.validate_trace_width(trace_width).unwrap_or_else(|err| {
200            panic!("assertion {assertion} is invalid: {err}");
201        });
202        assertion.validate_trace_length(trace_length).unwrap_or_else(|err| {
203            panic!("assertion {assertion} is invalid: {err}");
204        });
205        for a in result.iter().filter(|a| a.column == assertion.column) {
206            assert!(
207                !a.overlaps_with(&assertion),
208                "assertion {assertion} overlaps with assertion {a}"
209            );
210        }
211
212        result.insert(assertion);
213    }
214
215    result.into_iter().collect()
216}
