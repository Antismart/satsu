14/// Holds the randomly generated elements necessary to build the auxiliary trace.
15///
16/// Specifically, [`AuxRandElements`] currently supports 3 types of random elements:
17/// - the ones needed to build the Lagrange kernel column (when using GKR to accelerate LogUp),
18/// - the ones needed to build the "s" auxiliary column (when using GKR to accelerate LogUp),
19/// - the ones needed to build all the other auxiliary columns
20#[derive(Debug, Clone)]
21pub struct AuxRandElements<E> {
22    rand_elements: Vec<E>,
23    gkr: Option<GkrRandElements<E>>,
24}
25
26impl<E> AuxRandElements<E> {
27    /// Creates a new [`AuxRandElements`], where the auxiliary trace doesn't contain a Lagrange
28    /// kernel column.
29    pub fn new(rand_elements: Vec<E>) -> Self {
30        Self { rand_elements, gkr: None }
31    }
32
33    /// Creates a new [`AuxRandElements`], where the auxiliary trace contains columns needed when
34    /// using GKR to accelerate LogUp (i.e. a Lagrange kernel column and the "s" column).
35    pub fn new_with_gkr(rand_elements: Vec<E>, gkr: GkrRandElements<E>) -> Self {
36        Self { rand_elements, gkr: Some(gkr) }
37    }
38
39    /// Returns the random elements needed to build all columns other than the two GKR-related ones.
40    pub fn rand_elements(&self) -> &[E] {
41        &self.rand_elements
42    }
43
44    /// Returns the random elements needed to build the Lagrange kernel column.
45    pub fn lagrange(&self) -> Option<&LagrangeKernelRandElements<E>> {
46        self.gkr.as_ref().map(|gkr| &gkr.lagrange)
47    }
48
49    /// Returns the random values used to linearly combine the openings returned from the GKR proof.
50    ///
51    /// These correspond to the lambdas in our documentation.
52    pub fn gkr_openings_combining_randomness(&self) -> Option<&[E]> {
53        self.gkr.as_ref().map(|gkr| gkr.openings_combining_randomness.as_ref())
54    }
55}
56
57/// Holds all the random elements needed when using GKR to accelerate LogUp.
58///
59/// This consists of two sets of random values:
60/// 1. The Lagrange kernel random elements (expanded on in [`LagrangeKernelRandElements`]), and
61/// 2. The "openings combining randomness".
62///
63/// After the verifying the LogUp-GKR circuit, the verifier is left with unproven claims provided
64/// nondeterministically by the prover about the evaluations of the MLE of the main trace columns at
65/// the Lagrange kernel random elements. Those claims are (linearly) combined into one using the
66/// openings combining randomness.
67#[derive(Clone, Debug)]
68pub struct GkrRandElements<E> {
69    lagrange: LagrangeKernelRandElements<E>,
70    openings_combining_randomness: Vec<E>,
71}
72
73impl<E> GkrRandElements<E> {
74    /// Constructs a new [`GkrRandElements`] from [`LagrangeKernelRandElements`], and the openings
75    /// combining randomness.
76    ///
77    /// See [`GkrRandElements`] for a more detailed description.
78    pub fn new(
79        lagrange: LagrangeKernelRandElements<E>,
80        openings_combining_randomness: Vec<E>,
81    ) -> Self {
82        Self { lagrange, openings_combining_randomness }
83    }
84
85    /// Returns the random elements needed to build the Lagrange kernel column.
86    pub fn lagrange_kernel_rand_elements(&self) -> &LagrangeKernelRandElements<E> {
87        &self.lagrange
88    }
89
90    /// Returns the random values used to linearly combine the openings returned from the GKR proof.
91    pub fn openings_combining_randomness(&self) -> &[E] {
92        &self.openings_combining_randomness
93    }
94}

96/// A trait for verifying a GKR proof.
97///
98/// Specifically, the use case in mind is proving the constraints of a LogUp bus using GKR, as
99/// described in [Improving logarithmic derivative lookups using
100/// GKR](https://eprint.iacr.org/2023/1284.pdf).
101pub trait GkrVerifier {
102    /// The GKR proof.
103    type GkrProof: Deserializable;
104    /// The error that can occur during GKR proof verification.
105    type Error: ToString;
106
107    /// Verifies the GKR proof, and returns the random elements that were used in building
108    /// the Lagrange kernel auxiliary column.
109    fn verify<E, Hasher>(
110        &self,
111        gkr_proof: Self::GkrProof,
112        public_coin: &mut impl RandomCoin<BaseField = E::BaseField, Hasher = Hasher>,
113    ) -> Result<GkrRandElements<E>, Self::Error>
114    where
115        E: FieldElement,
116        Hasher: ElementHasher<BaseField = E::BaseField>;
117}
118
119impl GkrVerifier for () {
120    type GkrProof = ();
121    type Error = RandomCoinError;
122
123    fn verify<E, Hasher>(
124        &self,
125        _gkr_proof: Self::GkrProof,
126        _public_coin: &mut impl RandomCoin<BaseField = E::BaseField, Hasher = Hasher>,
127    ) -> Result<GkrRandElements<E>, Self::Error>
128    where
129        E: FieldElement,
130        Hasher: ElementHasher<BaseField = E::BaseField>,
131    {
132        Ok(GkrRandElements::new(LagrangeKernelRandElements::default(), Vec::new()))
133    }
134}
