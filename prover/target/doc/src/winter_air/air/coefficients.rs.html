10// CONSTRAINT COMPOSITION COEFFICIENTS
11// ================================================================================================
12/// Coefficients used in construction of constraint composition polynomial.
13///
14/// These coefficients are created by the
15/// [Air::get_constraint_composition_coefficients()](crate::Air::get_constraint_composition_coefficients)
16/// function. In the interactive version of the protocol, the verifier draws these coefficients
17/// uniformly at random from the extension field of the protocol.
18///
19/// There is one coefficient for each constraint so that we can compute a random linear
20/// combination of constraints as:
21/// $$
22/// \sum_{i = 0}^k{\alpha_i \cdot C_i(x)}
23/// $$
24/// where:
25/// * $\alpha_i$ is the coefficient for the $i$th constraint.
26/// * $C_i(x)$ is an evaluation of the $i$th constraint at $x$.
27///
28/// The coefficients are separated into two lists: one for transition constraints and another one
29/// for boundary constraints. This separation is done for convenience only.
30#[derive(Debug, Clone)]
31pub struct ConstraintCompositionCoefficients<E: FieldElement> {
32    pub transition: Vec<E>,
33    pub boundary: Vec<E>,
34    pub lagrange: Option<LagrangeConstraintsCompositionCoefficients<E>>,
35}
36
37/// Stores the constraint composition coefficients for the Lagrange kernel transition and boundary
38/// constraints.
39#[derive(Debug, Clone)]
40pub struct LagrangeConstraintsCompositionCoefficients<E: FieldElement> {
41    pub transition: Vec<E>,
42    pub boundary: E,
43}
44// DEEP COMPOSITION COEFFICIENTS
45// ================================================================================================
46/// Coefficients used in construction of DEEP composition polynomial.
47///
48/// These coefficients are created by the
49/// [Air::get_deep_composition_coefficients()](crate::Air::get_deep_composition_coefficients)
50/// function. In the interactive version of the protocol, the verifier draws these coefficients
51/// uniformly at random from the extension field of the protocol.
52///
53/// The coefficients are used in computing the DEEP composition polynomial as:
54/// $$
55/// Y(x) = \sum_{i=0}^k{(
56///     \alpha_i \cdot (\frac{T_i(x) - T_i(z)}{x - z} +
57///     \frac{T_i(x) - T_i(z \cdot g)}{x - z \cdot g})
58/// )} + \sum_{j=0}^m{\beta_j \cdot \frac{H_j(x) - H_j(z)}{x - z}}
59/// $$
60/// where:
61/// * $z$ is an out-of-domain point drawn randomly from the entire field. In the interactive
62///   version of the protocol, $z$ is provided by the verifier.
63/// * $g$ is the generator of the trace domain. This is the $n$th root of unity where
64///   $n$ is the length of the execution trace.
65/// * $T_i(x)$ is an evaluation of the $i$th trace polynomial at $x$, and $k$ is the total
66///   number of trace polynomials (which is equal to the width of the execution trace).
67/// * $H_i(x)$ is an evaluation of the $j$th constraint composition column polynomial at $x$,
68///   and $m$ is the total number of column polynomials.
69/// * $\alpha_i$ is a composition coefficient for the $i$th trace polynomial.
70/// * $\beta_j$ is a composition coefficient for the $j$th constraint column polynomial.
71///
72/// The soundness of the resulting protocol with batching as above is given in Theorem 8 in
73/// https://eprint.iacr.org/2022/1216 and it relies on two points:
74///
75/// 1. The evaluation proofs for each trace polynomial at $z$ and $g \cdot z$ can be batched using
76///    the non-normalized Lagrange kernel over the set $\{z, g \cdot z\}$. This, however, requires
77///    that the FRI protocol is run with a larger agreement parameter
78///    $\alpha^{+} = (1 + 1/2m)\cdot\sqrt{\rho^{+}}$ where $\rho^{+} := \frac{\kappa + 2}{\nu}$,
79///    $\kappa$ and $\nu$ are the length of the execution trace and the LDE domain size,
80///    respectively.
81/// 2. The resulting $Y(x)$ do not need to be degree adjusted but the soundness error of the
82///    protocol needs to be updated. For most combinations of batching parameters, this leads to a
83///    negligible increase in soundness error. The formula for the updated error can be found in
84///    Theorem 8 of https://eprint.iacr.org/2022/1216.
85///
86/// In the case when the trace polynomials contain a trace polynomial corresponding to a Lagrange
87/// kernel column, the above expression of $Y(x)$ includes the additional term given by
88///
89/// $$
90/// \gamma \cdot \frac{T_l(x) - p_S(x)}{Z_S(x)}
91/// $$
92///
93/// where:
94///
95/// 1. $\gamma$ is the composition coefficient for the Lagrange kernel trace polynomial.
96/// 2. $T_l(x) is the evaluation of the Lagrange trace polynomial at $x$.
97/// 3. $S$ is the set of opening points for the Lagrange kernel i.e.,
98///    $S := {z, z.g, z.g^2, ..., z.g^{2^{log_2(\nu) - 1}}}$.
99/// 4. $p_S(X)$ is the polynomial of minimal degree interpolating the set ${(a, T_l(a)): a \in S}$.
100/// 5. $Z_S(X)$ is the polynomial of minimal degree vanishing over the set $S$.
101///
102/// Note that, if a Lagrange kernel trace polynomial is present, then $\rho^{+}$ from above should
103/// be updated to be $\rho^{+} := \frac{\kappa + log_2(\nu) + 1}{\nu}$.
104#[derive(Debug, Clone)]
105pub struct DeepCompositionCoefficients<E: FieldElement> {
106    /// Trace polynomial composition coefficients $\alpha_i$.
107    pub trace: Vec<E>,
108    /// Constraint column polynomial composition coefficients $\beta_j$.
109    pub constraints: Vec<E>,
110    /// Lagrange kernel trace polynomial composition coefficient $\gamma$.
111    pub lagrange: Option<E>,
112}
