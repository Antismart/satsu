32/// Defines an extension field for the composition polynomial.
33///
34/// Choice of a field for a composition polynomial may impact proof soundness, and can also have
35/// a non-negligible impact on proof generation time and proof size. Specifically, for small
36/// fields, security offered by the base field itself may be inadequate or insufficient, and an
37/// extension of the base field may need to be used.
38///
39/// For example, if the size of base field is ~64-bits, a quadratic extension must be use to
40/// achieve ~100 bits of soundness, and a cubic extension must be used to achieve 128+ bits
41/// of soundness.
42///
43/// However, increasing extension degree will increase proof generation time and proof size by
44/// as much as 50%.
45#[repr(u8)]
46#[derive(Debug, Copy, Clone, Eq, PartialEq)]
47pub enum FieldExtension {
48    /// Composition polynomial is constructed in the base field.
49    None = 1,
50    /// Composition polynomial is constructed in the quadratic extension of the base field.
51    Quadratic = 2,
52    /// Composition polynomial is constructed in the cubic extension of the base field.
53    Cubic = 3,
54}
55
56/// STARK protocol parameters.
57///
58/// These parameters have a direct impact on proof soundness, proof generation time, and proof
59/// size. Specifically:
60///
61/// 1. Finite field - proof soundness depends on the size of finite field used by the protocol.
62///    This means, that for small fields (e.g. smaller than ~128 bits), field extensions must be
63///    used to achieve adequate security. And even for ~128 bit fields, to achieve security over
64///    100 bits, a field extension may be required.
65/// 2. Number of queries - higher values increase proof soundness, but also increase proof size.
66/// 3. Blowup factor - higher values increase proof soundness, but also increase proof generation
67///    time and proof size. However, higher blowup factors require fewer queries for the same
68///    security level. Thus, it is frequently possible to increase blowup factor and at the same
69///    time decrease the number of queries in such a way that the proofs become smaller.
70/// 4. Grinding factor - higher values increase proof soundness, but also may increase proof
71///    generation time. More precisely, conjectured proof soundness is bounded by
72///    `num_queries * log2(blowup_factor) + grinding_factor`.
73///
74/// Another important parameter in defining STARK security level, which is not a part of [ProofOptions]
75/// is the hash function used in the protocol. The soundness of a STARK proof is limited by the
76/// collision resistance of the hash function used by the protocol. For example, if a hash function
77/// with 128-bit collision resistance is used, soundness of a STARK proof cannot exceed 128 bits.
78///
79/// In addition, partition options (see [PartitionOptions]) can be provided to split traces during
80/// proving and distribute work across multiple devices. Taking the main segment trace as an example,
81/// the prover will split the main segment trace into `num_partitions` parts, and then proceed to hash
82/// each part row-wise resulting in `num_partitions` digests per row of the trace. Finally,
83/// `num_partitions` digests (per row) are combined into one digest (per row) and at this point
84/// a vector commitment scheme can be called. In the case when `num_partitions` is equal to `1` (default)
85/// the prover will hash each row in one go producing one digest per row of the trace.
86#[derive(Debug, Clone, Eq, PartialEq)]
87pub struct ProofOptions {
88    num_queries: u8,
89    blowup_factor: u8,
90    grinding_factor: u8,
91    field_extension: FieldExtension,
92    fri_folding_factor: u8,
93    fri_remainder_max_degree: u8,
94    partition_options: PartitionOptions,
95}

97// PROOF OPTIONS IMPLEMENTATION
98// ================================================================================================
99impl ProofOptions {
100    // CONSTANTS
101    // --------------------------------------------------------------------------------------------
102
103    /// Smallest allowed blowup factor which is currently set to 2.
104    ///
105    /// The smallest allowed blowup factor for a given computation is derived from degrees of
106    /// constraints defined for that computation and may be greater than 2. But no computation may
107    /// have a blowup factor smaller than 2.
108    pub const MIN_BLOWUP_FACTOR: usize = MIN_BLOWUP_FACTOR;
109
110    // CONSTRUCTORS
111    // --------------------------------------------------------------------------------------------
112    /// Returns a new instance of [ProofOptions] struct constructed from the specified parameters.
113    ///
114    /// # Panics
115    /// Panics if:
116    /// - `num_queries` is zero or greater than 255.
117    /// - `blowup_factor` is smaller than 2, greater than 128, or is not a power of two.
118    /// - `grinding_factor` is greater than 32.
119    /// - `fri_folding_factor` is not 2, 4, 8, or 16.
120    /// - `fri_remainder_max_degree` is greater than 255 or is not a power of two minus 1.
121    pub const fn new(
122        num_queries: usize,
123        blowup_factor: usize,
124        grinding_factor: u32,
125        field_extension: FieldExtension,
126        fri_folding_factor: usize,
127        fri_remainder_max_degree: usize,
128    ) -> ProofOptions {
129        // TODO: return errors instead of panicking
130        assert!(num_queries > 0, "number of queries must be greater than 0");
131        assert!(num_queries <= MAX_NUM_QUERIES, "number of queries cannot be greater than 255");
132
133        assert!(blowup_factor.is_power_of_two(), "blowup factor must be a power of 2");
134        assert!(blowup_factor >= MIN_BLOWUP_FACTOR, "blowup factor cannot be smaller than 2");
135        assert!(blowup_factor <= MAX_BLOWUP_FACTOR, "blowup factor cannot be greater than 128");
136
137        assert!(
138            grinding_factor <= MAX_GRINDING_FACTOR,
139            "grinding factor cannot be greater than 32"
140        );
141
142        assert!(fri_folding_factor.is_power_of_two(), "FRI folding factor must be a power of 2");
143        assert!(
144            fri_folding_factor >= FRI_MIN_FOLDING_FACTOR,
145            "FRI folding factor cannot be smaller than 2"
146        );
147        assert!(
148            fri_folding_factor <= FRI_MAX_FOLDING_FACTOR,
149            "FRI folding factor cannot be greater than 16"
150        );
151
152        assert!(
153            (fri_remainder_max_degree + 1).is_power_of_two(),
154            "FRI polynomial remainder degree must be one less than a power of two"
155        );
156        assert!(
157            fri_remainder_max_degree <= FRI_MAX_REMAINDER_DEGREE,
158            "FRI polynomial remainder degree cannot be greater than 255"
159        );
160
161        Self {
162            num_queries: num_queries as u8,
163            blowup_factor: blowup_factor as u8,
164            grinding_factor: grinding_factor as u8,
165            field_extension,
166            fri_folding_factor: fri_folding_factor as u8,
167            fri_remainder_max_degree: fri_remainder_max_degree as u8,
168            partition_options: PartitionOptions::new(1, 1),
169        }
170    }
171
172    /// Updates the provided [ProofOptions] instance with the specified partition parameters.
173    ///
174    /// # Panics
175    /// Panics if:
176    /// - `num_partitions` is zero or greater than 16.
177    /// - `hash_rate` is zero or greater than 256.
178    pub const fn with_partitions(
179        mut self,
180        num_partitions: usize,
181        hash_rate: usize,
182    ) -> ProofOptions {
183        self.partition_options = PartitionOptions::new(num_partitions, hash_rate);
184
185        self
186    }
187
188    // PUBLIC ACCESSORS
189    // --------------------------------------------------------------------------------------------
190
191    /// Returns number of queries for a STARK proof.
192    ///
193    /// This directly impacts proof soundness as each additional query adds roughly
194    /// `log2(blowup_factor)` bits of security to a proof. However, each additional query also
195    /// increases proof size.
196    pub const fn num_queries(&self) -> usize {
197        self.num_queries as usize
198    }
199
200    /// Returns trace blowup factor for a STARK proof.
201    ///
202    /// This is the factor by which the execution trace is extended during low-degree extension. It
203    /// has a direct impact on proof soundness as each query adds roughly `log2(blowup_factor)`
204    /// bits of security to a proof. However, higher blowup factors also increases prover runtime,
205    /// and may increase proof size.
206    pub const fn blowup_factor(&self) -> usize {
207        self.blowup_factor as usize
208    }
209
210    /// Returns query seed grinding factor for a STARK proof.
211    ///
212    /// Grinding applies Proof-of-Work to the query position seed. An honest prover needs to
213    /// perform this work only once, while a dishonest prover will need to perform it every time
214    /// they try to change a commitment. Thus, higher grinding factor makes it more difficult to
215    /// forge a STARK proof. However, setting grinding factor too high (e.g. higher than 20) will
216    /// adversely affect prover time.
217    pub const fn grinding_factor(&self) -> u32 {
218        self.grinding_factor as u32
219    }
220
221    /// Specifies whether composition polynomial should be constructed in an extension field
222    /// of STARK protocol.
223    ///
224    /// Using a field extension increases maximum security level of a proof, but also has
225    /// non-negligible impact on prover performance.
226    pub const fn field_extension(&self) -> FieldExtension {
227        self.field_extension
228    }
229
230    /// Returns the offset by which the low-degree extension domain is shifted in relation to the
231    /// trace domain.
232    ///
233    /// Currently, this is hard-coded to the primitive element of the underlying base field.
234    pub const fn domain_offset<B: StarkField>(&self) -> B {
235        B::GENERATOR
236    }
237
238    /// Returns options for FRI protocol instantiated with parameters from this proof options.
239    pub fn to_fri_options(&self) -> FriOptions {
240        let folding_factor = self.fri_folding_factor as usize;
241        let remainder_max_degree = self.fri_remainder_max_degree as usize;
242        FriOptions::new(self.blowup_factor(), folding_factor, remainder_max_degree)
243    }
244
245    /// Returns the `[PartitionOptions]` used in this instance of proof options.
246    pub fn partition_options(&self) -> PartitionOptions {
247        self.partition_options
248    }
249}
250
251impl<E: StarkField> ToElements<E> for ProofOptions {
252    fn to_elements(&self) -> Vec<E> {
253        // encode field extension and FRI parameters into a single field element
254        let mut buf = self.field_extension as u32;
255        buf = (buf << 8) | self.fri_folding_factor as u32;
256        buf = (buf << 8) | self.fri_remainder_max_degree as u32;
257
258        vec![
259            E::from(buf),
260            E::from(self.grinding_factor),
261            E::from(self.blowup_factor),
262            E::from(self.num_queries),
263        ]
264    }
265}

267impl Serializable for ProofOptions {
268    /// Serializes `self` and writes the resulting bytes into the `target`.
269    fn write_into<W: ByteWriter>(&self, target: &mut W) {
270        target.write_u8(self.num_queries);
271        target.write_u8(self.blowup_factor);
272        target.write_u8(self.grinding_factor);
273        target.write(self.field_extension);
274        target.write_u8(self.fri_folding_factor);
275        target.write_u8(self.fri_remainder_max_degree);
276        target.write_u8(self.partition_options.num_partitions);
277        target.write_u8(self.partition_options.hash_rate);
278    }
279}
280
281impl Deserializable for ProofOptions {
282    /// Reads proof options from the specified `source` and returns the result.
283    ///
284    /// # Errors
285    /// Returns an error of a valid proof options could not be read from the specified `source`.
286    fn read_from<R: ByteReader>(source: &mut R) -> Result<Self, DeserializationError> {
287        let result = ProofOptions::new(
288            source.read_u8()? as usize,
289            source.read_u8()? as usize,
290            source.read_u8()? as u32,
291            FieldExtension::read_from(source)?,
292            source.read_u8()? as usize,
293            source.read_u8()? as usize,
294        );
295        Ok(result.with_partitions(source.read_u8()? as usize, source.read_u8()? as usize))
296    }
297}

302impl FieldExtension {
303    /// Returns `true` if this field extension is set to `None`.
304    pub const fn is_none(&self) -> bool {
305        matches!(self, Self::None)
306    }
307
308    /// Returns extension degree of this field extension.
309    pub const fn degree(&self) -> u32 {
310        match self {
311            Self::None => 1,
312            Self::Quadratic => 2,
313            Self::Cubic => 3,
314        }
315    }
316}
317
318impl Serializable for FieldExtension {
319    /// Serializes `self` and writes the resulting bytes into the `target`.
320    fn write_into<W: ByteWriter>(&self, target: &mut W) {
321        target.write_u8(*self as u8);
322    }
323
324    /// Returns an estimate of how many bytes are needed to represent self.
325    fn get_size_hint(&self) -> usize {
326        1
327    }
328}
329
330impl Deserializable for FieldExtension {
331    /// Reads a field extension enum from the specified `source`.
332    fn read_from<R: ByteReader>(source: &mut R) -> Result<Self, DeserializationError> {
333        match source.read_u8()? {
334            1 => Ok(FieldExtension::None),
335            2 => Ok(FieldExtension::Quadratic),
336            3 => Ok(FieldExtension::Cubic),
337            value => Err(DeserializationError::InvalidValue(format!(
338                "value {value} cannot be deserialized as FieldExtension enum"
339            ))),
340        }
341    }
342}

347/// Defines the parameters used to calculate partition size when committing to the traces
348/// generated during the protocol.
349///
350/// Using multiple partitions will change how vector commitments are calculated:
351/// - Input matrix columns are split into at most num_partitions partitions
352/// - For each matrix row, a hash is calculated for each partition separately
353/// - The results are merged together by one more hash iteration
354///
355/// This is especially useful when proving with multiple GPU cards where each device holds
356/// a subset of data and allows less data reshuffling when generating commitments.
357///
358/// Hash_rate parameter is used to find the optimal partition size to minimize the number
359/// of hash iterations. It specifies how many field elements are consumed by each hash iteration.
360#[derive(Debug, Clone, Copy, Eq, PartialEq)]
361pub struct PartitionOptions {
362    num_partitions: u8,
363    hash_rate: u8,
364}
365
366impl PartitionOptions {
367    /// Returns a new instance of `[PartitionOptions]`.
368    pub const fn new(num_partitions: usize, hash_rate: usize) -> Self {
369        assert!(num_partitions >= 1, "number of partitions must be greater than or equal to 1");
370        assert!(num_partitions <= 16, "number of partitions must be smaller than or equal to 16");
371
372        assert!(hash_rate >= 1, "hash rate must be greater than or equal to 1");
373        assert!(hash_rate <= 256, "hash rate must be smaller than or equal to 256");
374
375        Self {
376            num_partitions: num_partitions as u8,
377            hash_rate: hash_rate as u8,
378        }
379    }
380
381    /// Returns the size of each partition used when committing to the main and auxiliary traces as
382    /// well as the constraint evaluation trace.
383    /// The returned size is given in terms of number of columns in the field `E`.
384    pub fn partition_size<E: FieldElement>(&self, num_columns: usize) -> usize {
385        if self.num_partitions == 1 {
386            return num_columns;
387        }
388
389        // Don't separate columns that would fit inside one hash iteration. min_partition_size is
390        // the number of `E` elements that can be consumed in one hash iteration.
391        let min_partition_size = self.hash_rate as usize / E::EXTENSION_DEGREE;
392
393        cmp::max(num_columns.div_ceil(self.num_partitions as usize), min_partition_size)
394    }
395
396    /// The actual number of partitions, after the min partition size implied
397    /// by the hash rate is taken into account.
398    pub fn num_partitions<E: FieldElement>(&self, num_columns: usize) -> usize {
399        num_columns.div_ceil(self.partition_size::<E>(num_columns))
400    }
401}
402
403impl Default for PartitionOptions {
404    fn default() -> Self {
405        Self { num_partitions: 1, hash_rate: 1 }
406    }
407}

412#[cfg(test)]
413mod tests {
414    use math::fields::{f64::BaseElement, CubeExtension};
415
416    use super::{FieldExtension, PartitionOptions, ProofOptions, ToElements};
417
418    #[test]
419    fn proof_options_to_elements() {
420        let field_extension = FieldExtension::None;
421        let fri_folding_factor = 8;
422        let fri_remainder_max_degree = 127;
423        let grinding_factor = 20;
424        let blowup_factor = 8;
425        let num_queries = 30;
426
427        let ext_fri = u32::from_le_bytes([
428            fri_remainder_max_degree,
429            fri_folding_factor,
430            field_extension as u8,
431            0,
432        ]);
433        let expected = vec![
434            BaseElement::from(ext_fri),
435            BaseElement::from(grinding_factor),
436            BaseElement::from(blowup_factor as u32),
437            BaseElement::from(num_queries as u32),
438        ];
439
440        let options = ProofOptions::new(
441            num_queries,
442            blowup_factor,
443            grinding_factor,
444            field_extension,
445            fri_folding_factor as usize,
446            fri_remainder_max_degree as usize,
447        );
448        assert_eq!(expected, options.to_elements());
449    }
450
451    #[test]
452    fn correct_partition_sizes() {
453        type E1 = BaseElement;
454        type E3 = CubeExtension<BaseElement>;
455
456        let options = PartitionOptions::new(4, 8);
457        let columns = 7;
458        assert_eq!(8, options.partition_size::<E1>(columns));
459        assert_eq!(1, options.num_partitions::<E1>(columns));
460
461        let options = PartitionOptions::new(4, 8);
462        let columns = 70;
463        assert_eq!(18, options.partition_size::<E1>(columns));
464        assert_eq!(4, options.num_partitions::<E1>(columns));
465
466        let options = PartitionOptions::new(2, 8);
467        let columns = 7;
468        assert_eq!(4, options.partition_size::<E3>(columns));
469        assert_eq!(2, options.num_partitions::<E3>(columns));
470
471        let options: PartitionOptions = PartitionOptions::new(4, 8);
472        let columns = 7;
473        assert_eq!(2, options.partition_size::<E3>(columns));
474        assert_eq!(4, options.num_partitions::<E3>(columns));
475
476        // don't use all partitions if it would result in sizes smaller than
477        // a single hash iteration can handle
478        let options: PartitionOptions = PartitionOptions::new(4, 8);
479        let columns = 3;
480        assert_eq!(2, options.partition_size::<E3>(columns));
481        assert_eq!(2, options.num_partitions::<E3>(columns));
482    }
