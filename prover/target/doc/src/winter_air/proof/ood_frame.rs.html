16// OUT-OF-DOMAIN FRAME
17// ================================================================================================
18
19/// Trace and constraint polynomial evaluations at an out-of-domain point.
20///
21/// This struct contains the following evaluations:
22/// * Evaluations of all trace polynomials at *z*.
23/// * Evaluations of all trace polynomials at *z * g*.
24/// * Evaluations of Lagrange kernel trace polynomial (if any) at *z*, *z * g*, *z * g^2*, ...,
25///   *z * g^(2^(v-1))*, where `v == log(trace_len)`
26/// * Evaluations of constraint composition column polynomials at *z*.
27///
28/// where *z* is an out-of-domain point and *g* is the generator of the trace domain.
29///
30/// Internally, the evaluations are stored as a sequence of bytes. Thus, to retrieve the
31/// evaluations, [parse()](OodFrame::parse) function should be used.
32#[derive(Clone, Debug, Default, Eq, PartialEq)]
33pub struct OodFrame {
34    trace_states: Vec<u8>,
35    lagrange_kernel_trace_states: Vec<u8>,
36    evaluations: Vec<u8>,
37}
38
39impl OodFrame {
40    // UPDATERS
41    // --------------------------------------------------------------------------------------------
42
43    /// Updates the trace state portion of this out-of-domain frame, and returns the hash of the
44    /// trace states.
45    ///
46    /// The out-of-domain frame is stored as one vector of interleaved values, one from the current
47    /// row and the other from the next row. Given the input frame
48    ///
49    ///    +-------+-------+-------+-------+-------+-------+-------+-------+
50    ///    |   a1  |   a2  |  ...  |  an   |  c1   |  c2   |  ...  |  cm   |
51    ///    +-------+-------+-------+-------+-------+-------+-------+-------+
52    ///    |   b1  |   b2  |  ...  |  bn   |  d1   |  d2   |  ...  |  dm   |
53    ///    +-------+-------+-------+-------+-------+-------+-------+-------+
54    ///
55    /// with n being the main trace width and m the auxiliary trace width, the values are stored as
56    ///
57    /// [a1, b1, a2, b2, ..., an, bn, c1, d1, c2, d2, ..., cm, dm]
58    ///
59    /// into `Self::trace_states` (as byte values).
60    ///
61    /// # Panics
62    /// Panics if evaluation frame has already been set.
63    pub fn set_trace_states<E, H>(&mut self, trace_ood_frame: &TraceOodFrame<E>) -> H::Digest
64    where
65        E: FieldElement,
66        H: ElementHasher<BaseField = E::BaseField>,
67    {
68        assert!(self.trace_states.is_empty(), "trace sates have already been set");
69
70        // save the evaluations with the current and next evaluations interleaved for each polynomial
71        let (main_and_aux_trace_states, lagrange_trace_states) = trace_ood_frame.to_trace_states();
72
73        // there are 2 frames: current and next
74        let frame_size: u8 = 2;
75        self.trace_states.write_u8(frame_size);
76        self.trace_states.write_many(&main_and_aux_trace_states);
77
78        // save the Lagrange kernel evaluation frame (if any)
79        {
80            // trace states length will be smaller than u8::MAX, since it is `== log2(trace_len) + 1`
81            debug_assert!(lagrange_trace_states.len() < u8::MAX.into());
82            self.lagrange_kernel_trace_states.write_u8(lagrange_trace_states.len() as u8);
83            self.lagrange_kernel_trace_states.write_many(&lagrange_trace_states);
84        };
85
86        let elements_to_hash: Vec<E> =
87            main_and_aux_trace_states.into_iter().chain(lagrange_trace_states).collect();
88
89        H::hash_elements(&elements_to_hash)
90    }
91
92    /// Updates constraint evaluation portion of this out-of-domain frame.
93    ///
94    /// # Panics
95    /// Panics if:
96    /// * Constraint evaluations have already been set.
97    /// * `evaluations` is an empty vector.
98    pub fn set_constraint_evaluations<E: FieldElement>(&mut self, evaluations: &[E]) {
99        assert!(self.evaluations.is_empty(), "constraint evaluations have already been set");
100        assert!(!evaluations.is_empty(), "cannot set to empty constraint evaluations");
101        self.evaluations.write_many(evaluations);
102    }
103
104    // PARSER
105    // --------------------------------------------------------------------------------------------
106    /// Returns an out-of-domain trace frame and a vector of out-of-domain constraint evaluations
107    /// contained in `self`.
108    ///
109    /// # Panics
110    /// Panics if either `main_trace_width` or `num_evaluations` are equal to zero.
111    ///
112    /// # Errors
113    /// Returns an error if:
114    /// * Valid [`crate::EvaluationFrame`]s for the specified `main_trace_width` and
115    ///   `aux_trace_width` could not be parsed from the internal bytes.
116    /// * A vector of evaluations specified by `num_evaluations` could not be parsed from the
117    ///   internal bytes.
118    /// * Any unconsumed bytes remained after the parsing was complete.
119    pub fn parse<E: FieldElement>(
120        self,
121        main_trace_width: usize,
122        aux_trace_width: usize,
123        num_evaluations: usize,
124    ) -> Result<(TraceOodFrame<E>, Vec<E>), DeserializationError> {
125        assert!(main_trace_width > 0, "trace width cannot be zero");
126        assert!(num_evaluations > 0, "number of evaluations cannot be zero");
127
128        // parse Lagrange kernel column trace, if any
129        let mut reader = SliceReader::new(&self.lagrange_kernel_trace_states);
130        let lagrange_kernel_frame_size = reader.read_u8()? as usize;
131        let lagrange_kernel_frame = if lagrange_kernel_frame_size > 0 {
132            let lagrange_kernel_trace = reader.read_many(lagrange_kernel_frame_size)?;
133
134            Some(LagrangeKernelEvaluationFrame::new(lagrange_kernel_trace))
135        } else {
136            None
137        };
138
139        // if there is a Lagrange kernel, we treat its associated entries separately above
140        let aux_trace_width = aux_trace_width - (lagrange_kernel_frame.is_some() as usize);
141
142        // parse main and auxiliary trace evaluation frames. This does the reverse operation done in
143        // `set_trace_states()`.
144        let (current_row, next_row) = {
145            let mut reader = SliceReader::new(&self.trace_states);
146            let frame_size = reader.read_u8()? as usize;
147            let trace = reader.read_many((main_trace_width + aux_trace_width) * frame_size)?;
148
149            if reader.has_more_bytes() {
150                return Err(DeserializationError::UnconsumedBytes);
151            }
152
153            let mut current_row = Vec::with_capacity(main_trace_width);
154            let mut next_row = Vec::with_capacity(main_trace_width);
155
156            for col in trace.chunks_exact(2) {
157                current_row.push(col[0]);
158                next_row.push(col[1]);
159            }
160
161            (current_row, next_row)
162        };
163
164        // parse the constraint evaluations
165        let mut reader = SliceReader::new(&self.evaluations);
166        let evaluations = reader.read_many(num_evaluations)?;
167        if reader.has_more_bytes() {
168            return Err(DeserializationError::UnconsumedBytes);
169        }
170
171        Ok((
172            TraceOodFrame::new(current_row, next_row, main_trace_width, lagrange_kernel_frame),
173            evaluations,
174        ))
175    }
176}

178// SERIALIZATION
179// ================================================================================================
180
181impl Serializable for OodFrame {
182    /// Serializes `self` and writes the resulting bytes into the `target`.
183    fn write_into<W: ByteWriter>(&self, target: &mut W) {
184        // write trace rows
185        target.write_u16(self.trace_states.len() as u16);
186        target.write_bytes(&self.trace_states);
187
188        // write Lagrange kernel column trace rows
189        target.write_u16(self.lagrange_kernel_trace_states.len() as u16);
190        target.write_bytes(&self.lagrange_kernel_trace_states);
191
192        // write constraint evaluations row
193        target.write_u16(self.evaluations.len() as u16);
194        target.write_bytes(&self.evaluations)
195    }
196
197    /// Returns an estimate of how many bytes are needed to represent self.
198    fn get_size_hint(&self) -> usize {
199        self.trace_states.len() + self.evaluations.len() + 4
200    }
201}
202
203impl Deserializable for OodFrame {
204    /// Reads a OOD frame from the specified `source` and returns the result
205    ///
206    /// # Errors
207    /// Returns an error of a valid OOD frame could not be read from the specified `source`.
208    fn read_from<R: ByteReader>(source: &mut R) -> Result<Self, DeserializationError> {
209        // read trace rows
210        let num_trace_state_bytes = source.read_u16()? as usize;
211        let trace_states = source.read_vec(num_trace_state_bytes)?;
212
213        // read Lagrange kernel column trace rows
214        let num_lagrange_state_bytes = source.read_u16()? as usize;
215        let lagrange_kernel_trace_states = source.read_vec(num_lagrange_state_bytes)?;
216
217        // read constraint evaluations row
218        let num_constraint_evaluation_bytes = source.read_u16()? as usize;
219        let evaluations = source.read_vec(num_constraint_evaluation_bytes)?;
220
221        Ok(OodFrame {
222            trace_states,
223            lagrange_kernel_trace_states,
224            evaluations,
225        })
226    }
227}

229// OOD FRAME TRACE STATES
230// ================================================================================================
231
232/// Trace evaluation frame at the out-of-domain point.
233///
234/// Stores the trace evaluations at `z` and `gz`, where `z` is a random Field element in
235/// `current_row` and `next_row`, respectively.
236///
237/// If the Air contains a Lagrange kernel auxiliary column, then that column interpolated polynomial
238/// will be evaluated at `z`, `gz`, `g^2 z`, ... `g^(2^(v-1)) z`, where `v == log(trace_len)`, and
239/// stored in `lagrange_kernel_frame`.
240pub struct TraceOodFrame<E: FieldElement> {
241    current_row: Vec<E>,
242    next_row: Vec<E>,
243    main_trace_width: usize,
244    lagrange_kernel_frame: Option<LagrangeKernelEvaluationFrame<E>>,
245}
246
247impl<E: FieldElement> TraceOodFrame<E> {
248    /// Creates a new [`TraceOodFrame`] from current, next and optionally Lagrange kernel frames.
249    pub fn new(
250        current_row: Vec<E>,
251        next_row: Vec<E>,
252        main_trace_width: usize,
253        lagrange_kernel_frame: Option<LagrangeKernelEvaluationFrame<E>>,
254    ) -> Self {
255        assert_eq!(current_row.len(), next_row.len());
256
257        Self {
258            current_row,
259            next_row,
260            main_trace_width,
261            lagrange_kernel_frame,
262        }
263    }
264
265    /// Returns the number of columns for the current and next frames.
266    pub fn num_columns(&self) -> usize {
267        self.current_row.len()
268    }
269
270    /// Returns the current row, consisting of both main and auxiliary columns.
271    pub fn current_row(&self) -> &[E] {
272        &self.current_row
273    }
274
275    /// Returns the next frame, consisting of both main and auxiliary columns.
276    pub fn next_row(&self) -> &[E] {
277        &self.next_row
278    }
279
280    /// Returns the evaluation frame for the main trace
281    pub fn main_frame(&self) -> EvaluationFrame<E> {
282        let current = self.current_row[0..self.main_trace_width].to_vec();
283        let next = self.next_row[0..self.main_trace_width].to_vec();
284
285        EvaluationFrame::from_rows(current, next)
286    }
287
288    /// Returns the evaluation frame for the auxiliary trace
289    pub fn aux_frame(&self) -> Option<EvaluationFrame<E>> {
290        if self.has_aux_frame() {
291            let current = self.current_row[self.main_trace_width..].to_vec();
292            let next = self.next_row[self.main_trace_width..].to_vec();
293
294            Some(EvaluationFrame::from_rows(current, next))
295        } else {
296            None
297        }
298    }
299
300    /// Hashes the main, auxiliary and Lagrange kernel frame in a manner consistent with
301    /// [`OodFrame::set_trace_states`], with the purpose of reseeding the public coin.
302    pub fn hash<H: ElementHasher<BaseField = E::BaseField>>(&self) -> H::Digest {
303        let (mut trace_states, mut lagrange_trace_states) = self.to_trace_states();
304        trace_states.append(&mut lagrange_trace_states);
305
306        H::hash_elements(&trace_states)
307    }
308
309    /// Returns the Lagrange kernel frame, if any.
310    pub fn lagrange_kernel_frame(&self) -> Option<&LagrangeKernelEvaluationFrame<E>> {
311        self.lagrange_kernel_frame.as_ref()
312    }
313
314    /// Returns true if an auxiliary frame is present
315    fn has_aux_frame(&self) -> bool {
316        self.current_row.len() > self.main_trace_width
317    }
318
319    /// Returns the main/aux frame and Lagrange kernel frame as element vectors. Specifically, the
320    /// main and auxiliary frames are interleaved, as described in [`OodFrame::set_trace_states`].
321    fn to_trace_states(&self) -> (Vec<E>, Vec<E>) {
322        let mut main_and_aux_frame_states = Vec::new();
323        for col in 0..self.current_row.len() {
324            main_and_aux_frame_states.push(self.current_row[col]);
325            main_and_aux_frame_states.push(self.next_row[col]);
326        }
327
328        let lagrange_frame_states = match self.lagrange_kernel_frame {
329            Some(ref lagrange_kernel_frame) => lagrange_kernel_frame.inner().to_vec(),
330            None => Vec::new(),
331        };
332
333        (main_and_aux_frame_states, lagrange_frame_states)
334    }
335}
