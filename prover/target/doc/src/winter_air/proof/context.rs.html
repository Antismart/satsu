13// PROOF CONTEXT
14// ================================================================================================
15/// Basic metadata about a specific execution of a computation.
16#[derive(Debug, Clone, Eq, PartialEq)]
17pub struct Context {
18    trace_info: TraceInfo,
19    field_modulus_bytes: Vec<u8>,
20    options: ProofOptions,
21}
22
23impl Context {
24    // CONSTRUCTOR
25    // --------------------------------------------------------------------------------------------
26    /// Creates a new context for a computation described by the specified field, trace info, and
27    /// proof options.
28    ///
29    /// # Panics
30    /// Panics if either trace length or the LDE domain size implied by the trace length and the
31    /// blowup factor is greater then [u32::MAX].
32    pub fn new<B: StarkField>(trace_info: TraceInfo, options: ProofOptions) -> Self {
33        // TODO: return errors instead of panicking?
34
35        let trace_length = trace_info.length();
36        assert!(trace_length <= u32::MAX as usize, "trace length too big");
37
38        let lde_domain_size = trace_length * options.blowup_factor();
39        assert!(lde_domain_size <= u32::MAX as usize, "LDE domain size too big");
40
41        Context {
42            trace_info,
43            field_modulus_bytes: B::get_modulus_le_bytes(),
44            options,
45        }
46    }
47
48    // PUBLIC ACCESSORS
49    // --------------------------------------------------------------------------------------------
50
51    /// Returns execution trace info for the computation described by this context.
52    pub fn trace_info(&self) -> &TraceInfo {
53        &self.trace_info
54    }
55
56    /// Returns the size of the LDE domain for the computation described by this context.
57    pub fn lde_domain_size(&self) -> usize {
58        self.trace_info.length() * self.options.blowup_factor()
59    }
60
61    /// Returns modulus of the field for the computation described by this context.
62    pub fn field_modulus_bytes(&self) -> &[u8] {
63        &self.field_modulus_bytes
64    }
65
66    /// Returns number of bits in the base field modulus for the computation described by this
67    /// context.
68    ///
69    /// The modulus is assumed to be encoded in little-endian byte order.
70    pub fn num_modulus_bits(&self) -> u32 {
71        let mut num_bits = self.field_modulus_bytes.len() as u32 * 8;
72        for &byte in self.field_modulus_bytes.iter().rev() {
73            if byte != 0 {
74                num_bits -= byte.leading_zeros();
75                return num_bits;
76            }
77            num_bits -= 8;
78        }
79
80        0
81    }
82
83    /// Returns proof options which were used to a proof in this context.
84    pub fn options(&self) -> &ProofOptions {
85        &self.options
86    }
87}
88
89impl<E: StarkField> ToElements<E> for Context {
90    /// Converts this [Context] into a vector of field elements.
91    ///
92    /// The elements are laid out as follows:
93    /// - trace info [2 or more elements].
94    /// - field modulus bytes [2 field elements].
95    /// - field extension and FRI parameters [1 element].
96    /// - grinding factor [1 element].
97    /// - blowup factor [1 element].
98    /// - number of queries [1 element].
99    fn to_elements(&self) -> Vec<E> {
100        // convert trace layout
101        let mut result = self.trace_info.to_elements();
102
103        // convert field modulus bytes into 2 elements
104        let num_modulus_bytes = self.field_modulus_bytes.len();
105        let (m1, m2) = self.field_modulus_bytes.split_at(num_modulus_bytes / 2);
106        result.push(E::from_bytes_with_padding(m1));
107        result.push(E::from_bytes_with_padding(m2));
108
109        // convert proof options to elements
110        result.append(&mut self.options.to_elements());
111
112        result
113    }
114}

116// SERIALIZATION
117// ================================================================================================
118
119impl Serializable for Context {
120    /// Serializes `self` and writes the resulting bytes into the `target`.
121    fn write_into<W: ByteWriter>(&self, target: &mut W) {
122        self.trace_info.write_into(target);
123        assert!(self.field_modulus_bytes.len() < u8::MAX as usize);
124        target.write_u8(self.field_modulus_bytes.len() as u8);
125        target.write_bytes(&self.field_modulus_bytes);
126        self.options.write_into(target);
127    }
128}
129
130impl Deserializable for Context {
131    /// Reads proof context from the specified `source` and returns the result.
132    ///
133    /// # Errors
134    /// Returns an error of a valid Context struct could not be read from the specified `source`.
135    fn read_from<R: ByteReader>(source: &mut R) -> Result<Self, DeserializationError> {
136        // read and validate trace info
137        let trace_info = TraceInfo::read_from(source)?;
138
139        // read and validate field modulus bytes
140        let num_modulus_bytes = source.read_u8()? as usize;
141        if num_modulus_bytes == 0 {
142            return Err(DeserializationError::InvalidValue(
143                "field modulus cannot be an empty value".to_string(),
144            ));
145        }
146        let field_modulus_bytes = source.read_vec(num_modulus_bytes)?;
147
148        // read options
149        let options = ProofOptions::read_from(source)?;
150
151        Ok(Context { trace_info, field_modulus_bytes, options })
152    }
153}

158#[cfg(test)]
159mod tests {
160    use math::fields::f64::BaseElement;
161
162    use super::{Context, ProofOptions, ToElements, TraceInfo};
163    use crate::FieldExtension;
164
165    #[test]
166    fn context_to_elements() {
167        let field_extension = FieldExtension::None;
168        let fri_folding_factor = 8;
169        let fri_remainder_max_degree = 127;
170        let grinding_factor = 20;
171        let blowup_factor = 8;
172        let num_queries = 30;
173
174        let main_width = 20;
175        let aux_width = 9;
176        let aux_rands = 12;
177        let trace_length = 4096;
178
179        let ext_fri = u32::from_le_bytes([
180            fri_remainder_max_degree,
181            fri_folding_factor,
182            field_extension as u8,
183            0,
184        ]);
185
186        let expected = {
187            let trace_info = TraceInfo::new_multi_segment(
188                main_width,
189                aux_width,
190                aux_rands,
191                trace_length,
192                vec![],
193            );
194
195            let mut expected = trace_info.to_elements();
196            expected.extend(vec![
197                BaseElement::from(1_u32),    // lower bits of field modulus
198                BaseElement::from(u32::MAX), // upper bits of field modulus
199                BaseElement::from(ext_fri),
200                BaseElement::from(grinding_factor),
201                BaseElement::from(blowup_factor as u32),
202                BaseElement::from(num_queries as u32),
203            ]);
204
205            expected
206        };
207
208        let options = ProofOptions::new(
209            num_queries,
210            blowup_factor,
211            grinding_factor,
212            field_extension,
213            fri_folding_factor as usize,
214            fri_remainder_max_degree as usize,
215        );
216        let trace_info =
217            TraceInfo::new_multi_segment(main_width, aux_width, aux_rands, trace_length, vec![]);
218        let context = Context::new::<BaseElement>(trace_info, options);
219        assert_eq!(expected, context.to_elements());
220    }
221}
