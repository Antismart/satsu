6//! Basic polynomial operations.
7//!
8//! This module provides a set of function for basic polynomial operations, including:
9//! - Polynomial evaluation using Horner method.
10//! - Polynomial interpolation using Lagrange method.
11//! - Polynomial addition, subtraction, multiplication, and division.
12//! - Synthetic polynomial division for efficient division by polynomials of the form
13//!   `x`^`a` - `b`.
14//!
15//! In the context of this module any slice of field elements is considered to be a polynomial
16//! in reverse coefficient form. A few examples:
17//!
18//! ```
19//! # use winter_math::{fields::{f128::BaseElement}, FieldElement};
20//! // p(x) = 2 * x + 1
21//! let p = vec![BaseElement::new(1), BaseElement::new(2)];
22//!
23//! // p(x) = 4 * x^2 + 3
24//! let p = [BaseElement::new(3), BaseElement::ZERO, BaseElement::new(4)];
25//! ```

40/// Evaluates a polynomial at a single point and returns the result.
41///
42/// Evaluates polynomial `p` at coordinate `x` using
43/// [Horner's method](https://en.wikipedia.org/wiki/Horner%27s_method).
44///
45/// # Examples
46/// ```
47/// # use winter_math::polynom::*;
48/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
49/// // define polynomial: f(x) = 3 * x^2 + 2 * x + 1
50/// let p = (1u32..4).map(BaseElement::from).collect::<Vec<_>>();
51///
52/// // evaluate the polynomial at point 4
53/// let x = BaseElement::new(4);
54/// assert_eq!(BaseElement::new(57), eval(&p, x));
55/// ```
56pub fn eval<B, E>(p: &[B], x: E) -> E
57where
58    B: FieldElement,
59    E: FieldElement + From<B>,
60{
61    // Horner evaluation
62    p.iter().rev().fold(E::ZERO, |acc, &coeff| acc * x + E::from(coeff))
63}
64
65/// Evaluates a polynomial at multiple points and returns a vector of results.
66///
67/// Evaluates polynomial `p` at all coordinates in `xs` slice by repeatedly invoking
68/// `polynom::eval()` function.
69///
70/// # Examples
71/// ```
72/// # use winter_math::polynom::*;
73/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
74/// // define polynomial: f(x) = 3 * x^2 + 2 * x + 1
75/// let p = (1_u32..4).map(BaseElement::from).collect::<Vec<_>>();
76/// let xs = (3_u32..6).map(BaseElement::from).collect::<Vec<_>>();
77///
78/// let expected = xs.iter().map(|x| eval(&p, *x)).collect::<Vec<_>>();
79/// assert_eq!(expected, eval_many(&p, &xs));
80/// ```
81pub fn eval_many<B, E>(p: &[B], xs: &[E]) -> Vec<E>
82where
83    B: FieldElement,
84    E: FieldElement + From<B>,
85{
86    xs.iter().map(|x| eval(p, *x)).collect()
87}

92/// Returns a polynomial in coefficient form interpolated from a set of X and Y coordinates.
93///
94/// Uses [Lagrange interpolation](https://en.wikipedia.org/wiki/Lagrange_polynomial) to build a
95/// polynomial from X and Y coordinates. If `remove_leading_zeros = true`, all leading coefficients
96/// which are ZEROs will be truncated; otherwise, the length of result will be equal to the number
97/// of X coordinates.
98///
99/// # Panics
100/// Panics if number of X and Y coordinates is not the same.
101///
102/// # Example
103/// ```
104/// # use winter_math::polynom::*;
105/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
106/// # use rand_utils::rand_vector;
107/// let xs: Vec<BaseElement> = rand_vector(16);
108/// let ys: Vec<BaseElement> = rand_vector(16);
109///
110/// let p = interpolate(&xs, &ys, false);
111/// assert_eq!(ys, eval_many(&p, &xs));
112/// ```
113pub fn interpolate<E>(xs: &[E], ys: &[E], remove_leading_zeros: bool) -> Vec<E>
114where
115    E: FieldElement,
116{
117    debug_assert!(xs.len() == ys.len(), "number of X and Y coordinates must be the same");
118
119    let roots = poly_from_roots(xs);
120    let numerators: Vec<Vec<E>> = xs.iter().map(|&x| syn_div(&roots, 1, x)).collect();
121
122    let denominators: Vec<E> = numerators.iter().zip(xs).map(|(e, &x)| eval(e, x)).collect();
123    let denominators = batch_inversion(&denominators);
124
125    let mut result = vec![E::ZERO; xs.len()];
126    for i in 0..xs.len() {
127        let y_slice = ys[i] * denominators[i];
128        for (j, res) in result.iter_mut().enumerate() {
129            *res += numerators[i][j] * y_slice;
130        }
131    }
132
133    if remove_leading_zeros {
134        crate::polynom::remove_leading_zeros(&result)
135    } else {
136        result
137    }
138}

140/// Returns a vector of polynomials interpolated from the provided X and Y coordinate batches.
141///
142/// Uses [Lagrange interpolation](https://en.wikipedia.org/wiki/Lagrange_polynomial) to build a
143/// vector of polynomial from X and Y coordinate batches (one polynomial per batch).
144///
145/// When the number of batches is larger, this function is significantly faster than using
146/// `polynom::interpolate()` function individually for each batch of coordinates. The speed-up
147/// is primarily due to computing all inversions as a single batch inversion across all
148/// coordinate batches.
149///
150/// # Panics
151/// Panics if the number of X coordinate batches and Y coordinate batches is not the same.
152///
153/// # Examples
154/// ```
155/// # use winter_math::polynom::*;
156/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
157/// # use rand_utils::rand_array;
158/// let x_batches: Vec<[BaseElement; 8]> = vec![
159///     rand_array(),
160///     rand_array(),
161/// ];
162/// let y_batches: Vec<[BaseElement; 8]> = vec![
163///     rand_array(),
164///     rand_array(),
165/// ];
166///
167/// let polys = interpolate_batch(&x_batches, &y_batches);
168/// for ((p, xs), ys) in polys.iter().zip(x_batches).zip(y_batches) {
169///     assert_eq!(ys.to_vec(), eval_many(p, &xs));
170/// }
171/// ```
172pub fn interpolate_batch<E, const N: usize>(xs: &[[E; N]], ys: &[[E; N]]) -> Vec<[E; N]>
173where
174    E: FieldElement,
175{
176    debug_assert!(
177        xs.len() == ys.len(),
178        "number of X coordinate batches and Y coordinate batches must be the same"
179    );
180
181    let n = xs.len();
182    let mut equations = vec![[E::ZERO; N]; n * N];
183    let mut inverses = vec![E::ZERO; n * N];
184
185    // TODO: converting this to an array results in about 5% speed-up, but unfortunately, complex
186    // generic constraints are not yet supported: https://github.com/rust-lang/rust/issues/76560
187    let mut roots = vec![E::ZERO; N + 1];
188
189    for (i, xs) in xs.iter().enumerate() {
190        fill_zero_roots(xs, &mut roots);
191        for (j, &x) in xs.iter().enumerate() {
192            let equation = &mut equations[i * N + j];
193            // optimized synthetic division for this context
194            equation[N - 1] = roots[N];
195            for k in (0..N - 1).rev() {
196                equation[k] = roots[k + 1] + equation[k + 1] * x;
197            }
198            inverses[i * N + j] = eval(equation, x);
199        }
200    }
201    let equations = group_slice_elements::<[E; N], N>(&equations);
202    let inverses_vec = batch_inversion(&inverses);
203    let inverses = group_slice_elements::<E, N>(&inverses_vec);
204
205    let mut result = vec![[E::ZERO; N]; n];
206    for (i, poly) in result.iter_mut().enumerate() {
207        for j in 0..N {
208            let inv_y = ys[i][j] * inverses[i][j];
209            for (res_coeff, &eq_coeff) in poly.iter_mut().zip(equations[i][j].iter()) {
210                *res_coeff += eq_coeff * inv_y;
211            }
212        }
213    }
214
215    result
216}

221/// Returns a polynomial resulting from adding two polynomials together.
222///
223/// Polynomials `a` and `b` are expected to be in the coefficient form, and the returned
224/// polynomial will be in the coefficient form as well. The length of the returned vector
225/// will be max(a.len(), b.len()).
226///
227/// # Examples
228/// ```
229/// # use winter_math::polynom::*;
230/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
231/// // p1(x) = 4 * x^2 + 3 * x + 2
232/// let p1 = (2_u32..5).map(BaseElement::from).collect::<Vec<_>>();
233/// // p2(x) = 2 * x + 1
234/// let p2 = (1_u32..3).map(BaseElement::from).collect::<Vec<_>>();
235///
236/// // expected result = 4 * x^2 + 5 * x + 3
237/// let expected = vec![
238///     BaseElement::new(3),
239///     BaseElement::new(5),
240///     BaseElement::new(4),
241/// ];
242/// assert_eq!(expected, add(&p1, &p2));
243/// ```
244pub fn add<E>(a: &[E], b: &[E]) -> Vec<E>
245where
246    E: FieldElement,
247{
248    let result_len = core::cmp::max(a.len(), b.len());
249    let mut result = Vec::with_capacity(result_len);
250    for i in 0..result_len {
251        let c1 = if i < a.len() { a[i] } else { E::ZERO };
252        let c2 = if i < b.len() { b[i] } else { E::ZERO };
253        result.push(c1 + c2);
254    }
255    result
256}

258/// Returns a polynomial resulting from subtracting one polynomial from another.
259///
260/// Specifically, subtracts polynomial `b` from polynomial `a` and returns the result. Both
261/// polynomials are expected to be in the coefficient form, and the returned polynomial will
262/// be in the coefficient form as well.  The length of the returned vector will be
263/// max(a.len(), b.len()).
264///
265/// # Examples
266/// ```
267/// # use winter_math::polynom::*;
268/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
269/// // p1(x) = 4 * x^2 + 3 * x + 2
270/// let p1 = (2_u32..5).map(BaseElement::from).collect::<Vec<_>>();
271/// // p2(x) = 2 * x + 1
272/// let p2 = (1_u32..3).map(BaseElement::from).collect::<Vec<_>>();
273///
274/// // expected result = 4 * x^2 + x + 1
275/// let expected = vec![
276///     BaseElement::new(1),
277///     BaseElement::new(1),
278///     BaseElement::new(4),
279/// ];
280/// assert_eq!(expected, sub(&p1, &p2));
281/// ```
282pub fn sub<E>(a: &[E], b: &[E]) -> Vec<E>
283where
284    E: FieldElement,
285{
286    let result_len = core::cmp::max(a.len(), b.len());
287    let mut result = Vec::with_capacity(result_len);
288    for i in 0..result_len {
289        let c1 = if i < a.len() { a[i] } else { E::ZERO };
290        let c2 = if i < b.len() { b[i] } else { E::ZERO };
291        result.push(c1 - c2);
292    }
293    result
294}

296/// Returns a polynomial resulting from multiplying two polynomials together.
297///
298/// Polynomials `a` and `b` are expected to be in the coefficient form, and the returned
299/// polynomial will be in the coefficient form as well. The length of the returned vector
300/// will be a.len() + b.len() - 1.
301///
302/// # Examples
303/// ```
304/// # use winter_math::polynom::*;
305/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
306/// // p1(x) = x + 1
307/// let p1 = [BaseElement::ONE, BaseElement::ONE];
308/// // p2(x) = x^2 + 2
309/// let p2 = [BaseElement::new(2), BaseElement::ZERO, BaseElement::ONE];
310///
311/// // expected result = x^3 + x^2 + 2 * x + 2
312/// let expected = vec![
313///     BaseElement::new(2),
314///     BaseElement::new(2),
315///     BaseElement::new(1),
316///     BaseElement::new(1),
317/// ];
318/// assert_eq!(expected, mul(&p1, &p2));
319/// ```
320pub fn mul<E>(a: &[E], b: &[E]) -> Vec<E>
321where
322    E: FieldElement,
323{
324    let result_len = a.len() + b.len() - 1;
325    let mut result = vec![E::ZERO; result_len];
326    for i in 0..a.len() {
327        for j in 0..b.len() {
328            let s = a[i] * b[j];
329            result[i + j] += s;
330        }
331    }
332    result
333}

335/// Returns a polynomial resulting from multiplying a given polynomial by a scalar value.
336///
337/// Specifically, multiplies every coefficient of polynomial `p` by constant `k` and returns
338/// the resulting vector.
339///
340/// # Examples
341/// ```
342/// # use winter_math::polynom::*;
343/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
344/// let p = [
345///     BaseElement::new(1),
346///     BaseElement::new(2),
347///     BaseElement::new(3),
348/// ];
349/// let k = BaseElement::new(2);
350///
351/// let expected = vec![
352///     BaseElement::new(2),
353///     BaseElement::new(4),
354///     BaseElement::new(6),
355/// ];
356/// assert_eq!(expected, mul_by_scalar(&p, k));
357/// ```
358pub fn mul_by_scalar<E>(p: &[E], k: E) -> Vec<E>
359where
360    E: FieldElement,
361{
362    let mut result = Vec::with_capacity(p.len());
363    for coeff in p {
364        result.push(*coeff * k);
365    }
366    result
367}

369/// Returns a polynomial resulting from dividing one polynomial by another.
370///
371/// Specifically, divides polynomial `a` by polynomial `b` and returns the result. If the
372/// polynomials don't divide evenly, the remainder is ignored. Both polynomials are expected to
373/// be in the coefficient form, and the returned polynomial will be in the coefficient form as
374/// well. The length of the returned vector will be a.len() - b.len() + 1.
375///
376/// # Panics
377/// Panics if:
378/// * Polynomial `b` is empty.
379/// * Degree of polynomial `b` is zero and the constant coefficient is ZERO.
380/// * The degree of polynomial `b` is greater than the degree of polynomial `a`.
381///
382/// # Examples
383/// ```
384/// # use winter_math::polynom::*;
385/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
386/// // p1(x) = x^3 + x^2 + 2 * x + 2
387/// let p1 = [
388///     BaseElement::new(2),
389///     BaseElement::new(2),
390///     BaseElement::new(1),
391///     BaseElement::new(1),
392/// ];
393/// // p2(x) = x^2 + 2
394/// let p2 = [BaseElement::new(2), BaseElement::ZERO, BaseElement::ONE];
395///
396/// // expected result = x + 1
397/// let expected = vec![BaseElement::ONE, BaseElement::ONE];
398/// assert_eq!(expected, div(&p1, &p2));
399/// ```
400pub fn div<E>(a: &[E], b: &[E]) -> Vec<E>
401where
402    E: FieldElement,
403{
404    let mut apos = degree_of(a);
405    let mut a = a.to_vec();
406
407    let bpos = degree_of(b);
408    assert!(apos >= bpos, "cannot divide by polynomial of higher degree");
409    if bpos == 0 {
410        assert!(!b.is_empty(), "cannot divide by empty polynomial");
411        assert!(b[0] != E::ZERO, "cannot divide polynomial by zero");
412    }
413
414    let mut result = vec![E::ZERO; apos - bpos + 1];
415    for i in (0..result.len()).rev() {
416        let quot = a[apos] / b[bpos];
417        result[i] = quot;
418        for j in (0..bpos).rev() {
419            a[i + j] -= b[j] * quot;
420        }
421        apos = apos.wrapping_sub(1);
422    }
423
424    result
425}

427/// Returns a polynomial resulting from dividing a polynomial by a polynomial of special form.
428///
429/// Specifically, divides polynomial `p` by polynomial (x^`a` - `b`) using
430/// [synthetic division](https://en.wikipedia.org/wiki/Synthetic_division) method; if the
431/// polynomials don't divide evenly, the remainder is ignored. Polynomial `p` is expected
432/// to be in the coefficient form, and the result will be in the coefficient form as well.
433/// The length of the resulting polynomial will be equal to `p.len()`.
434///
435/// This function is significantly faster than the generic `polynom::div()` function.
436///
437/// # Panics
438/// Panics if:
439/// * `a` is zero;
440/// * `b` is zero;
441/// * `p.len()` is smaller than or equal to `a`.
442///
443/// # Examples
444/// ```
445/// # use winter_math::polynom::*;
446/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
447/// // p(x) = x^3 + x^2 + 2 * x + 2
448/// let p = [
449///     BaseElement::new(2),
450///     BaseElement::new(2),
451///     BaseElement::new(1),
452///     BaseElement::new(1),
453/// ];
454///
455/// // expected result = x^2 + 2
456/// let expected = vec![
457///     BaseElement::new(2),
458///     BaseElement::ZERO,
459///     BaseElement::new(1),
460///     BaseElement::ZERO,
461/// ];
462///
463/// // divide by x + 1
464/// assert_eq!(expected, syn_div(&p, 1, -BaseElement::ONE));
465/// ```
466pub fn syn_div<E>(p: &[E], a: usize, b: E) -> Vec<E>
467where
468    E: FieldElement,
469{
470    let mut result = p.to_vec();
471    syn_div_in_place(&mut result, a, b);
472    result
473}

475/// Divides a polynomial by a polynomial of special form and saves the result into the original
476/// polynomial.
477///
478/// Specifically, divides polynomial `p` by polynomial (x^`a` - `b`) using
479/// [synthetic division](https://en.wikipedia.org/wiki/Synthetic_division) method and saves the
480/// result into `p`. If the polynomials don't divide evenly, the remainder is ignored. Polynomial
481/// `p` is expected to be in the coefficient form, and the result will be in coefficient form as
482/// well.
483///
484/// This function is significantly faster than the generic `polynom::div()` function, and as
485/// compared to `polynom::syn_div()` function, this function does not allocate any additional
486/// memory.
487///
488/// # Panics
489/// Panics if:
490/// * `a` is zero;
491/// * `b` is zero;
492/// * `p.len()` is smaller than or equal to `a`.
493///
494/// # Examples
495/// ```
496/// # use winter_math::polynom::*;
497/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
498/// // p(x) = x^3 + x^2 + 2 * x + 2
499/// let mut p = [
500///     BaseElement::new(2),
501///     BaseElement::new(2),
502///     BaseElement::new(1),
503///     BaseElement::new(1),
504/// ];
505///
506/// // divide by x + 1
507/// syn_div_in_place(&mut p, 1, -BaseElement::ONE);
508///
509/// // expected result = x^2 + 2
510/// let expected = [
511///     BaseElement::new(2),
512///     BaseElement::ZERO,
513///     BaseElement::new(1),
514///     BaseElement::ZERO,
515/// ];
516///
517/// assert_eq!(expected, p);
518pub fn syn_div_in_place<E>(p: &mut [E], a: usize, b: E)
519where
520    E: FieldElement,
521{
522    assert!(a != 0, "divisor degree cannot be zero");
523    assert!(b != E::ZERO, "constant cannot be zero");
524    assert!(p.len() > a, "divisor degree cannot be greater than dividend size");
525
526    if a == 1 {
527        // if we are dividing by (x - `b`), we can use a single variable to keep track
528        // of the remainder; this way, we can avoid shifting the values in the slice later
529        let mut c = E::ZERO;
530        for coeff in p.iter_mut().rev() {
531            *coeff += b * c;
532            mem::swap(coeff, &mut c);
533        }
534    } else {
535        // if we are dividing by a polynomial of higher power, we need to keep track of the
536        // full remainder. we do that in place, but then need to shift the values at the end
537        // to discard the remainder
538        let degree_offset = p.len() - a;
539        if b == E::ONE {
540            // if `b` is 1, no need to multiply by `b` in every iteration of the loop
541            for i in (0..degree_offset).rev() {
542                p[i] += p[i + a];
543            }
544        } else {
545            for i in (0..degree_offset).rev() {
546                p[i] += p[i + a] * b;
547            }
548        }
549        // discard the remainder
550        p.copy_within(a.., 0);
551        p[degree_offset..].fill(E::ZERO);
552    }
553}

555/// Divides a polynomial by a polynomial given its roots and saves the result into the original
556/// polynomial.
557///
558/// Specifically, divides polynomial `p` by polynomial \prod_{i = 1}^m (x - `x_i`) using
559/// [synthetic division](https://en.wikipedia.org/wiki/Synthetic_division) method and saves the
560/// result into `p`. If the polynomials don't divide evenly, the remainder is ignored. Polynomial
561/// `p` is expected to be in the coefficient form, and the result will be in coefficient form as
562/// well.
563///
564/// This function is significantly faster than the generic `polynom::div()` function, using
565/// the coefficients of the divisor.
566///
567/// # Panics
568/// Panics if:
569/// * `roots.len()` is zero;
570/// * `p.len()` is smaller than or equal to `roots.len()`.
571///
572/// # Examples
573/// ```
574/// # use winter_math::polynom::*;
575/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
576/// // p(x) = x^3 - 7 * x + 6
577/// let mut p = [
578///     BaseElement::new(6),
579///     -BaseElement::new(7),
580///     BaseElement::new(0),
581///     BaseElement::new(1),
582/// ];
583///
584/// // divide by (x - 1) * (x - 2)
585/// let zeros = vec![BaseElement::new(1), BaseElement::new(2)];
586/// syn_div_roots_in_place(&mut p, &zeros);
587///
588/// // expected result = x + 3
589/// let expected = [
590///     BaseElement::new(3),
591///     BaseElement::new(1),
592///     BaseElement::ZERO,
593///     BaseElement::ZERO,
594/// ];
595///
596/// assert_eq!(expected, p);
597pub fn syn_div_roots_in_place<E>(p: &mut [E], roots: &[E])
598where
599    E: FieldElement,
600{
601    assert!(!roots.is_empty(), "divisor should contain at least one linear factor");
602    assert!(p.len() > roots.len(), "divisor degree cannot be greater than dividend size");
603
604    for root in roots {
605        let mut c = E::ZERO;
606        for coeff in p.iter_mut().rev() {
607            *coeff += *root * c;
608            mem::swap(coeff, &mut c);
609        }
610    }
611}

616/// Returns the degree of the provided polynomial.
617///
618/// If the size of the provided slice is much larger than the degree of the polynomial (i.e.,
619/// a large number of leading coefficients is ZERO), this operation can be quite inefficient.
620///
621/// # Examples
622/// ```
623/// # use winter_math::polynom::*;
624/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
625/// assert_eq!(0, degree_of::<BaseElement>(&[]));
626/// assert_eq!(0, degree_of(&[BaseElement::ONE]));
627/// assert_eq!(1, degree_of(&[BaseElement::ONE, BaseElement::new(2)]));
628/// assert_eq!(1, degree_of(&[BaseElement::ONE, BaseElement::new(2), BaseElement::ZERO]));
629/// assert_eq!(2, degree_of(&[BaseElement::ONE, BaseElement::new(2), BaseElement::new(3)]));
630/// assert_eq!(
631///     2,
632///     degree_of(&[
633///         BaseElement::ONE,
634///         BaseElement::new(2),
635///         BaseElement::new(3),
636///         BaseElement::ZERO
637///     ])
638/// );
639/// ```
640pub fn degree_of<E>(poly: &[E]) -> usize
641where
642    E: FieldElement,
643{
644    for i in (0..poly.len()).rev() {
645        if poly[i] != E::ZERO {
646            return i;
647        }
648    }
649    0
650}

652/// Returns a polynomial with all leading ZERO coefficients removed.
653///
654/// # Examples
655/// ```
656/// # use winter_math::polynom::*;
657/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
658/// let a = vec![1u128, 2, 3, 4, 5, 6, 0, 0]
659///     .into_iter()
660///     .map(BaseElement::new)
661///     .collect::<Vec<_>>();
662/// let b = remove_leading_zeros(&a);
663/// assert_eq!(6, b.len());
664/// assert_eq!(a[..6], b);
665///
666/// let a = vec![0u128, 0, 0, 0]
667///     .into_iter()
668///     .map(BaseElement::new)
669///     .collect::<Vec<_>>();
670/// let b = remove_leading_zeros(&a);
671/// assert_eq!(0, b.len());
672/// ```
673pub fn remove_leading_zeros<E>(values: &[E]) -> Vec<E>
674where
675    E: FieldElement,
676{
677    for i in (0..values.len()).rev() {
678        if values[i] != E::ZERO {
679            return values[..(i + 1)].to_vec();
680        }
681    }
682    vec![]
683}

685/// Returns the coefficients of polynomial given its roots.
686///
687/// # Examples
688/// ```
689/// # use winter_math::polynom::*;
690/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
691/// let xs = vec![1u128, 2]
692///     .into_iter()
693///     .map(BaseElement::new)
694///     .collect::<Vec<_>>();
695///
696/// let mut expected_poly = vec![2u128, 3, 1]
697///     .into_iter()
698///     .map(BaseElement::new)
699///     .collect::<Vec<_>>();
700/// expected_poly[1] *= -BaseElement::ONE;
701///
702/// let poly = poly_from_roots(&xs);
703/// assert_eq!(expected_poly, poly);
704/// ```
705pub fn poly_from_roots<E: FieldElement>(xs: &[E]) -> Vec<E> {
706    let mut result = unsafe { utils::uninit_vector(xs.len() + 1) };
707    fill_zero_roots(xs, &mut result);
708    result
709}

711// HELPER FUNCTIONS
712// ================================================================================================
713
714fn fill_zero_roots<E: FieldElement>(xs: &[E], result: &mut [E]) {
715    let mut n = result.len();
716    n -= 1;
717    result[n] = E::ONE;
718
719    for i in 0..xs.len() {
720        n -= 1;
721        result[n] = E::ZERO;
722        #[allow(clippy::assign_op_pattern)]
723        for j in n..xs.len() {
724            result[j] = result[j] - result[j + 1] * xs[i];
725        }
726    }
727}
