16/// Evaluates polynomial `p` in-place over the domain of length `p.len()` in the field specified
17/// by `B` using the FFT algorithm.
18pub fn evaluate_poly<B, E>(p: &mut [E], twiddles: &[B])
19where
20    B: StarkField,
21    E: FieldElement<BaseField = B>,
22{
23    p.fft_in_place(twiddles);
24    p.permute();
25}
26
27/// Evaluates polynomial `p` over the domain of length `p.len()` * `blowup_factor` shifted by
28/// `domain_offset` in the field specified `B` using the FFT algorithm and returns the result.
29pub fn evaluate_poly_with_offset<B, E>(
30    p: &[E],
31    twiddles: &[B],
32    domain_offset: B,
33    blowup_factor: usize,
34) -> Vec<E>
35where
36    B: StarkField,
37    E: FieldElement<BaseField = B>,
38{
39    let domain_size = p.len() * blowup_factor;
40    let g = B::get_root_of_unity(domain_size.ilog2());
41    let mut result = unsafe { uninit_vector(domain_size) };
42
43    result.as_mut_slice().chunks_mut(p.len()).enumerate().for_each(|(i, chunk)| {
44        let idx = super::permute_index(blowup_factor, i) as u64;
45        let offset = g.exp(idx.into()) * domain_offset;
46        let mut factor = E::BaseField::ONE;
47        for (d, c) in chunk.iter_mut().zip(p.iter()) {
48            *d = (*c).mul_base(factor);
49            factor *= offset;
50        }
51        chunk.fft_in_place(twiddles);
52    });
53
54    result.permute();
55    result
56}

58// POLYNOMIAL INTERPOLATION

61/// Interpolates `evaluations` over a domain of length `evaluations.len()` in the field specified
62/// `B` into a polynomial in coefficient form using the FFT algorithm.
63///
64/// # Panics
65/// Panics if the length of `evaluations` is greater than [u32::MAX].
66pub fn interpolate_poly<B, E>(evaluations: &mut [E], inv_twiddles: &[B])
67where
68    B: StarkField,
69    E: FieldElement<BaseField = B>,
70{
71    assert!(evaluations.len() <= u32::MAX as usize, "too many evaluations");
72    let inv_length = B::inv((evaluations.len() as u32).into());
73    evaluations.fft_in_place(inv_twiddles);
74    evaluations.shift_by(inv_length);
75    evaluations.permute();
76}
77
78/// Interpolates `evaluations` over a domain of length `evaluations.len()` and shifted by
79/// `domain_offset` in the field specified by `B` into a polynomial in coefficient form using
80/// the FFT algorithm.
81///
82/// # Panics
83/// Panics if the length of `evaluations` is greater than [u32::MAX].
84pub fn interpolate_poly_with_offset<B, E>(
85    evaluations: &mut [E],
86    inv_twiddles: &[B],
87    domain_offset: B,
88) where
89    B: StarkField,
90    E: FieldElement<BaseField = B>,
91{
92    assert!(evaluations.len() <= u32::MAX as usize, "too many evaluations");
93
94    evaluations.fft_in_place(inv_twiddles);
95    evaluations.permute();
96
97    let domain_offset = B::inv(domain_offset);
98    let offset = B::inv((evaluations.len() as u32).into());
99
100    evaluations.shift_by_series(offset, domain_offset);
101}
