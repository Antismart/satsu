17/// Returns a vector containing successive powers of a given base.
18///
19/// More precisely, for base `b`, generates a vector with values [1, b, b^2, b^3, ..., b^(n-1)].
20///
21/// When `concurrent` feature is enabled, series generation is done concurrently in multiple
22/// threads.
23///
24/// # Examples
25/// ```
26/// # use winter_math::get_power_series;
27/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
28/// let n = 2048;
29/// let b = BaseElement::from(3u8);
30///
31/// let expected = (0..n)
32///     .map(|p| b.exp((p as u64).into()))
33///     .collect::<Vec<_>>();
34///
35/// let actual = get_power_series(b, n);
36/// assert_eq!(expected, actual);
37/// ```
38pub fn get_power_series<E>(b: E, n: usize) -> Vec<E>
39where
40    E: FieldElement,
41{
42    let mut result = unsafe { uninit_vector(n) };
43    batch_iter_mut!(&mut result, 1024, |batch: &mut [E], batch_offset: usize| {
44        let start = b.exp((batch_offset as u64).into());
45        fill_power_series(batch, b, start);
46    });
47    result
48}

50/// Returns a vector containing successive powers of a given base offset by the specified value.
51///
52/// More precisely, for base `b` and offset `s`, generates a vector with values
53/// [s, s * b, s * b^2, s * b^3, ..., s * b^(n-1)].
54///
55/// When `concurrent` feature is enabled, series generation is done concurrently in multiple
56/// threads.
57///
58/// # Examples
59/// ```
60/// # use winter_math::get_power_series_with_offset;
61/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
62/// let n = 2048;
63/// let b = BaseElement::from(3u8);
64/// let s = BaseElement::from(7u8);
65///
66/// let expected = (0..n)
67///     .map(|p| s * b.exp((p as u64).into()))
68///     .collect::<Vec<_>>();
69///
70/// let actual = get_power_series_with_offset(b, s, n);
71/// assert_eq!(expected, actual);
72/// ```
73pub fn get_power_series_with_offset<E>(b: E, s: E, n: usize) -> Vec<E>
74where
75    E: FieldElement,
76{
77    let mut result = unsafe { uninit_vector(n) };
78    batch_iter_mut!(&mut result, 1024, |batch: &mut [E], batch_offset: usize| {
79        let start = s * b.exp((batch_offset as u64).into());
80        fill_power_series(batch, b, start);
81    });
82    result
83}

85/// Computes element-wise sum of the provided vectors, and stores the result in the first vector.
86///
87/// When `concurrent` feature is enabled, the summation is performed concurrently in multiple
88/// threads.
89///
90/// # Panics
91/// Panics if lengths of `a` and `b` vectors are not the same.
92///
93/// # Examples
94/// ```
95/// # use winter_math::add_in_place;
96/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
97/// # use rand_utils::rand_vector;
98/// let a: Vec<BaseElement> = rand_vector(2048);
99/// let b: Vec<BaseElement> = rand_vector(2048);
100///
101/// let mut c = a.clone();
102/// add_in_place(&mut c, &b);
103///
104/// for ((a, b), c) in a.into_iter().zip(b).zip(c) {
105///     assert_eq!(a + b, c);
106/// }
107/// ```
108pub fn add_in_place<E>(a: &mut [E], b: &[E])
109where
110    E: FieldElement,
111{
112    assert!(a.len() == b.len(), "number of values must be the same for both operands");
113    iter_mut!(a).zip(b).for_each(|(a, &b)| *a += b);
114}

116/// Multiplies a sequence of values by a scalar and accumulates the results.
117///
118/// More precisely, computes `a[i]` + `b[i]` * `c` for all `i` and saves result into `a[i]`.
119///
120/// When `concurrent` feature is enabled, the computation is performed concurrently in multiple
121/// threads.
122///
123/// # Panics
124/// Panics if lengths of `a` and `b` slices are not the same.
125///
126/// # Examples
127/// ```
128/// # use winter_math::mul_acc;
129/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
130/// # use rand_utils::rand_vector;
131/// let a: Vec<BaseElement> = rand_vector(2048);
132/// let b: Vec<BaseElement> = rand_vector(2048);
133/// let c = BaseElement::new(12345);
134///
135/// let mut d = a.clone();
136/// mul_acc(&mut d, &b, c);
137///
138/// for ((a, b), d) in a.into_iter().zip(b).zip(d) {
139///     assert_eq!(a + b * c, d);
140/// }
141/// ```
142pub fn mul_acc<F, E>(a: &mut [E], b: &[F], c: E)
143where
144    F: FieldElement,
145    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
146{
147    assert!(a.len() == b.len(), "number of values must be the same for both slices");
148    iter_mut!(a).zip(b).for_each(|(a, &b)| *a += c.mul_base(b));
149}

151/// Computes a multiplicative inverse of a sequence of elements using batch inversion method.
152///
153/// Any ZEROs in the provided sequence are ignored.
154///
155/// When `concurrent` feature is enabled, the inversion is performed concurrently in multiple
156/// threads.
157///
158/// This function is significantly faster than inverting elements one-by-one because it
159/// essentially transforms `n` inversions into `3 * n` multiplications + 1 inversion.
160///
161/// # Examples
162/// ```
163/// # use winter_math::batch_inversion;
164/// # use winter_math::{fields::{f128::BaseElement}, FieldElement};
165/// # use rand_utils::rand_vector;
166/// let a: Vec<BaseElement> = rand_vector(2048);
167/// let b = batch_inversion(&a);
168///
169/// for (&a, &b) in a.iter().zip(b.iter()) {
170///     assert_eq!(a.inv(), b);
171/// }
172/// ```
173pub fn batch_inversion<E>(values: &[E]) -> Vec<E>
174where
175    E: FieldElement,
176{
177    let mut result: Vec<E> = unsafe { uninit_vector(values.len()) };
178    batch_iter_mut!(&mut result, 1024, |batch: &mut [E], batch_offset: usize| {
179        let start = batch_offset;
180        let end = start + batch.len();
181        serial_batch_inversion(&values[start..end], batch);
182    });
183    result
184}

189#[inline(always)]
190fn fill_power_series<E: FieldElement>(result: &mut [E], base: E, start: E) {
191    result[0] = start;
192    for i in 1..result.len() {
193        result[i] = result[i - 1] * base;
194    }
195}
196
197fn serial_batch_inversion<E: FieldElement>(values: &[E], result: &mut [E]) {
198    let mut last = E::ONE;
199    for (result, &value) in result.iter_mut().zip(values.iter()) {
200        *result = last;
201        if value != E::ZERO {
202            last *= value;
203        }
204    }
205
206    last = last.inv();
207
208    for i in (0..values.len()).rev() {
209        if values[i] == E::ZERO {
210            result[i] = E::ZERO;
211        } else {
212            result[i] *= last;
213            last *= values[i];
214        }
215    }
216}
