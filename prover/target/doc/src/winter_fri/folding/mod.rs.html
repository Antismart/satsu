21// DEGREE-RESPECTING PROJECTION
22// ================================================================================================
23/// Applies degree-respecting projection to evaluations of a polynomial.
24///
25/// This process reduces the  degree of the polynomial implied by the `evaluations` by `N`.
26/// For example, for `N` = 4, this is equivalent to the following:
27/// - Let `evaluations` contain the evaluations of polynomial *f*(x) of degree *k*.
28/// - Group coefficients of *f* so that *f*(x) = a(x) + x * b(x) + x^2 * c(x) + x^3 * d(x)
29/// - Compute random linear combination of polynomials a, b, c, d as
30///   *f'*(x) = a + α * b + α^2 * c + α^3 * d, where α is an element drawn uniformly at random from
31///   the entire field.
32/// - evaluate f'(x) on a domain which consists of x^4 from the original domain (and thus is
33///   1/4 the size).
34///
35/// However, the reduction is performed without converting the polynomials into coefficient form.
36/// That is, we can go directly form evaluations to folded evaluations. For this, for each
37/// evaluation in the folded domain, we need `N` evaluations in the source domain. For example,
38/// for the case of `N` = 4, to compute the evaluation of *f'*(x) we need to have evaluations of
39/// *f*(x) from the source domain at x^{1/4}, x^{2/4}, x^{3/4}, x.
40///
41/// This function expect the `evaluations` to be already in a transposed form such that all
42/// evaluations needed to compute a single evaluation in the folded domain are next to each other.
43///
44/// The example below shows the equivalence of performing the projection via coefficient form and
45/// via evaluation form for `N` = 2.
46/// ```
47/// # use math::{StarkField, FieldElement, fields::f128::BaseElement, get_power_series_with_offset, polynom};
48/// # use rand_utils::{rand_value, rand_vector};
49/// # use utils::transpose_slice;
50/// # use winter_fri::folding::apply_drp;
51/// // generate random alpha
52/// let alpha: BaseElement = rand_value();
53///
54/// // degree 7 polynomial f(x)
55/// let poly: Vec<BaseElement> = rand_vector(8);
56///
57/// // f'(x) = g(x) + alpha * h(x) such that g(x) consists of even coefficients of f(x) and h(x)
58/// // consists of odd coefficients of f(x). This is equivalent to using `folding_factor = 2`.
59/// let mut folded_poly = Vec::new();
60/// for i in 0..poly.len() / 2 {
61///     folded_poly.push(poly[2 * i] + alpha * poly[2 * i + 1]);
62/// }
63///
64/// // build a domain of 32 elements
65/// let n = 32_usize;
66/// let offset = BaseElement::GENERATOR;
67/// let g = BaseElement::get_root_of_unity(n.trailing_zeros());
68/// let domain = get_power_series_with_offset(g, offset, n);
69///
70/// // build a folded domain of half the size, such that x in the source domain maps to
71/// // x^2 in the folded domain.
72/// let g = BaseElement::get_root_of_unity((n / 2).trailing_zeros());
73/// let folded_domain = get_power_series_with_offset(g, offset.exp(2), n / 2);
74///
75/// // evaluate the polynomials over their respective domains
76/// let evaluations = polynom::eval_many(&poly, &domain);
77/// let folded_evaluations = polynom::eval_many(&folded_poly, &folded_domain);
78///
79/// // use DRP to perform folding without converting polynomials into coefficient form
80/// let transposed_evaluations = transpose_slice::<BaseElement, 2>(&evaluations);
81/// let drp_evaluations = apply_drp(&transposed_evaluations, offset, alpha);
82///
83/// // applying DRP should be equivalent to folding polynomials in coefficient form
84/// assert_eq!(folded_evaluations, drp_evaluations);
85/// ```
86pub fn apply_drp<B, E, const N: usize>(values: &[[E; N]], domain_offset: B, alpha: E) -> Vec<E>
87where
88    B: StarkField,
89    E: FieldElement<BaseField = B>,
90{
91    // build offset inverses and twiddles used during polynomial interpolation
92    let inv_offsets = get_inv_offsets(values.len(), domain_offset, N);
93    let inv_twiddles = get_inv_twiddles::<B>(N);
94    let len_offset = E::inv((N as u32).into());
95
96    let mut result = unsafe { uninit_vector(values.len()) };
97    iter_mut!(result)
98        .zip(values)
99        .zip(inv_offsets)
100        .for_each(|((result, values), domain_offset)| {
101            // interpolate the values into a polynomial; this is similar to interpolation with
102            // offset implemented in math::fft module
103            let mut poly = *values;
104            serial_fft(&mut poly, &inv_twiddles);
105
106            let mut offset = len_offset;
107            let domain_offset = E::from(domain_offset);
108            for coeff in poly.iter_mut() {
109                *coeff *= offset;
110                offset *= domain_offset;
111            }
112
113            // evaluate the polynomial at alpha, and save the result
114            *result = polynom::eval(&poly, alpha)
115        });
116
117    result
118}

120// POSITION FOLDING
121// ================================================================================================
122/// Maps positions in the source domain, to positions in the folded domain.
123///
124/// The size of the folded domain is reduced by the `folding_factor` as compared to the size of the
125/// source domain. Thus, the original positions may fall outside of the folded domain. To map
126/// positions in the source domain to positions in the folded domain, we simply compute
127/// `new_position = old_position % folded_domain_size` and discard duplicates.
128///
129/// ```
130/// # use winter_fri::folding::fold_positions;
131/// let positions = [1, 9, 12, 20];
132/// let folded_positions = fold_positions(&positions, 32, 4);
133///
134/// // 1 and 9 should map to 1; 12 and 20 should map to 4
135/// assert_eq!(vec![1, 4], folded_positions);
136/// ```
137///
138/// The domains we usually work with have the following form: c, c * ω, c * ω^2, ..., c * ω^(*n*-1),
139/// where *n* is the size of the domain, ω is *n*th root of unity, and c is the domain offset. Thus,
140/// the procedure described above maps c * ω^*i* to (c * ω^*i*)^*k*, where *k* is the folding factor.
141/// ```
142/// # use math::{fields::f128::BaseElement, StarkField, FieldElement, get_power_series_with_offset};
143/// // build a domain of size 32
144/// let n = 32usize;
145/// let c = BaseElement::GENERATOR;
146/// let g = BaseElement::get_root_of_unity(n.trailing_zeros());
147/// let source_domain = get_power_series_with_offset(g, c, n);
148///
149/// // build a domain of size 8 (folding_factor = 4)
150/// let g = BaseElement::get_root_of_unity((n / 4).trailing_zeros());
151/// let folded_domain = get_power_series_with_offset(g, c.exp(4_u32.into()), (n / 4));
152///
153/// // position 1 in the source domain maps to position 1 in the folded domain
154/// assert_eq!(folded_domain[1], source_domain[1].exp(4));
155///
156/// // position 9 in the source domain also maps to position 1 in the folded domain
157/// assert_eq!(folded_domain[1], source_domain[9].exp(4));
158/// ```
159pub fn fold_positions(
160    positions: &[usize],
161    source_domain_size: usize,
162    folding_factor: usize,
163) -> Vec<usize> {
164    let target_domain_size = source_domain_size / folding_factor;
165
166    let mut result = Vec::new();
167    for position in positions {
168        let position = position % target_domain_size;
169        // make sure we don't record duplicated values
170        if !result.contains(&position) {
171            result.push(position);
172        }
173    }
174
175    result
176}

178// HELPER FUNCTIONS
179// ================================================================================================
180
181fn get_inv_offsets<B>(domain_size: usize, domain_offset: B, folding_factor: usize) -> Vec<B>
182where
183    B: StarkField,
184{
185    let n = domain_size * folding_factor;
186    let g = B::get_root_of_unity(n.ilog2());
187    get_power_series_with_offset(g.inv(), domain_offset.inv(), domain_size)
188}
