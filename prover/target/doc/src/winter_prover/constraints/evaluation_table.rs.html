24// CONSTRAINT EVALUATION TABLE
25// ================================================================================================
26
27pub struct ConstraintEvaluationTable<'a, E: FieldElement> {
28    evaluations: Vec<Vec<E>>,
29    divisors: Vec<ConstraintDivisor<E::BaseField>>,
30    domain: &'a StarkDomain<E::BaseField>,
31
32    #[cfg(debug_assertions)]
33    main_transition_evaluations: Vec<Vec<E::BaseField>>,
34    #[cfg(debug_assertions)]
35    aux_transition_evaluations: Vec<Vec<E>>,
36    #[cfg(debug_assertions)]
37    expected_transition_degrees: Vec<usize>,
38}
39
40impl<'a, E: FieldElement> ConstraintEvaluationTable<'a, E> {
41    // CONSTRUCTOR
42    // --------------------------------------------------------------------------------------------
43    /// Returns a new constraint evaluation table with number of columns equal to the number of
44    /// specified divisors, and number of rows equal to the size of constraint evaluation domain.
45    #[cfg(not(debug_assertions))]
46    pub fn new(
47        domain: &'a StarkDomain<E::BaseField>,
48        divisors: Vec<ConstraintDivisor<E::BaseField>>,
49    ) -> Self {
50        let num_columns = divisors.len();
51        let num_rows = domain.ce_domain_size();
52        ConstraintEvaluationTable {
53            evaluations: uninit_matrix(num_columns, num_rows),
54            divisors,
55            domain,
56        }
57    }
58
59    /// Similar to the as above constructor but used in debug mode. In debug mode we also want
60    /// to keep track of all evaluated transition constraints so that we can verify that their
61    /// expected degrees match their actual degrees.
62    #[cfg(debug_assertions)]
63    pub fn new(
64        domain: &'a StarkDomain<E::BaseField>,
65        divisors: Vec<ConstraintDivisor<E::BaseField>>,
66        transition_constraints: &TransitionConstraints<E>,
67    ) -> Self {
68        let num_columns = divisors.len();
69        let num_rows = domain.ce_domain_size();
70        let num_tm_columns = transition_constraints.num_main_constraints();
71        let num_ta_columns = transition_constraints.num_aux_constraints();
72
73        // collect expected degrees for all transition constraints to compare them against actual
74        // degrees; we do this in debug mode only because this comparison is expensive
75        let expected_transition_degrees =
76            build_transition_constraint_degrees(transition_constraints, domain.trace_length());
77
78        ConstraintEvaluationTable {
79            evaluations: uninit_matrix(num_columns, num_rows),
80            divisors,
81            domain,
82            main_transition_evaluations: uninit_matrix(num_tm_columns, num_rows),
83            aux_transition_evaluations: uninit_matrix(num_ta_columns, num_rows),
84            expected_transition_degrees,
85        }
86    }
87
88    // PUBLIC ACCESSORS
89    // --------------------------------------------------------------------------------------------
90
91    /// Returns the number of rows in this table. This is the same as the size of the constraint
92    /// evaluation domain.
93    pub fn num_rows(&self) -> usize {
94        self.evaluations[0].len()
95    }
96
97    /// Returns number of columns in this table.
98    ///
99    /// The first column always contains the value of combined transition constraint evaluations;
100    /// the remaining columns contain values of assertion constraint evaluations combined based on
101    /// common divisors.
102    #[allow(dead_code)]
103    pub fn num_columns(&self) -> usize {
104        self.evaluations.len()
105    }
106
107    // TABLE FRAGMENTS
108    // --------------------------------------------------------------------------------------------
109
110    /// Break the table into the number of specified fragments. All fragments can be updated
111    /// independently - e.g. in different threads.
112    pub fn fragments(&mut self, num_fragments: usize) -> Vec<EvaluationTableFragment<E>> {
113        let fragment_size = self.num_rows() / num_fragments;
114        assert!(
115            fragment_size >= MIN_FRAGMENT_SIZE,
116            "fragment size must be at least {MIN_FRAGMENT_SIZE}, but was {fragment_size}"
117        );
118
119        // break evaluations into fragments
120        let evaluation_data = make_fragments(&mut self.evaluations, num_fragments);
121
122        #[cfg(debug_assertions)]
123        let result = {
124            // in debug mode, also break individual transition evaluations into fragments
125            let tm_evaluation_data =
126                make_fragments(&mut self.main_transition_evaluations, num_fragments);
127            let ta_evaluation_data =
128                make_fragments(&mut self.aux_transition_evaluations, num_fragments);
129
130            evaluation_data
131                .into_iter()
132                .zip(tm_evaluation_data)
133                .zip(ta_evaluation_data)
134                .enumerate()
135                .map(|(i, ((evaluations, tm_evaluations), ta_evaluations))| {
136                    EvaluationTableFragment {
137                        offset: i * fragment_size,
138                        evaluations,
139                        tm_evaluations,
140                        ta_evaluations,
141                    }
142                })
143                .collect()
144        };
145
146        #[cfg(not(debug_assertions))]
147        let result = evaluation_data
148            .into_iter()
149            .enumerate()
150            .map(|(i, evaluations)| EvaluationTableFragment {
151                offset: i * fragment_size,
152                evaluations,
153            })
154            .collect();
155
156        result
157    }
158
159    // CONSTRAINT COMPOSITION
160    // --------------------------------------------------------------------------------------------
161    /// Divides constraint evaluation columns by their respective divisor (in evaluation form) and
162    /// combines the results into a single column.
163    pub fn combine(self) -> Vec<E> {
164        // allocate memory for the combined polynomial
165        let mut combined_poly = vec![E::ZERO; self.num_rows()];
166
167        // iterate over all columns of the constraint evaluation table, divide each column
168        // by the evaluations of its corresponding divisor, and add all resulting evaluations
169        // together into a single vector
170        for (column, divisor) in self.evaluations.into_iter().zip(self.divisors.iter()) {
171            // divide the column by the divisor and accumulate the result into combined_poly
172            acc_column(column, divisor, self.domain, &mut combined_poly);
173        }
174
175        combined_poly
176    }
177
178    // DEBUG HELPERS
179    // --------------------------------------------------------------------------------------------
180
181    #[cfg(debug_assertions)]
182    pub fn validate_transition_degrees(&mut self) {
183        // evaluate transition constraint divisor (which is assumed to be the first one in the
184        // divisor list) over the constraint evaluation domain. this is used later to compute
185        // actual degrees of transition constraint evaluations.
186        let div_values = evaluate_divisor::<E::BaseField>(
187            &self.divisors[0],
188            self.num_rows(),
189            self.domain.offset(),
190        );
191
192        // collect actual degrees for all transition constraints by interpolating saved
193        // constraint evaluations into polynomials and checking their degree; also
194        // determine max transition constraint degree
195        let mut actual_degrees = Vec::with_capacity(self.expected_transition_degrees.len());
196        let mut max_degree = 0;
197        let inv_twiddles = fft::get_inv_twiddles::<E::BaseField>(self.num_rows());
198
199        // first process transition constraint evaluations for the main trace segment
200        for evaluations in self.main_transition_evaluations.iter() {
201            let degree = get_transition_poly_degree(evaluations, &inv_twiddles, &div_values);
202            actual_degrees.push(degree);
203            max_degree = core::cmp::max(max_degree, degree);
204        }
205
206        // then process transition constraint evaluations for the auxiliary trace segment
207        for evaluations in self.aux_transition_evaluations.iter() {
208            let degree = get_transition_poly_degree(evaluations, &inv_twiddles, &div_values);
209            actual_degrees.push(degree);
210            max_degree = core::cmp::max(max_degree, degree);
211        }
212
213        // make sure expected and actual degrees are equal
214        assert_eq!(
215            self.expected_transition_degrees, actual_degrees,
216            "transition constraint degrees didn't match\nexpected: {:>3?}\nactual:   {:>3?}",
217            self.expected_transition_degrees, actual_degrees
218        );
219
220        // make sure evaluation domain size does not exceed the size required by max degree
221        let expected_domain_size =
222            core::cmp::max(max_degree, self.domain.trace_length() + 1).next_power_of_two();
223        assert_eq!(
224            expected_domain_size,
225            self.num_rows(),
226            "incorrect constraint evaluation domain size; expected {}, but was {}",
227            expected_domain_size,
228            self.num_rows()
229        );
230    }
231}

236pub struct EvaluationTableFragment<'a, E: FieldElement> {
237    offset: usize,
238    evaluations: Vec<&'a mut [E]>,
239
240    #[cfg(debug_assertions)]
241    tm_evaluations: Vec<&'a mut [E::BaseField]>,
242    #[cfg(debug_assertions)]
243    ta_evaluations: Vec<&'a mut [E]>,
244}
245
246impl<E: FieldElement> EvaluationTableFragment<'_, E> {
247    /// Returns the row at which the fragment starts.
248    pub fn offset(&self) -> usize {
249        self.offset
250    }
251
252    /// Returns the number of evaluation rows in the fragment.
253    pub fn num_rows(&self) -> usize {
254        self.evaluations[0].len()
255    }
256
257    /// Returns the number of columns in every evaluation row.
258    pub fn num_columns(&self) -> usize {
259        self.evaluations.len()
260    }
261
262    /// Updates a single row in the fragment with provided data.
263    pub fn update_row(&mut self, row_idx: usize, row_data: &[E]) {
264        for (column, &value) in self.evaluations.iter_mut().zip(row_data) {
265            column[row_idx] = value;
266        }
267    }
268
269    /// Updates transition evaluations row with the provided data; available only in debug mode.
270    #[cfg(debug_assertions)]
271    pub fn update_transition_evaluations(
272        &mut self,
273        row_idx: usize,
274        main_evaluations: &[E::BaseField],
275        aux_evaluations: &[E],
276    ) {
277        for (column, &value) in self.tm_evaluations.iter_mut().zip(main_evaluations) {
278            column[row_idx] = value;
279        }
280        for (column, &value) in self.ta_evaluations.iter_mut().zip(aux_evaluations) {
281            column[row_idx] = value;
282        }
283    }
284}

289/// Allocates memory for a two-dimensional data structure without initializing it.
290fn uninit_matrix<E: FieldElement>(num_cols: usize, num_rows: usize) -> Vec<Vec<E>> {
291    unsafe { (0..num_cols).map(|_| uninit_vector(num_rows)).collect() }
292}
293
294/// Breaks the source data into a mutable set of fragments such that each fragment has the same
295/// number of columns as the source data, and the number of rows equal to `num_fragments`
296/// parameter.
297///
298/// If the source data is empty, the returned vector will contain number of empty vectors equal
299/// to `num_fragments` parameter.
300fn make_fragments<E: FieldElement>(
301    source: &mut [Vec<E>],
302    num_fragments: usize,
303) -> Vec<Vec<&mut [E]>> {
304    let mut result = (0..num_fragments).map(|_| Vec::new()).collect::<Vec<_>>();
305    if !source.is_empty() {
306        let fragment_size = source[0].len() / num_fragments;
307        source.iter_mut().for_each(|column| {
308            for (i, fragment) in column.chunks_mut(fragment_size).enumerate() {
309                result[i].push(fragment);
310            }
311        });
312    }
313    result
314}

316#[allow(clippy::many_single_char_names)]
317fn acc_column<E: FieldElement>(
318    column: Vec<E>,
319    divisor: &ConstraintDivisor<E::BaseField>,
320    domain: &StarkDomain<E::BaseField>,
321    result: &mut [E],
322) {
323    let numerator = divisor.numerator();
324    assert_eq!(numerator.len(), 1, "complex divisors are not yet supported");
325
326    // compute inverse evaluations of the divisor's numerator, which has the form (x^a - b)
327    let z = get_inv_evaluation(divisor, domain);
328
329    // divide column values by the divisor; for boundary constraints this computed simply as
330    // multiplication of column value by the inverse of divisor numerator; for transition
331    // constraints, it is computed similarly, but the result is also multiplied by the divisor's
332    // denominator (exclusion point).
333    if divisor.exemptions().is_empty() {
334        // the column represents merged evaluations of boundary constraints, and divisor has the
335        // form of (x^a - b); thus to divide the column by the divisor, we compute: value * z,
336        // where z = 1 / (x^a - 1) and has already been computed above.
337        iter_mut!(result, 1024)
338            .zip(column)
339            .enumerate()
340            .for_each(|(i, (acc_value, value))| {
341                // determine which value of z corresponds to the current domain point
342                let z = z[i % z.len()];
343                // compute value * z and add it to the result
344                *acc_value += value.mul_base(z);
345            });
346    } else {
347        // the column represents merged evaluations of transition constraints, and divisor has the
348        // form of (x^a - 1) / e(x), where e(x) describes the exemption points; thus, to divide
349        // the column by the divisor, we compute: value * e(x) * z, where z = 1 / (x^a - 1) and has
350        // already been computed above.
351        batch_iter_mut!(
352            result,
353            128, // min batch size
354            |batch: &mut [E], batch_offset: usize| {
355                for (i, acc_value) in batch.iter_mut().enumerate() {
356                    // compute value of e(x) and compute next value of x
357                    let x = domain.get_ce_x_at(batch_offset + i);
358                    let e = divisor.evaluate_exemptions_at(x);
359                    // determine which value of z corresponds to the current domain point
360                    let z = z[i % z.len()];
361                    // compute value * e(x) * z and add it to the result
362                    *acc_value += column[batch_offset + i].mul_base(z * e);
363                }
364            }
365        );
366    }
367}

369/// Computes evaluations of the divisor's numerator over the domain of the specified size and offset.
370fn get_inv_evaluation<B: StarkField>(
371    divisor: &ConstraintDivisor<B>,
372    domain: &StarkDomain<B>,
373) -> Vec<B> {
374    let numerator = divisor.numerator();
375    let a = numerator[0].0 as u64; // numerator degree
376    let b = numerator[0].1;
377
378    // this guarantees that we can use get_ce_x_power_at() below but limits execution trace length
379    // to be at most 2^32. in the future, we should revisit this to allow execution traces of
380    // greater length.
381    assert!(
382        a <= u32::MAX as u64,
383        "constraint divisor numerator degree cannot exceed {}, but was {}",
384        u32::MAX,
385        a
386    );
387
388    let n = domain.ce_domain_size() / a as usize;
389    let domain_offset_exp = domain.offset().exp(a.into());
390
391    // compute x^a - b for all x
392    let mut evaluations = unsafe { uninit_vector(n) };
393    batch_iter_mut!(
394        &mut evaluations,
395        128, // min batch size
396        |batch: &mut [B], batch_offset: usize| {
397            for (i, evaluation) in batch.iter_mut().enumerate() {
398                let x = domain.get_ce_x_power_at(batch_offset + i, a, domain_offset_exp);
399                *evaluation = x - b;
400            }
401        }
402    );
403
404    // compute 1 / (x^a - b)
405    batch_inversion(&evaluations)
406}

411/// Returns evaluation degrees of all transition constraints.
412///
413/// An evaluation degree is defined as degree of transition constraints in the context of a given
414/// execution trace accounting for constraint divisor degree. For most constraints, this degree is
415/// computed as `([trace_length - 1] * [constraint degree]) - [divisor degree]`. However, for
416/// constraints which rely on periodic columns this computation is slightly more complex.
417///
418/// The general idea is that evaluation degree is the degree of rational function `C(x) / z(x)`,
419/// where `C(x)` is the constraint polynomial and `z(x)` is the divisor polynomial.
420#[cfg(debug_assertions)]
421fn build_transition_constraint_degrees<E: FieldElement>(
422    constraints: &TransitionConstraints<E>,
423    trace_length: usize,
424) -> Vec<usize> {
425    let mut result = Vec::new();
426
427    for degree in constraints.main_constraint_degrees() {
428        result.push(degree.get_evaluation_degree(trace_length) - constraints.divisor().degree())
429    }
430
431    for degree in constraints.aux_constraint_degrees() {
432        result.push(degree.get_evaluation_degree(trace_length) - constraints.divisor().degree())
433    }
434
435    result
436}

438/// Computes the actual degree of a transition polynomial described by the provided evaluations.
439///
440/// The degree is computed as follows:
441/// - First, we divide the polynomial evaluations by the evaluations of transition constraint
442///   divisor (`div_values`). This is needed because it is possible for the numerator portions of
443///   transition constraints to have a degree which is larger than the size of the evaluation
444///   domain (and thus, interpolating the numerator would yield an incorrect result). However,
445///   once the divisor values are divided out, the degree of the resulting polynomial should be
446///   smaller than the size of the evaluation domain, and thus, we can interpolate safely.
447/// - Then, we interpolate the polynomial over the domain specified by `inv_twiddles`.
448/// - And finally, we get the degree from the interpolated polynomial.
449#[cfg(debug_assertions)]
450fn get_transition_poly_degree<E: FieldElement>(
451    evaluations: &[E],
452    inv_twiddles: &[E::BaseField],
453    div_values: &[E::BaseField],
454) -> usize {
455    let mut evaluations = evaluations
456        .iter()
457        .zip(div_values)
458        .map(|(&c, &d)| c / E::from(d))
459        .collect::<Vec<_>>();
460    fft::interpolate_poly(&mut evaluations, inv_twiddles);
461    math::polynom::degree_of(&evaluations)
462}

464/// Evaluates constraint divisor over the specified domain. This is similar to [get_inv_evaluation]
465/// function above but uses a more straight-forward but less efficient evaluation methodology and
466/// also does not invert the results.
467#[cfg(debug_assertions)]
468fn evaluate_divisor<E: FieldElement>(
469    divisor: &ConstraintDivisor<E::BaseField>,
470    domain_size: usize,
471    domain_offset: E::BaseField,
472) -> Vec<E> {
473    let g = E::BaseField::get_root_of_unity(domain_size.trailing_zeros());
474    let domain = math::get_power_series_with_offset(g, domain_offset, domain_size);
475    domain.into_iter().map(|x| E::from(divisor.evaluate_at(x))).collect()
476}
