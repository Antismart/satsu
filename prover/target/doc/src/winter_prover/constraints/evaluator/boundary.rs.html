16/// Boundary polynomials with this degree or smaller will be evaluated on the fly, while for
17/// larger polynomials all evaluations over the constraint evaluation domain will be pre-computed.
18const SMALL_POLY_DEGREE: usize = 63;
19
20// BOUNDARY CONSTRAINTS
21// ================================================================================================
22
23/// Contains all boundary constraints defined for an instance of a computation. This includes
24/// constraints against the main segment of the execution trace as well as constraints against
25/// the auxiliary trace segment (if any).
26///
27/// We transform the constraints defined in the [air] crate into specialized constraints here
28/// to make evaluation of these constraints more efficient in the prover context.
29pub struct BoundaryConstraints<E: FieldElement>(Vec<BoundaryConstraintGroup<E>>);
30
31impl<E: FieldElement> BoundaryConstraints<E> {
32    // CONSTRUCTOR
33    // --------------------------------------------------------------------------------------------
34    /// Returns a new instance of [BoundaryConstraints] constructed from the constraints defined
35    /// by an instance of AIR for a specific computation.
36    pub fn new<A: Air<BaseField = E::BaseField>>(
37        air: &A,
38        aux_rand_elements: Option<&AuxRandElements<E>>,
39        composition_coefficients: &[E],
40    ) -> Self {
41        // get constraints from the AIR instance
42        let source = air.get_boundary_constraints(aux_rand_elements, composition_coefficients);
43
44        // initialize a map of twiddles here so that we can keep track of already computed
45        // twiddles; this helps us avoid building twiddles over and over again for constraints
46        // defined over the same domain. twiddles are relevant only for large polynomial
47        // constraints.
48        let mut twiddle_map = BTreeMap::new();
49
50        // transform constraints against the main segment of the execution trace into specialized
51        // constraints
52        let mut result = source
53            .main_constraints()
54            .iter()
55            .map(|group| {
56                BoundaryConstraintGroup::from_main_constraints(group, air, &mut twiddle_map)
57            })
58            .collect::<Vec<BoundaryConstraintGroup<E>>>();
59
60        // transform constraints against the auxiliary trace segment (if any) into specialized
61        // constraints. this also checks if a group with the same divisor has already been
62        // transformed (when processing constraints against the main trace above), and if so,
63        // appends constraints to that group rather than creating a new group. this ensures
64        // that we always end up with a single constraint group for the same divisor.
65        for group in source.aux_constraints() {
66            match result.iter_mut().find(|g| &g.divisor == group.divisor()) {
67                Some(x) => x.add_aux_constraints(group, air, &mut twiddle_map),
68                None => {
69                    let group =
70                        BoundaryConstraintGroup::from_aux_constraints(group, air, &mut twiddle_map);
71                    result.push(group);
72                },
73            };
74        }
75
76        Self(result)
77    }
78
79    // PUBLIC ACCESSORS
80    // --------------------------------------------------------------------------------------------
81
82    /// Returns a vector of all boundary constraint divisors.
83    pub fn get_divisors(&self) -> Vec<ConstraintDivisor<E::BaseField>> {
84        self.0.iter().map(|g| g.divisor.clone()).collect()
85    }
86
87    // EVALUATORS
88    // --------------------------------------------------------------------------------------------
89
90    /// Evaluates boundary constraints against the main segment of an execution trace at the
91    /// specified step of constraint evaluation domain.
92    pub fn evaluate_main(
93        &self,
94        main_state: &[E::BaseField],
95        domain: &StarkDomain<E::BaseField>,
96        step: usize,
97        result: &mut [E],
98    ) {
99        let x = domain.get_ce_x_at(step);
100        for (group, result) in self.0.iter().zip(result.iter_mut()) {
101            *result = group.evaluate_main(main_state, step, x);
102        }
103    }
104
105    /// Evaluates boundary constraints against all segments of an execution trace at the
106    /// specified step of constraint evaluation domain.
107    pub fn evaluate_all(
108        &self,
109        main_state: &[E::BaseField],
110        aux_state: &[E],
111        domain: &StarkDomain<E::BaseField>,
112        step: usize,
113        result: &mut [E],
114    ) {
115        let x = domain.get_ce_x_at(step);
116        for (group, result) in self.0.iter().zip(result.iter_mut()) {
117            *result = group.evaluate_all(main_state, aux_state, step, x);
118        }
119    }
120}

122// BOUNDARY CONSTRAINT GROUP
123// ================================================================================================
124
125/// Contains constraints all having the same divisor. The constraints are separated into single
126/// value constraints, small polynomial constraints, and large polynomial constraints.
127///
128/// The constraints are also separated into constraints against the main segment of the execution
129/// and the constraints against auxiliary segments of the execution trace (if any).
130pub struct BoundaryConstraintGroup<E: FieldElement> {
131    divisor: ConstraintDivisor<E::BaseField>,
132    // main trace constraints
133    main_single_value: Vec<SingleValueConstraint<E::BaseField, E>>,
134    main_small_poly: Vec<SmallPolyConstraint<E::BaseField, E>>,
135    main_large_poly: Vec<LargePolyConstraint<E::BaseField, E>>,
136    // auxiliary trace constraints
137    aux_single_value: Vec<SingleValueConstraint<E, E>>,
138    aux_small_poly: Vec<SmallPolyConstraint<E, E>>,
139    aux_large_poly: Vec<LargePolyConstraint<E, E>>,
140}
141
142impl<E: FieldElement> BoundaryConstraintGroup<E> {
143    // CONSTRUCTORS
144    // --------------------------------------------------------------------------------------------
145
146    /// Returns an empty [BoundaryConstraintGroup] instantiated with the specified divisor and
147    /// degree adjustment factor.
148    fn new(divisor: ConstraintDivisor<E::BaseField>) -> Self {
149        Self {
150            divisor,
151            main_single_value: Vec::new(),
152            main_small_poly: Vec::new(),
153            main_large_poly: Vec::new(),
154            aux_single_value: Vec::new(),
155            aux_small_poly: Vec::new(),
156            aux_large_poly: Vec::new(),
157        }
158    }
159
160    /// Returns a [BoundaryConstraintGroup] created from the specified group of constraints against
161    /// the main segment of an execution trace. Constraints against auxiliary trace segment in this
162    /// group will be empty.
163    ///
164    /// Twiddles and [Air] instance are passed in for evaluating large polynomial constraints
165    /// (if any).
166    pub fn from_main_constraints<A: Air<BaseField = E::BaseField>>(
167        source: &air::BoundaryConstraintGroup<E::BaseField, E>,
168        air: &A,
169        twiddle_map: &mut BTreeMap<usize, Vec<E::BaseField>>,
170    ) -> Self {
171        let mut result = Self::new(source.divisor().clone());
172
173        for constraint in source.constraints() {
174            if constraint.poly().len() == 1 {
175                let constraint = SingleValueConstraint::new(constraint);
176                result.main_single_value.push(constraint);
177            } else if constraint.poly().len() < SMALL_POLY_DEGREE {
178                let constraint = SmallPolyConstraint::new(constraint);
179                result.main_small_poly.push(constraint);
180            } else {
181                let constraint = LargePolyConstraint::new(constraint, air, twiddle_map);
182                result.main_large_poly.push(constraint);
183            }
184        }
185
186        result
187    }
188
189    /// Returns a [BoundaryConstraintGroup] created from the specified group of constraints against
190    /// auxiliary segments of an execution trace. Constraints against the main trace segment in this
191    /// group will be empty.
192    ///
193    /// Twiddles and [Air] instance are passed in for evaluating large polynomial constraints
194    /// (if any).
195    pub fn from_aux_constraints<A: Air<BaseField = E::BaseField>>(
196        group: &air::BoundaryConstraintGroup<E, E>,
197        air: &A,
198        twiddle_map: &mut BTreeMap<usize, Vec<E::BaseField>>,
199    ) -> Self {
200        let mut result = Self::new(group.divisor().clone());
201        result.add_aux_constraints(group, air, twiddle_map);
202        result
203    }
204
205    // STATE MUTATORS
206    // --------------------------------------------------------------------------------------------
207
208    /// Adds the provided constraints against auxiliary segments of an execution trace to this
209    /// group.
210    ///
211    /// Twiddles and [Air] instance are passed in for evaluating large polynomial constraints
212    /// (if any).
213    ///
214    /// # Panics
215    /// Panics if the divisor of the provided constraints doesn't match the divisor of this group.
216    pub fn add_aux_constraints<A: Air<BaseField = E::BaseField>>(
217        &mut self,
218        group: &air::BoundaryConstraintGroup<E, E>,
219        air: &A,
220        twiddle_map: &mut BTreeMap<usize, Vec<E::BaseField>>,
221    ) {
222        assert_eq!(group.divisor(), &self.divisor, "inconsistent constraint divisor");
223
224        for constraint in group.constraints() {
225            if constraint.poly().len() == 1 {
226                let constraint = SingleValueConstraint::new(constraint);
227                self.aux_single_value.push(constraint);
228            } else if constraint.poly().len() < SMALL_POLY_DEGREE {
229                let constraint = SmallPolyConstraint::new(constraint);
230                self.aux_small_poly.push(constraint);
231            } else {
232                let constraint = LargePolyConstraint::new(constraint, air, twiddle_map);
233                self.aux_large_poly.push(constraint);
234            }
235        }
236    }
237
238    // EVALUATORS
239    // --------------------------------------------------------------------------------------------
240
241    /// Evaluates the constraints against the main segment of the execution trace contained in
242    /// this group at the specified step of the trace.
243    pub fn evaluate_main(&self, state: &[E::BaseField], ce_step: usize, x: E::BaseField) -> E {
244        let mut result = E::ZERO;
245
246        // evaluate all single-value constraints
247        for constraint in self.main_single_value.iter() {
248            result += constraint.evaluate(state);
249        }
250
251        // evaluate all small polynomial constraints
252        for constraint in self.main_small_poly.iter() {
253            result += constraint.evaluate(state, x);
254        }
255
256        // evaluate all large polynomial constraints
257        for constraint in self.main_large_poly.iter() {
258            result += constraint.evaluate(state, ce_step);
259        }
260
261        result
262    }
263
264    /// Evaluates all constraints contained in this group at the specified step of the
265    /// execution trace.
266    pub fn evaluate_all(
267        &self,
268        main_state: &[E::BaseField],
269        aux_state: &[E],
270        ce_step: usize,
271        x: E::BaseField,
272    ) -> E {
273        let mut result = self.evaluate_main(main_state, ce_step, x);
274
275        // evaluate all single-value constraints
276        for constraint in self.aux_single_value.iter() {
277            result += constraint.evaluate(aux_state);
278        }
279
280        // evaluate all small polynomial constraints
281        for constraint in self.aux_small_poly.iter() {
282            result += constraint.evaluate(aux_state, x);
283        }
284
285        // evaluate all large polynomial constraints
286        for constraint in self.aux_large_poly.iter() {
287            result += constraint.evaluate(aux_state, ce_step);
288        }
289
290        result
291    }
292}

297/// A constraint where the numerator can be represented by p(x) - v, where v is the asserted value,
298/// and p(x) is the trace polynomial for the column against which the constraint is applied.
299struct SingleValueConstraint<F, E>
300where
301    F: FieldElement,
302    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
303{
304    column: usize,
305    value: F,
306    coefficients: E,
307}
308
309impl<F, E> SingleValueConstraint<F, E>
310where
311    F: FieldElement,
312    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
313{
314    /// Returns an new instance of [SingleValueConstraint] created from the specified source
315    /// boundary constraint.
316    pub fn new(source: &air::BoundaryConstraint<F, E>) -> Self {
317        debug_assert!(source.poly().len() == 1, "not a single constraint");
318        Self {
319            column: source.column(),
320            value: source.poly()[0],
321            coefficients: *source.cc(),
322        }
323    }
324
325    /// Evaluates this constraint over the specified state and returns the result.
326    ///
327    /// This also multiplies by the composition coefficient.
328    pub fn evaluate(&self, state: &[F]) -> E {
329        let evaluation = state[self.column] - self.value;
330        self.coefficients.mul_base(evaluation)
331    }
332}
333
334/// A constraint where the numerator can be represented by p(x) - c(x), where b(x) is the
335/// polynomial describing a set of asserted values. This specialization is useful when the
336/// degree of b(x) is relatively small, and thus, is cheap to evaluate on the fly.
337///
338/// TODO: investigate whether we get any significant improvement vs. [LargePolyConstraint], and if
339/// so, what is the appropriate value for SMALL_POLY_DEGREE.
340struct SmallPolyConstraint<F, E>
341where
342    F: FieldElement,
343    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
344{
345    column: usize,
346    poly: Vec<F>,
347    x_offset: F::BaseField,
348    coefficients: E,
349}
350
351impl<F, E> SmallPolyConstraint<F, E>
352where
353    F: FieldElement,
354    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
355{
356    /// Returns an new instance of [SmallPolyConstraint] created from the specified source
357    /// boundary constraint.
358    pub fn new(source: &air::BoundaryConstraint<F, E>) -> Self {
359        debug_assert!(
360            source.poly().len() > 1 && source.poly().len() < SMALL_POLY_DEGREE,
361            "not a small poly constraint"
362        );
363        Self {
364            column: source.column(),
365            poly: source.poly().to_vec(),
366            x_offset: source.poly_offset().1,
367            coefficients: *source.cc(),
368        }
369    }
370
371    /// Evaluates this constraint over the specified state and returns the result.
372    ///
373    /// We need the point in the domain ('x') corresponding to the passed-in state because to
374    /// evaluate the constraint we need to evaluate its value polynomial at `x`.
375    pub fn evaluate(&self, state: &[F], x: F::BaseField) -> E {
376        let x = x * self.x_offset;
377        // evaluate constraint polynomial as x * offset using Horner evaluation
378        let assertion_value =
379            self.poly.iter().rev().fold(F::ZERO, |acc, &coeff| acc.mul_base(x) + coeff);
380        // evaluate the constraint
381        let evaluation = state[self.column] - assertion_value;
382        self.coefficients.mul_base(evaluation)
383    }
384}
385
386/// A constraint where the numerator can be represented by p(x) - b(x), where b(x) is a large
387/// polynomial. In such cases, we pre-compute evaluations of b(x) by evaluating it over the
388/// entire constraint evaluation domain (using FFT).
389struct LargePolyConstraint<F, E>
390where
391    F: FieldElement,
392    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
393{
394    column: usize,
395    values: Vec<F>,
396    step_offset: usize,
397    coefficients: E,
398}
399
400impl<F, E> LargePolyConstraint<F, E>
401where
402    F: FieldElement,
403    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
404{
405    /// Returns a new instance of [LargePolyConstraint] created from the specified source
406    /// boundary constraint.
407    pub fn new<A: Air<BaseField = F::BaseField>>(
408        source: &air::BoundaryConstraint<F, E>,
409        air: &A,
410        twiddle_map: &mut BTreeMap<usize, Vec<F::BaseField>>,
411    ) -> Self {
412        debug_assert!(source.poly().len() >= SMALL_POLY_DEGREE, "not a large poly constraint");
413        // evaluate the polynomial over the entire constraint evaluation domain; first
414        // get twiddles for the evaluation; if twiddles haven't been built yet, build them
415        let poly_length = source.poly().len();
416        let twiddles =
417            twiddle_map.entry(poly_length).or_insert_with(|| fft::get_twiddles(poly_length));
418
419        let values = fft::evaluate_poly_with_offset(
420            source.poly(),
421            twiddles,
422            air.domain_offset(),
423            air.ce_domain_size() / poly_length,
424        );
425
426        LargePolyConstraint {
427            column: source.column(),
428            values,
429            step_offset: source.poly_offset().0 * air.ce_blowup_factor(),
430            coefficients: *source.cc(),
431        }
432    }
433
434    /// Evaluates this constraint at the specified step of the constraint evaluation domain.
435    ///
436    /// This also applies composition coefficients as well as the degree adjustment factor
437    /// (defined by `xp` parameter) to the evaluation before it is returned.
438    pub fn evaluate(&self, state: &[F], ce_step: usize) -> E {
439        let value_index = if self.step_offset > 0 {
440            // if the assertion happens on steps which are not a power of 2, we need to offset the
441            // evaluation; the below basically computes (ce_step - step_offset) % values.len();
442            // this is equivalent to evaluating the polynomial at x * x_offset coordinate.
443            if self.step_offset > ce_step {
444                self.values.len() + ce_step - self.step_offset
445            } else {
446                ce_step - self.step_offset
447            }
448        } else {
449            ce_step
450        };
451        let evaluation = state[self.column] - self.values[value_index];
452        (self.coefficients).mul_base(evaluation)
453    }
454}
