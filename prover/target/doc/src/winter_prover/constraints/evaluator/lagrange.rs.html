16/// Contains a specific strategy for evaluating the Lagrange kernel boundary and transition
17/// constraints where the divisors' evaluation is batched.
18///
19/// Specifically, [`batch_inversion`] is used to reduce the number of divisions performed.
20pub struct LagrangeKernelConstraintsBatchEvaluator<E: FieldElement> {
21    lagrange_kernel_constraints: LagrangeKernelConstraints<E>,
22    rand_elements: LagrangeKernelRandElements<E>,
23}
24
25impl<E: FieldElement> LagrangeKernelConstraintsBatchEvaluator<E> {
26    /// Constructs a new [`LagrangeConstraintsBatchEvaluator`].
27    pub fn new<A: Air>(
28        air: &A,
29        lagrange_kernel_rand_elements: LagrangeKernelRandElements<E>,
30        lagrange_composition_coefficients: LagrangeConstraintsCompositionCoefficients<E>,
31    ) -> Self
32    where
33        E: FieldElement<BaseField = A::BaseField>,
34    {
35        Self {
36            lagrange_kernel_constraints: air
37                .get_lagrange_kernel_constraints(
38                    lagrange_composition_coefficients,
39                    &lagrange_kernel_rand_elements,
40                )
41                .expect("expected Lagrange kernel constraints to be present"),
42            rand_elements: lagrange_kernel_rand_elements,
43        }
44    }
45
46    /// Evaluates the transition and boundary constraints. Specifically, the constraint evaluations
47    /// are divided by their corresponding divisors, and the resulting terms are linearly combined
48    /// using the composition coefficients.
49    ///
50    /// Writes the evaluations in `combined_evaluations_acc` at the corresponding (constraint
51    /// evaluation) domain index.
52    pub fn evaluate_constraints<T>(
53        &self,
54        trace: &T,
55        domain: &StarkDomain<E::BaseField>,
56        combined_evaluations_acc: &mut [E],
57    ) where
58        T: TraceLde<E>,
59    {
60        let lde_shift = domain.ce_to_lde_blowup().trailing_zeros();
61        let trans_constraints_divisors = LagrangeKernelTransitionConstraintsDivisor::new(
62            self.lagrange_kernel_constraints.transition.num_constraints(),
63            domain,
64        );
65        let boundary_divisors_inv = self.compute_boundary_divisors_inv(domain);
66
67        let mut frame = LagrangeKernelEvaluationFrame::new_empty();
68
69        for step in 0..domain.ce_domain_size() {
70            // compute Lagrange kernel frame
71            trace.read_lagrange_kernel_frame_into(
72                step << lde_shift,
73                self.lagrange_kernel_constraints.lagrange_kernel_col_idx,
74                &mut frame,
75            );
76
77            // Compute the combined transition and boundary constraints evaluations for this row
78            let combined_evaluations = {
79                let mut combined_evaluations = E::ZERO;
80
81                // combine transition constraints
82                for trans_constraint_idx in
83                    0..self.lagrange_kernel_constraints.transition.num_constraints()
84                {
85                    let numerator = self
86                        .lagrange_kernel_constraints
87                        .transition
88                        .evaluate_ith_numerator(&frame, &self.rand_elements, trans_constraint_idx);
89                    let inv_divisor = trans_constraints_divisors
90                        .get_inverse_divisor_eval(trans_constraint_idx, step);
91
92                    combined_evaluations += numerator * inv_divisor;
93                }
94
95                // combine boundary constraints
96                {
97                    let boundary_numerator =
98                        self.lagrange_kernel_constraints.boundary.evaluate_numerator_at(&frame);
99
100                    combined_evaluations += boundary_numerator * boundary_divisors_inv[step];
101                }
102
103                combined_evaluations
104            };
105
106            combined_evaluations_acc[step] += combined_evaluations;
107        }
108    }
109
110    // HELPERS
111    // ---------------------------------------------------------------------------------------------
112
113    /// Computes the inverse boundary divisor at every point of the constraint evaluation domain.
114    /// That is, returns a vector of the form `[1 / div_0, ..., 1 / div_n]`, where `div_i` is the
115    /// divisor for the Lagrange kernel boundary constraint at the i'th row of the constraint
116    /// evaluation domain.
117    fn compute_boundary_divisors_inv(&self, domain: &StarkDomain<E::BaseField>) -> Vec<E> {
118        let mut boundary_denominator_evals = Vec::with_capacity(domain.ce_domain_size());
119        for step in 0..domain.ce_domain_size() {
120            let domain_point = domain.get_ce_x_at(step);
121            let boundary_denominator = self
122                .lagrange_kernel_constraints
123                .boundary
124                .evaluate_denominator_at(domain_point.into());
125            boundary_denominator_evals.push(boundary_denominator);
126        }
127
128        batch_inversion(&boundary_denominator_evals)
129    }
130}

132/// Holds all the transition constraint inverse divisor evaluations over the constraint evaluation
133/// domain.
134///
135/// [`LagrangeKernelTransitionConstraintsDivisor`] takes advantage of some structure in the
136/// divisors' evaluations. Recall that the divisor for the i'th transition constraint is `x^(2^i) -
137/// 1`. When substituting `x` for each value of the constraint evaluation domain, for constraints
138/// `i>0`, the divisor evaluations "wrap-around" such that some values repeat. For example,
139///
140/// i=0: no repetitions
141/// i=1: the first half of the buffer is equal to the second half
142/// i=2: each 1/4th of the buffer are equal
143/// i=3: each 1/8th of the buffer are equal
144/// ...
145/// Therefore, we only compute the non-repeating section of the buffer in each iteration, and index
146/// into it accordingly.
147struct LagrangeKernelTransitionConstraintsDivisor<E: FieldElement> {
148    divisor_evals_inv: Vec<E>,
149
150    // Precompute the indices into `divisors_evals_inv` of the slices that correspond to each
151    // transition constraint.
152    //
153    // For example, for a CE domain size `n=8`, `slice_indices_precomputes = [0, 8, 12, 14]`, such
154    // that transition constraint `idx` owns the range:
155    // idx=0: [0, 8)
156    // idx=1: [8, 12)
157    // idx=2: [12, 14)
158    slice_indices_precomputes: Vec<usize>,
159}
160
161impl<E: FieldElement> LagrangeKernelTransitionConstraintsDivisor<E> {
162    pub fn new(
163        num_lagrange_transition_constraints: usize,
164        domain: &StarkDomain<E::BaseField>,
165    ) -> Self {
166        let divisor_evals_inv = {
167            let divisor_evaluator = TransitionDivisorEvaluator::<E>::new(
168                num_lagrange_transition_constraints,
169                domain.offset(),
170            );
171
172            // The number of divisor evaluations is
173            // `ce_domain_size + ce_domain_size/2 + ce_domain_size/4 + ... + ce_domain_size/(log(ce_domain_size)-1)`,
174            // which is slightly smaller than `ce_domain_size * 2`
175            let mut divisor_evals: Vec<E> = Vec::with_capacity(domain.ce_domain_size() * 2);
176
177            for trans_constraint_idx in 0..num_lagrange_transition_constraints {
178                let num_non_repeating_denoms =
179                    domain.ce_domain_size() / 2_usize.pow(trans_constraint_idx as u32);
180
181                for step in 0..num_non_repeating_denoms {
182                    let divisor_eval =
183                        divisor_evaluator.evaluate_ith_divisor(trans_constraint_idx, domain, step);
184
185                    divisor_evals.push(divisor_eval.into());
186                }
187            }
188
189            batch_inversion(&divisor_evals)
190        };
191
192        let slice_indices_precomputes = {
193            let num_indices = num_lagrange_transition_constraints + 1;
194            let mut slice_indices_precomputes = Vec::with_capacity(num_indices);
195
196            slice_indices_precomputes.push(0);
197
198            let mut current_slice_len = domain.ce_domain_size();
199            for i in 1..num_indices {
200                let next_precompute_index = slice_indices_precomputes[i - 1] + current_slice_len;
201                slice_indices_precomputes.push(next_precompute_index);
202
203                current_slice_len /= 2;
204            }
205
206            slice_indices_precomputes
207        };
208
209        Self {
210            divisor_evals_inv,
211            slice_indices_precomputes,
212        }
213    }
214
215    /// Returns the evaluation `1 / divisor`, where `divisor` is the divisor for the given
216    /// transition constraint, at the given row of the constraint evaluation domain
217    pub fn get_inverse_divisor_eval(&self, trans_constraint_idx: usize, row_idx: usize) -> E {
218        let inv_divisors_slice_for_constraint =
219            self.get_transition_constraint_slice(trans_constraint_idx);
220
221        inv_divisors_slice_for_constraint[row_idx % inv_divisors_slice_for_constraint.len()]
222    }
223
224    // HELPERS
225    // ---------------------------------------------------------------------------------------------
226
227    /// Returns a slice containing all the inverse divisor evaluations for the given transition
228    /// constraint.
229    fn get_transition_constraint_slice(&self, trans_constraint_idx: usize) -> &[E] {
230        let start = self.slice_indices_precomputes[trans_constraint_idx];
231        let end = self.slice_indices_precomputes[trans_constraint_idx + 1];
232
233        &self.divisor_evals_inv[start..end]
234    }
235}

237/// Encapsulates the efficient evaluation of the Lagrange kernel transition constraints divisors.
238///
239/// `s` stands for the domain offset (i.e. coset shift element). The key concept in this
240/// optimization is to realize that the computation of the first transition constraint divisor can
241/// be reused for all the other divisors (call the evaluations `d`).
242///
243/// Specifically, each subsequent transition constraint divisor evaluation is equivalent to
244/// multiplying an element `d` by a fixed number. For example, the multiplier for the transition
245/// constraints are:
246///
247/// - transition constraint 1's multiplier: s
248/// - transition constraint 2's multiplier: s^3
249/// - transition constraint 3's multiplier: s^7
250/// - transition constraint 4's multiplier: s^15
251/// - ...
252///
253/// This is what `s_precomputes` stores.
254///
255/// Finally, recall that the ith Lagrange kernel divisor is `x^(2^i) - 1`.
256/// [`TransitionDivisorEvaluator`] is only concerned with values of `x` in the constraint evaluation
257/// domain, where the j'th element is `s * g^j`, where `g` is the group generator. To understand the
258/// implementation of [`Self::evaluate_ith_divisor`], plug in `x = s * g^j` into `x^(2^i) - 1`.
259pub struct TransitionDivisorEvaluator<E: FieldElement> {
260    s_precomputes: Vec<E::BaseField>,
261}
262
263impl<E: FieldElement> TransitionDivisorEvaluator<E> {
264    /// Constructs a new [`TransitionDivisorEvaluator`]
265    pub fn new(num_lagrange_transition_constraints: usize, domain_offset: E::BaseField) -> Self {
266        let s_precomputes = {
267            // s_precomputes = [1, s, s^3, s^7, s^15, ...] (where s = domain_offset)
268            let mut s_precomputes = Vec::with_capacity(num_lagrange_transition_constraints);
269
270            let mut s_exp = E::BaseField::ONE;
271            for _ in 0..num_lagrange_transition_constraints {
272                s_precomputes.push(s_exp);
273                s_exp = s_exp * s_exp * domain_offset;
274            }
275
276            s_precomputes
277        };
278
279        Self { s_precomputes }
280    }
281
282    /// Evaluates the divisor of the `trans_constraint_idx`'th transition constraint. See
283    /// [`TransitionDivisorEvaluator`] for a more in-depth description of the algorithm.
284    pub fn evaluate_ith_divisor(
285        &self,
286        trans_constraint_idx: usize,
287        domain: &StarkDomain<E::BaseField>,
288        ce_domain_step: usize,
289    ) -> E::BaseField {
290        let domain_idx = ((1 << trans_constraint_idx) * ce_domain_step) % domain.ce_domain_size();
291
292        self.s_precomputes[trans_constraint_idx] * domain.get_ce_x_at(domain_idx)
293            - E::BaseField::ONE
294    }
