19// DEEP COMPOSITION POLYNOMIAL
20// ================================================================================================
21pub struct DeepCompositionPoly<E: FieldElement> {
22    coefficients: Vec<E>,
23    cc: DeepCompositionCoefficients<E>,
24    z: E,
25}
26
27impl<E: FieldElement> DeepCompositionPoly<E> {
28    // CONSTRUCTOR
29    // --------------------------------------------------------------------------------------------
30    /// Returns a new DEEP composition polynomial. Initially, this polynomial will be empty, and
31    /// the intent is to populate the coefficients via add_trace_polys() and add_constraint_polys()
32    /// methods.
33    pub fn new(z: E, cc: DeepCompositionCoefficients<E>) -> Self {
34        DeepCompositionPoly { coefficients: vec![], cc, z }
35    }
36
37    // ACCESSORS
38    // --------------------------------------------------------------------------------------------
39
40    /// Returns the size of the DEEP composition polynomial.
41    pub fn poly_size(&self) -> usize {
42        self.coefficients.len()
43    }
44
45    /// Returns the degree of the composition polynomial.
46    pub fn degree(&self) -> usize {
47        polynom::degree_of(&self.coefficients)
48    }
49
50    // TRACE POLYNOMIAL COMPOSITION
51    // --------------------------------------------------------------------------------------------
52    /// Combines all trace polynomials into a single polynomial and saves the result into
53    /// the DEEP composition polynomial. The combination is done as follows:
54    ///
55    /// - Compute polynomials T'_i(x) = (T_i(x) - T_i(z)) / (x - z) and
56    ///   T''_i(x) = (T_i(x) - T_i(z * g)) / (x - z * g) for all i, where T_i(x) is a trace
57    ///   polynomial for column i.
58    /// - Then, combine together all T'_i(x) and T''_i(x) polynomials using a random linear
59    ///   combination as T(x) = sum((T'_i(x) + T''_i(x)) * cc_i) for all i, where cc_i is
60    ///   the coefficient for the random linear combination drawn from the public coin.
61    /// - If a Lagrange kernel is present, combine one additional term defined as
62    ///   (T_l(x) - p_S(x)) / Z_S(x), where:
63    ///
64    /// 1. $T_l(x) is the evaluation of the Lagrange trace polynomial at $x$.
65    /// 2. $S$ is the set of opening points for the Lagrange kernel i.e.,
66    ///    $S := {z, z.g, z.g^2, ..., z.g^{2^{log_2(\nu) - 1}}}$.
67    /// 3. $p_S(X)$ is the polynomial of minimal degree interpolating the set
68    ///    ${(a, T_l(a)): a \in S}$.
69    /// 4. $Z_S(X)$ is the polynomial of minimal degree vanishing over the set $S$.
70    ///
71    /// Note that evaluations of T_i(z) and T_i(z * g) are passed in via the `ood_trace_state`
72    /// parameter.
73    /// If a Lagrange kernel is present, the evaluations of $T_l$ over the set $S$ are provided
74    /// separately via `ood_trace_state`.
75    pub fn add_trace_polys(
76        &mut self,
77        trace_polys: TracePolyTable<E>,
78        ood_trace_states: TraceOodFrame<E>,
79    ) {
80        assert!(self.coefficients.is_empty());
81
82        // compute a second out-of-domain point offset from z by exactly trace generator; this
83        // point defines the "next" computation state in relation to point z
84        let trace_length = trace_polys.poly_size();
85        let g = E::from(E::BaseField::get_root_of_unity(trace_length.ilog2()));
86        let next_z = self.z * g;
87
88        // combine trace polynomials into 2 composition polynomials T'(x) and T''(x)
89        let mut t1_composition = vec![E::ZERO; trace_length];
90        let mut t2_composition = vec![E::ZERO; trace_length];
91
92        // index of a trace polynomial; we declare it here so that we can maintain index continuity
93        // across all trace segments
94        let mut i = 0;
95
96        // --- merge polynomials of the main trace segment ----------------------------------------
97        for poly in trace_polys.main_trace_polys() {
98            // compute T'(x) = T(x) - T(z), multiply it by a pseudo-random coefficient,
99            // and add the result into composition polynomial
100            acc_trace_poly::<E::BaseField, E>(
101                &mut t1_composition,
102                poly,
103                ood_trace_states.current_row()[i],
104                self.cc.trace[i],
105            );
106
107            // compute T''(x) = T(x) - T(z * g), multiply it by a pseudo-random coefficient,
108            // and add the result into composition polynomial
109            acc_trace_poly::<E::BaseField, E>(
110                &mut t2_composition,
111                poly,
112                ood_trace_states.next_row()[i],
113                self.cc.trace[i],
114            );
115
116            i += 1;
117        }
118
119        // --- merge polynomials of the auxiliary trace segment ----------------------------------
120        for poly in trace_polys.aux_trace_polys() {
121            // compute T'(x) = T(x) - T(z), multiply it by a pseudo-random coefficient,
122            // and add the result into composition polynomial
123            acc_trace_poly::<E, E>(
124                &mut t1_composition,
125                poly,
126                ood_trace_states.current_row()[i],
127                self.cc.trace[i],
128            );
129
130            // compute T''(x) = T(x) - T(z * g), multiply it by a pseudo-random coefficient,
131            // and add the result into composition polynomial
132            acc_trace_poly::<E, E>(
133                &mut t2_composition,
134                poly,
135                ood_trace_states.next_row()[i],
136                self.cc.trace[i],
137            );
138
139            i += 1;
140        }
141
142        // divide the composition polynomials by (x - z) and (x - z * g), respectively,
143        // and add the resulting polynomials together; the output of this step
144        // is a single trace polynomial T(x) and deg(T(x)) = trace_length - 2.
145        let mut trace_poly =
146            merge_trace_compositions(vec![t1_composition, t2_composition], vec![self.z, next_z]);
147
148        // finally compose the final term associated to the Lagrange kernel trace polynomial if
149        // there is one present.
150        // TODO: Investigate using FFT to speed up this block (see #281).
151        if let Some(poly) = trace_polys.lagrange_kernel_poly() {
152            let ood_eval_frame = ood_trace_states.lagrange_kernel_frame().expect(
153                "should contain OOD values for Lagrange kernel trace polynomial if we are here",
154            );
155
156            let log_trace_len = poly.len().ilog2();
157            let g = E::from(E::BaseField::get_root_of_unity(log_trace_len));
158            let mut xs = Vec::with_capacity(log_trace_len as usize + 1);
159
160            // push z
161            xs.push(self.z);
162
163            // compute the values (z * g), (z * g^2), (z * g^4), ..., (z * g^(2^(v-1)))
164            let mut g_exp = g;
165            for _ in 0..log_trace_len {
166                let x = g_exp * self.z;
167                xs.push(x);
168                g_exp *= g_exp;
169            }
170
171            // compute the numerator
172            let p_s = polynom::interpolate(&xs, ood_eval_frame.inner(), true);
173            let mut numerator = polynom::sub(poly, &p_s);
174
175            // divide by the zero polynomial of the set S
176            syn_div_roots_in_place(&mut numerator, &xs);
177
178            // multiply by constraint composition randomness
179            let quotient = numerator;
180            let scaled_with_randomness =
181                polynom::mul_by_scalar(&quotient, self.cc.lagrange.unwrap());
182
183            trace_poly = polynom::add(&scaled_with_randomness, &trace_poly);
184        };
185
186        // set the coefficients of the DEEP composition polynomial
187        self.coefficients = trace_poly;
188        assert_eq!(self.poly_size() - 2, self.degree());
189    }
190
191    // CONSTRAINT POLYNOMIAL COMPOSITION
192    // --------------------------------------------------------------------------------------------
193    /// Divides out OOD point z from the constraint composition polynomial and saves the result
194    /// into the DEEP composition polynomial. This method is intended to be called only after the
195    /// add_trace_polys() method has been executed. The composition is done as follows:
196    ///
197    /// - For each H_i(x), compute H'_i(x) = (H_i(x) - H(z)) / (x - z), where H_i(x) is the
198    ///   ith composition polynomial column.
199    /// - Then, combine all H_i(x) polynomials together by computing H(x) = sum(H_i(x) * cc_i) for
200    ///   all i, where cc_i is the coefficient for the random linear combination drawn from the
201    ///   public coin.
202    ///
203    /// Note that evaluations of H_i(x) at z are passed in via the `ood_evaluations` parameter.
204    pub fn add_composition_poly(
205        &mut self,
206        composition_poly: CompositionPoly<E>,
207        ood_evaluations: Vec<E>,
208    ) {
209        assert!(!self.coefficients.is_empty());
210
211        let z = self.z;
212
213        let mut column_polys = composition_poly.into_columns();
214
215        // Divide out the OOD point z from column polynomials
216        iter_mut!(column_polys).zip(ood_evaluations).for_each(|(poly, value_at_z)| {
217            // compute H'_i(x) = (H_i(x) - H_i(z)) / (x - z)
218            poly[0] -= value_at_z;
219            polynom::syn_div_in_place(poly, 1, z);
220        });
221
222        // add H'_i(x) * cc_i for all i into the DEEP composition polynomial
223        for (i, poly) in column_polys.into_iter().enumerate() {
224            mul_acc::<E, E>(&mut self.coefficients, &poly, self.cc.constraints[i]);
225        }
226        assert_eq!(self.poly_size() - 2, self.degree());
227    }
228
229    // LOW-DEGREE EXTENSION
230    // --------------------------------------------------------------------------------------------
231    /// Evaluates DEEP composition polynomial over the specified LDE domain and returns the result.
232    pub fn evaluate(self, domain: &StarkDomain<E::BaseField>) -> Vec<E> {
233        fft::evaluate_poly_with_offset(
234            &self.coefficients,
235            domain.trace_twiddles(),
236            domain.offset(),
237            domain.trace_to_lde_blowup(),
238        )
239    }
240}

245/// Divides each polynomial in the list by the corresponding divisor, and computes the
246/// coefficient-wise sum of all resulting polynomials.
247fn merge_trace_compositions<E: FieldElement>(mut polys: Vec<Vec<E>>, divisors: Vec<E>) -> Vec<E> {
248    // divide all polynomials by their corresponding divisor
249    iter_mut!(polys).zip(divisors).for_each(|(poly, divisor)| {
250        polynom::syn_div_in_place(poly, 1, divisor);
251    });
252
253    // add all polynomials together into a single polynomial
254    let mut result = polys.remove(0);
255    for poly in polys.iter() {
256        add_in_place(&mut result, poly);
257    }
258
259    result
260}
261
262/// Computes (P(x) - value) * k and saves the result into the accumulator.
263fn acc_trace_poly<F, E>(accumulator: &mut [E], poly: &[F], value: E, k: E)
264where
265    F: FieldElement,
266    E: FieldElement<BaseField = F::BaseField> + ExtensionOf<F>,
267{
268    mul_acc(accumulator, poly, k);
269    let adjusted_tz = value * k;
270    accumulator[0] -= adjusted_tz;
271}
